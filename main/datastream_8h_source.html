874 template <typename T> void swapIfNeeded(T &val) const {
875 if constexpr (sizeof(T) == 1) return;
876 if (_byteOrder == nativeByteOrder()) return;
877 uint8_t *p = reinterpret_cast<uint8_t *>(&val);
878 if constexpr (sizeof(T) == 2) {
879 std::swap(p[0], p[1]);
880 } else if constexpr (sizeof(T) == 4) {
881 std::swap(p[0], p[3]);
882 std::swap(p[1], p[2]);
883 } else if constexpr (sizeof(T) == 8) {
884 std::swap(p[0], p[7]);
885 std::swap(p[1], p[6]);
886 std::swap(p[2], p[5]);
887 std::swap(p[3], p[4]);
888 }
889 }

896 static const uint16_t val = 1;
897 return (*reinterpret_cast<const uint8_t *>(&val) == 1) ? LittleEndian : BigEndian;

922// ============================================================================
923// Geometry template operators
924// ============================================================================
925//
926// These share a single tag per kind (TypeSize2D, TypeRect, TypePoint,
927// TypeRational) and rely on the inner values being written/read via their
928// own primitive operators, which means the inner values carry their own
929// type tags. This lets one Size2DTemplate<T> template cover uint32_t,
930// int32_t, float, etc. transparently, and a reader automatically catches
931// element-type mismatches via the existing tag-validation path.

940template <typename T> DataStream &operator<<(DataStream &stream, const Size2DTemplate<T> &sz) {
941 stream.writeTag(DataStream::TypeSize2D);
942 stream << sz.width() << sz.height();
943 return stream;
944}

1246template <typename T> DataStream &operator>>(DataStream &stream, List<T> &list) {
1247 list.clear();
1248 if (!stream.readTag(DataStream::TypeList)) return stream;
1249 uint32_t count = 0;
1250 stream >> count;
1251 if (stream.status() != DataStream::Ok) return stream;
1252 if (count > detail::DataStreamMaxContainerCount) {
1253 stream.setError(DataStream::ReadCorruptData, String("List element count exceeds sanity limit"));
1254 return stream;
1255 }
1256 list.reserve(count);
1257 for (uint32_t i = 0; i < count; ++i) {
1258 T item{};
1259 stream >> item;
1260 if (stream.status() != DataStream::Ok) return stream;
1261 list.pushToBack(std::move(item));
1262 }
1263 return stream;
1264}
