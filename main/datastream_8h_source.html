792 template <typename T>
793 void swapIfNeeded(T &val) const {
794 if constexpr (sizeof(T) == 1) return;
795 if(_byteOrder == nativeByteOrder()) return;
796 uint8_t *p = reinterpret_cast<uint8_t *>(&val);
797 if constexpr (sizeof(T) == 2) {
798 std::swap(p[0], p[1]);
799 } else if constexpr (sizeof(T) == 4) {
800 std::swap(p[0], p[3]);
801 std::swap(p[1], p[2]);
802 } else if constexpr (sizeof(T) == 8) {
803 std::swap(p[0], p[7]);
804 std::swap(p[1], p[6]);
805 std::swap(p[2], p[5]);
806 std::swap(p[3], p[4]);
807 }
808 }

814 static ByteOrder nativeByteOrder() {
815 static const uint16_t val = 1;
816 return (*reinterpret_cast<const uint8_t *>(&val) == 1)
817 ? LittleEndian : BigEndian;
818 }

842// ============================================================================
843// Geometry template operators
844// ============================================================================
845//
846// These share a single tag per kind (TypeSize2D, TypeRect, TypePoint,
847// TypeRational) and rely on the inner values being written/read via their
848// own primitive operators, which means the inner values carry their own
849// type tags. This lets one Size2DTemplate<T> template cover uint32_t,
850// int32_t, float, etc. transparently, and a reader automatically catches
851// element-type mismatches via the existing tag-validation path.

860template <typename T>
861DataStream &operator<<(DataStream &stream, const Size2DTemplate<T> &sz) {
862 stream.writeTag(DataStream::TypeSize2D);
863 stream << sz.width() << sz.height();
864 return stream;
865}
