693 template <typename T>
694 void swapIfNeeded(T &val) const {
695 if constexpr (sizeof(T) == 1) return;
696 if(_byteOrder == nativeByteOrder()) return;
697 uint8_t *p = reinterpret_cast<uint8_t *>(&val);
698 if constexpr (sizeof(T) == 2) {
699 std::swap(p[0], p[1]);
700 } else if constexpr (sizeof(T) == 4) {
701 std::swap(p[0], p[3]);
702 std::swap(p[1], p[2]);
703 } else if constexpr (sizeof(T) == 8) {
704 std::swap(p[0], p[7]);
705 std::swap(p[1], p[6]);
706 std::swap(p[2], p[5]);
707 std::swap(p[3], p[4]);
708 }
709 }

743// ============================================================================
744// Geometry template operators
745// ============================================================================
746//
747// These share a single tag per kind (TypeSize2D, TypeRect, TypePoint,
748// TypeRational) and rely on the inner values being written/read via their
749// own primitive operators, which means the inner values carry their own
750// type tags. This lets one Size2DTemplate<T> template cover uint32_t,
751// int32_t, float, etc. transparently, and a reader automatically catches
752// element-type mismatches via the existing tag-validation path.
753
761template <typename T>
762DataStream &operator<<(DataStream &stream, const Size2DTemplate<T> &sz) {
763 stream.writeTag(DataStream::TypeSize2D);
764 stream << sz.width() << sz.height();
765 return stream;
766}

775template <typename T>
776DataStream &operator>>(DataStream &stream, Size2DTemplate<T> &sz) {
777 if(!stream.readTag(DataStream::TypeSize2D)) { sz = Size2DTemplate<T>(); return stream; }
778 T w{}, h{};
779 stream >> w >> h;
780 if(stream.status() != DataStream::Ok) { sz = Size2DTemplate<T>(); return stream; }
781 sz = Size2DTemplate<T>(w, h);
782 return stream;
783}

877template <typename T, size_t N>
878DataStream &operator>>(DataStream &stream, Point<T, N> &point) {
879 if(!stream.readTag(DataStream::TypePoint)) { point = Point<T, N>(); return stream; }
880 uint32_t dims = 0;
881 stream >> dims;
882 if(stream.status() != DataStream::Ok) { point = Point<T, N>(); return stream; }
883 if(dims != N) {
884 stream.setError(DataStream::ReadCorruptData,
885 String::sprintf(

887 N, static_cast<unsigned>(dims)));

892 for(size_t i = 0; i < N; ++i) {

894 stream >> val;
895 if(stream.status() != DataStream::Ok) { point = Point<T, N>(); return stream; }
896 arr[i] = val;
897 }
898 point = Point<T, N>(arr);

960template <typename T>
961DataStream &operator>>(DataStream &stream, List<T> &list) {
962 list.clear();
963 if(!stream.readTag(DataStream::TypeList)) return stream;
964 uint32_t count = 0;
965 stream >> count;
966 if(stream.status() != DataStream::Ok) return stream;
967 if(count > detail::DataStreamMaxContainerCount) {
968 stream.setError(DataStream::ReadCorruptData,
969 String("List element count exceeds sanity limit"));
970 return stream;

972 list.reserve(count);

974 T item{};

977 list.pushToBack(std::move(item));
