760 template <typename T>
761 void swapIfNeeded(T &val) const {
762 if constexpr (sizeof(T) == 1) return;
763 if(_byteOrder == nativeByteOrder()) return;
764 uint8_t *p = reinterpret_cast<uint8_t *>(&val);
765 if constexpr (sizeof(T) == 2) {
766 std::swap(p[0], p[1]);
767 } else if constexpr (sizeof(T) == 4) {
768 std::swap(p[0], p[3]);
769 std::swap(p[1], p[2]);
770 } else if constexpr (sizeof(T) == 8) {
771 std::swap(p[0], p[7]);
772 std::swap(p[1], p[6]);
773 std::swap(p[2], p[5]);
774 std::swap(p[3], p[4]);
775 }
776 }
777
782 static ByteOrder nativeByteOrder() {
783 static const uint16_t val = 1;
784 return (*reinterpret_cast<const uint8_t *>(&val) == 1)
785 ? LittleEndian : BigEndian;
786 }

810// ============================================================================
811// Geometry template operators
812// ============================================================================
813//
814// These share a single tag per kind (TypeSize2D, TypeRect, TypePoint,
815// TypeRational) and rely on the inner values being written/read via their
816// own primitive operators, which means the inner values carry their own
817// type tags. This lets one Size2DTemplate<T> template cover uint32_t,
818// int32_t, float, etc. transparently, and a reader automatically catches
819// element-type mismatches via the existing tag-validation path.
820
828template <typename T>
829DataStream &operator<<(DataStream &stream, const Size2DTemplate<T> &sz) {
830 stream.writeTag(DataStream::TypeSize2D);
831 stream << sz.width() << sz.height();
832 return stream;
833}

845 T w{}, h{};
846 stream >> w >> h;
847 if(stream.status() != DataStream::Ok) { sz = Size2DTemplate<T>(); return stream; }
848 sz = Size2DTemplate<T>(w, h);

944template <typename T, size_t N>
945DataStream &operator>>(DataStream &stream, Point<T, N> &point) {
946 if(!stream.readTag(DataStream::TypePoint)) { point = Point<T, N>(); return stream; }
947 uint32_t dims = 0;
948 stream >> dims;
949 if(stream.status() != DataStream::Ok) { point = Point<T, N>(); return stream; }
950 if(dims != N) {
951 stream.setError(DataStream::ReadCorruptData,
952 String::sprintf(
953 "Point dimension mismatch: expected %zu, got %u",
954 N, static_cast<unsigned>(dims)));
955 point = Point<T, N>();
956 return stream;
957 }
958 Array<T, N> arr;
959 for(size_t i = 0; i < N; ++i) {
960 T val{};
961 stream >> val;
962 if(stream.status() != DataStream::Ok) { point = Point<T, N>(); return stream; }
963 arr[i] = val;
964 }
965 point = Point<T, N>(arr);
966 return stream;
967}

1112 list.clear();
1113 if(!stream.readTag(DataStream::TypeList)) return stream;
1114 uint32_t count = 0;
1115 stream >> count;
1116 if(stream.status() != DataStream::Ok) return stream;
1117 if(count > detail::DataStreamMaxContainerCount) {
1118 stream.setError(DataStream::ReadCorruptData,
1119 String("List element count exceeds sanity limit"));
1120 return stream;

1123 for(uint32_t i = 0; i < count; ++i) {

1126 if(stream.status() != DataStream::Ok) return stream;
1127 list.pushToBack(std::move(item));
1128 }
1129 return stream;
