729 template <typename T>
730 void swapIfNeeded(T &val) const {
731 if constexpr (sizeof(T) == 1) return;
732 if(_byteOrder == nativeByteOrder()) return;
733 uint8_t *p = reinterpret_cast<uint8_t *>(&val);
734 if constexpr (sizeof(T) == 2) {
735 std::swap(p[0], p[1]);
736 } else if constexpr (sizeof(T) == 4) {
737 std::swap(p[0], p[3]);
738 std::swap(p[1], p[2]);
739 } else if constexpr (sizeof(T) == 8) {
740 std::swap(p[0], p[7]);
741 std::swap(p[1], p[6]);
742 std::swap(p[2], p[5]);
743 std::swap(p[3], p[4]);
744 }
745 }

779// ============================================================================
780// Geometry template operators
781// ============================================================================
782//
783// These share a single tag per kind (TypeSize2D, TypeRect, TypePoint,
784// TypeRational) and rely on the inner values being written/read via their
785// own primitive operators, which means the inner values carry their own
786// type tags. This lets one Size2DTemplate<T> template cover uint32_t,
787// int32_t, float, etc. transparently, and a reader automatically catches
788// element-type mismatches via the existing tag-validation path.
789
797template <typename T>
798DataStream &operator<<(DataStream &stream, const Size2DTemplate<T> &sz) {
799 stream.writeTag(DataStream::TypeSize2D);
800 stream << sz.width() << sz.height();
801 return stream;
802}
