187 Duration frameDuration() const {

189 // Period in ns = denom * 1e9 / num. Use 64-bit
190 // math so 1001 * 1e9 doesn't overflow.
191 int64_t num = static_cast<int64_t>(_fps.numerator());
192 int64_t den = static_cast<int64_t>(_fps.denominator());
193 int64_t ns = (den * INT64_C(1'000'000'000) + num / 2) / num;
194 return Duration::fromNanoseconds(ns);

248 int64_t cumulativeTicks(int64_t tickRate, int64_t frameIndex) const {
249 if (!isValid() || tickRate <= 0 || frameIndex < 0) return 0;
250 const int64_t num = static_cast<int64_t>(_fps.numerator());
251 const int64_t den = static_cast<int64_t>(_fps.denominator());
252 return (frameIndex * tickRate * den) / num;

278 size_t samplesPerFrame(int64_t sampleRate, int64_t frameIndex) const {
279 if (!isValid() || sampleRate <= 0 || frameIndex < 0) return 0;
280 const int64_t cumNow = cumulativeTicks(sampleRate, frameIndex);
281 const int64_t cumNext = cumulativeTicks(sampleRate, frameIndex + 1);
