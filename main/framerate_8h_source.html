163 Duration frameDuration() const {

165 // Period in ns = denom * 1e9 / num. Use 64-bit
166 // math so 1001 * 1e9 doesn't overflow.
167 int64_t num = static_cast<int64_t>(_fps.numerator());
168 int64_t den = static_cast<int64_t>(_fps.denominator());
169 int64_t ns = (den * INT64_C(1'000'000'000) + num / 2) / num;
170 return Duration::fromNanoseconds(ns);

224 int64_t cumulativeTicks(int64_t tickRate, int64_t frameIndex) const {
225 if(!isValid() || tickRate <= 0 || frameIndex < 0) return 0;
226 const int64_t num = static_cast<int64_t>(_fps.numerator());
227 const int64_t den = static_cast<int64_t>(_fps.denominator());

254 size_t samplesPerFrame(int64_t sampleRate, int64_t frameIndex) const {
