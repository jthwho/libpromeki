204 size_t samplesPerFrame(int64_t sampleRate, int64_t frameIndex) const {
205 if(!isValid() || sampleRate <= 0 || frameIndex < 0) return 0;
206 const int64_t num = static_cast<int64_t>(_fps.numerator());
207 const int64_t den = static_cast<int64_t>(_fps.denominator());
208 // Cumulative samples through end of frame N = N * sr * den / num.
209 // Cast to __int128 would be overkill — for any plausible
210 // sample rate (<= 768 kHz) and fps numerator (<= 120000),
211 // sr * den fits in 60 bits and a multiply by frameIndex up
212 // to 2^31 still fits in int64_t.
213 const int64_t cumNext = ((frameIndex + 1) * sampleRate * den) / num;
214 const int64_t cumNow = (frameIndex * sampleRate * den) / num;
215 return static_cast<size_t>(cumNext - cumNow);
216 }
