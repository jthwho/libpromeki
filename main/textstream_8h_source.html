471// These templated operators live here (rather than alongside their
472// respective class definitions) so the @ref TextStream definition is
473// fully visible at template-parse time. The corresponding class
474// header (atomic.h, pair.h, span.h) is forward-declared below — actual
475// usage requires both the class header and this header to be included.
476// ============================================================================

479template <typename A, typename B> class Pair;

492template <typename T> inline TextStream &operator<<(TextStream &s, const Atomic<T> &a) {
493 s << a.value();
494 return s;
495}

506template <typename A, typename B> inline TextStream &operator<<(TextStream &s, const Pair<A, B> &p) {
507 s << "(" << p.first() << ", " << p.second() << ")";
508 return s;
509}
510
521template <typename T> inline TextStream &operator<<(TextStream &s, const Span<T> &span) {
522 s << "[";
523 for (size_t i = 0; i < span.size(); ++i) {
524 if (i != 0) s << ", ";
525 s << span[i];
526 }
527 s << "]";
528 return s;
