103 inline bool parseLeadingSegment(const String &key, VariantLookupSegment &out, Error *err = nullptr) {
104 if (err != nullptr) *err = Error::Ok;
105 const size_t len = key.byteCount();
106 if (len == 0) {
107 if (err != nullptr) *err = Error::ParseFailed;
108 return false;
109 }
110 const char *s = key.cstr();
111 size_t i = 0;
112 auto isNameChar = [](char c) -> bool {
113 return (c >= 'A' && c <= 'Z') || (c >= 'a' && c <= 'z') || (c >= '0' && c <= '9') || c == '_' ||
114 c == '@';
115 };
116 while (i < len && isNameChar(s[i])) ++i;
117 if (i == 0) {
118 if (err != nullptr) *err = Error::ParseFailed;
119 return false;
120 }
121 out.name = String(s, i);
122 out.hasIndex = false;
123 out.hasRest = false;
124 out.rest = String();
125 out.index = 0;
126
127 if (i < len && s[i] == '[') {
128 ++i;
129 const size_t numStart = i;
130 while (i < len && s[i] >= '0' && s[i] <= '9') ++i;
131 if (i == numStart) {
132 if (err != nullptr) *err = Error::ParseFailed;
133 return false;
134 }
135 if (i >= len || s[i] != ']') {
136 if (err != nullptr) *err = Error::ParseFailed;
137 return false;
138 }
139 char *endp = nullptr;
140 out.index = static_cast<size_t>(std::strtoull(s + numStart, &endp, 10));
141 out.hasIndex = true;
142 ++i;
143 }
144
145 if (i == len) return true;
146
147 if (s[i] != '.') {
148 if (err != nullptr) *err = Error::ParseFailed;
149 return false;
150 }
151 ++i;
152 if (i >= len) {
153 if (err != nullptr) *err = Error::ParseFailed;
154 return false;
155 }
156 out.rest = String(s + i, len - i);
157 out.hasRest = true;
158 return true;
159 }

183 inline const VariantSpec *walkSpecPath(const VariantSpec *cur, const String &path, Error *err = nullptr) {
184 if (cur == nullptr) {
185 if (err != nullptr) *err = Error::IdNotFound;
186 return nullptr;
187 }
188 if (err != nullptr) *err = Error::Ok;
189 String remaining = path;
190 while (!remaining.isEmpty() && cur != nullptr) {
191 VariantLookupSegment seg;
192 // parseLeadingSegment requires a leading name; we
193 // accept paths that start with a bracket (root is the
194 // list) by handling that case explicitly.
195 bool bracketStart = (remaining[0] == '[');
196 if (bracketStart) {
197 // Index-only step: e.g. "[3]" or "[3].rest".
198 size_t end = 1;
199 const char *p = remaining.cstr();
200 const size_t len = remaining.byteCount();
201 while (end < len && p[end] >= '0' && p[end] <= '9') ++end;
202 if (end == 1 || end >= len || p[end] != ']') {
203 if (err != nullptr) *err = Error::ParseFailed;
204 return nullptr;
205 }
206 cur = cur->elementSpec();
207 if (cur == nullptr) {
208 if (err != nullptr) *err = Error::IdNotFound;
209 return nullptr;
210 }
211 ++end; // past ']'
212 if (end == len) {
213 remaining = String();
214 } else if (p[end] == '.') {
215 remaining = remaining.mid(end + 1);
216 } else {
217 if (err != nullptr) *err = Error::ParseFailed;
218 return nullptr;
219 }
220 continue;
221 }
222 if (!parseLeadingSegment(remaining, seg, err)) return nullptr;
223 // Name step always descends via valueSpec. If the
224 // segment also has an index, descend once more via
225 // elementSpec — matches the runtime resolver's
226 // (map → list) traversal for "name[N]" segments.
227 cur = cur->valueSpec();
228 if (cur == nullptr) {
229 if (err != nullptr) *err = Error::IdNotFound;
230 return nullptr;
231 }
232 if (seg.hasIndex) {
233 cur = cur->elementSpec();
234 if (cur == nullptr) {
235 if (err != nullptr) *err = Error::IdNotFound;
236 return nullptr;
237 }
238 }
239 remaining = seg.hasRest ? seg.rest : String();
240 }
241 return cur;
242 }

946 // Resolve a "Name.rest" lookup: borrow the
947 // Variant from the user, descend via the
948 // shared tree-walker. Empty rest returns
949 // the whole Variant, with IdNotFound when

951 // (invalid Variant).

953 Error *err) -> std::optional<Variant> {
954 Variant root = get(t);
955 if (!root.isValid()) {
956 if (err != nullptr) *err = Error::IdNotFound;
957 return std::nullopt;
958 }
959 if (rest.isEmpty()) return root;
960 Error pe;
961 Variant out = promekiResolveVariantPath(root, rest, &pe);
962 if (pe.isError()) {
963 if (err != nullptr) *err = pe;
964 return std::nullopt;
965 }
966 return out;

976 if (err != nullptr) *err = Error::IdNotFound;

1818 template <typename Fn> static void forEachScalar(Fn &&fn) {
1819 const Registry &r = registry();
1820 StringList names;
1821 std::function<void(const std::function<void(const String &)> &)> inheritForEach;
1822 {
1823 ReadWriteLock::ReadLocker lock(r.lock);
1824 for (auto it = r.scalars.cbegin(); it != r.scalars.cend(); ++it) {
1825 auto n = r.names.find(it->first);
1826 if (n != r.names.end()) names.pushToBack(n->second);
1827 }
1828 inheritForEach = r.inherit.forEachScalar;
1829 }
1830 // Pull inherited names in through the fallback —
1831 // this recurses transparently because
1832 // @c VariantLookup<Base>::forEachScalar also
1833 // consults its own inherit chain.
1834 if (inheritForEach) {
1835 inheritForEach([&names](const String &n) { names.pushToBack(n); });
1836 }
1837 names = names.sort();
1838 // De-duplicate — a derived class may shadow a
1839 // base's scalar with the same name; emit only
1840 // once so dumps don't print the key twice.
1841 String prev;
1842 for (const String &name : names) {
1843 if (!prev.isEmpty() && name == prev) continue;
1844 fn(name);
1845 prev = name;
1846 }
1847 }

1915 static StringList dumpComposites(const T &instance, const String &indent = String()) {
1916 StringList out;
1917
1918 // Snapshot per-entry dump lambdas + names under
1919 // the read lock so we can drop the lock before
1920 // recursing into VariantLookup<U>::dump (which
1921 // takes its own VariantLookup<U> registry lock;
1922 // holding two locks invites hidden deadlocks as
1923 // the hierarchy grows).
1924 struct Item {
1925 String name;
1926 std::function<StringList(const T &, const String &)> dump;
1927 };
1928 List<Item> children;
1929 List<Item> indexedChildren;
1930 List<Item> databases;
1931 List<Item> variantTrees;
1932 std::function<StringList(const T &, const String &)> inheritFn;
1933 {
1934 const Registry &r = registry();
1935 ReadWriteLock::ReadLocker lock(r.lock);
1936 for (auto it = r.children.cbegin(); it != r.children.cend(); ++it) {
1937 if (!it->second.dump) continue;
1938 auto n = r.names.find(it->first);
1939 if (n == r.names.end()) continue;
1940 children.pushToBack(Item{n->second, it->second.dump});
1941 }
1942 for (auto it = r.indexedChildren.cbegin(); it != r.indexedChildren.cend(); ++it) {
1943 if (!it->second.dump) continue;
1944 auto n = r.names.find(it->first);
1945 if (n == r.names.end()) continue;
1946 indexedChildren.pushToBack(Item{n->second, it->second.dump});
1947 }
1948 for (auto it = r.databases.cbegin(); it != r.databases.cend(); ++it) {
1949 if (!it->second.dump) continue;
1950 auto n = r.names.find(it->first);
1951 if (n == r.names.end()) continue;
1952 databases.pushToBack(Item{n->second, it->second.dump});
1953 }
1954 // variantTree bindings get included with the
1955 // same dump shape as databases — one section
