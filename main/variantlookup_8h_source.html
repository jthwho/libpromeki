76inline bool parseLeadingSegment(const String &key,
77 VariantLookupSegment &out,
78 Error *err = nullptr) {
79 if(err != nullptr) *err = Error::Ok;
80 const size_t len = key.byteCount();
81 if(len == 0) {
82 if(err != nullptr) *err = Error::ParseFailed;
83 return false;
84 }
85 const char *s = key.cstr();
86 size_t i = 0;
87 auto isNameChar = [](char c) -> bool {
88 return (c >= 'A' && c <= 'Z')
89 || (c >= 'a' && c <= 'z')
90 || (c >= '0' && c <= '9')
91 || c == '_' || c == '@';
92 };
93 while(i < len && isNameChar(s[i])) ++i;
94 if(i == 0) {
95 if(err != nullptr) *err = Error::ParseFailed;
96 return false;
97 }
98 out.name = String(s, i);
99 out.hasIndex = false;
100 out.hasRest = false;
101 out.rest = String();
102 out.index = 0;
103
104 if(i < len && s[i] == '[') {
105 ++i;
106 const size_t numStart = i;
107 while(i < len && s[i] >= '0' && s[i] <= '9') ++i;
108 if(i == numStart) {
109 if(err != nullptr) *err = Error::ParseFailed;
110 return false;
111 }
112 if(i >= len || s[i] != ']') {
113 if(err != nullptr) *err = Error::ParseFailed;
114 return false;
115 }
116 char *endp = nullptr;
117 out.index = static_cast<size_t>(
118 std::strtoull(s + numStart, &endp, 10));
119 out.hasIndex = true;
120 ++i;
121 }
122
123 if(i == len) return true;
124
125 if(s[i] != '.') {
126 if(err != nullptr) *err = Error::ParseFailed;
127 return false;
128 }
129 ++i;
130 if(i >= len) {
131 if(err != nullptr) *err = Error::ParseFailed;
132 return false;
133 }
134 out.rest = String(s + i, len - i);
135 out.hasRest = true;
136 return true;
137}

641 static std::optional<Variant> resolve(const T &instance, const String &key, Error *err = nullptr) {
642 if(err != nullptr) *err = Error::Ok;
643 detail::VariantLookupSegment seg;
644 if(!detail::parseLeadingSegment(key, seg, err)) return std::nullopt;
645
646 const Registry &r = registry();
647 ReadWriteLock::ReadLocker lock(r.lock);
648 const uint64_t id = r.findId(seg.name);
649 if(id == Registry::Invalid) {
650 if(err != nullptr) *err = Error::IdNotFound;
651 return std::nullopt;
652 }
653
654 if(!seg.hasRest) {
655 if(seg.hasIndex) {
656 auto it = r.indexedScalars.find(id);
657 if(it == r.indexedScalars.end()) {
658 if(err != nullptr) *err = Error::IdNotFound;
659 return std::nullopt;
660 }
661 auto v = it->second.get(instance, seg.index);
662 if(!v.has_value()) {
663 if(err != nullptr) *err = Error::OutOfRange;
664 }
665 return v;
666 }
667 auto it = r.scalars.find(id);
668 if(it == r.scalars.end()) {
669 if(err != nullptr) *err = Error::IdNotFound;
670 return std::nullopt;
671 }
672 auto v = it->second.get(instance);
673 if(!v.has_value()) {
674 if(err != nullptr) *err = Error::IdNotFound;
675 }
676 return v;
677 }
678
679 if(seg.hasIndex) {
680 auto it = r.indexedChildren.find(id);
681 if(it == r.indexedChildren.end()) {
682 if(err != nullptr) *err = Error::IdNotFound;
683 return std::nullopt;
684 }
685 return it->second.resolve(instance, seg.index, seg.rest, err);
686 }
687
688 auto itChild = r.children.find(id);
689 if(itChild != r.children.end()) {
690 return itChild->second.resolve(instance, seg.rest, err);
691 }
692 auto itDb = r.databases.find(id);
693 if(itDb != r.databases.end()) {
694 return itDb->second.resolve(instance, seg.rest, err);
695 }
696 if(err != nullptr) *err = Error::IdNotFound;
697 return std::nullopt;
698 }

1078 template <typename Resolver>
1079 static String format(const T &instance, const String &tmpl,
1080 Resolver &&resolver, Error *err = nullptr) {
1081 if(err != nullptr) *err = Error::Ok;
1082 std::string out;
1083 out.reserve(tmpl.byteCount());
1084 const char *src = tmpl.cstr();
1085 const size_t len = tmpl.byteCount();
1086 bool sawUnresolved = false;
1087 size_t i = 0;
1088 while(i < len) {
1089 char c = src[i];
1090 if(c == '{') {
1091 if(i + 1 < len && src[i + 1] == '{') {
1092 out.push_back('{');
1093 i += 2;
1094 continue;
1095 }
1096 size_t end = i + 1;
1097 while(end < len && src[end] != '}') ++end;
1098 if(end >= len) {
1099 out.append(src + i, len - i);
1100 break;
1101 }
1102 std::string_view body(src + i + 1, end - (i + 1));
1103 size_t colon = body.find(':');

1105 ? body : body.substr(0, colon);
1106 std::string_view specView = (colon == std::string_view::npos)
1107 ? std::string_view() : body.substr(colon + 1);
1108 String keyName(keyView.data(), keyView.size());
1109 String specStr(specView.data(), specView.size());
1110 auto v = resolve(instance, keyName);
1111 if(v.has_value()) {
1112 String rendered = v->format(specStr);
1113 out.append(rendered.cstr(), rendered.byteCount());
