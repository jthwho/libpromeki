195 Error sync(unsigned int timeoutMs = 0) {
196 auto p = std::make_shared<std::promise<void>>();
197 auto f = p->get_future();
198 Command cmd;
199 cmd.cmd = CmdSync;
200 cmd.syncPromise = std::move(p);
201 _queue.push(std::move(cmd));
202 if(timeoutMs == 0) {
203 f.wait();
204 return Error::Ok;
205 }
206 if(f.wait_for(std::chrono::milliseconds(timeoutMs)) == std::future_status::ready) {
207 return Error::Ok;
208 }
209 return Error::Timeout;
210 }
