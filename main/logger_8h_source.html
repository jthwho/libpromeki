262 Error sync(unsigned int timeoutMs = 0) {
263 auto p = std::make_shared<std::promise<void>>();
264 auto f = p->get_future();
265 _queue.emplace(CmdSync{std::move(p)});
266 if(timeoutMs == 0) {
267 f.wait();
268 return Error::Ok;
269 }
270 if(f.wait_for(std::chrono::milliseconds(timeoutMs)) == std::future_status::ready) {
271 return Error::Ok;
272 }
273 return Error::Timeout;
274 }
