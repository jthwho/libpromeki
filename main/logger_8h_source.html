447 Error sync(unsigned int timeoutMs = 0) {
448 if (_terminating.value()) return Error::Ok;
449 auto p = std::make_shared<Promise<void>>();
450 Future<void> f = p->future();
451 _queue.emplace(CmdSync{std::move(p)});
452 if (timeoutMs == 0) {
453 f.waitForFinished();
454 return Error::Ok;
455 }
456 return f.waitForFinished(timeoutMs);
457 }
