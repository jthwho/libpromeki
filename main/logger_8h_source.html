289 Error sync(unsigned int timeoutMs = 0) {
290 auto p = std::make_shared<std::promise<void>>();
291 auto f = p->get_future();
292 _queue.emplace(CmdSync{std::move(p)});
293 if(timeoutMs == 0) {
294 f.wait();
295 return Error::Ok;
296 }
297 if(f.wait_for(std::chrono::milliseconds(timeoutMs)) == std::future_status::ready) {
298 return Error::Ok;
299 }
300 return Error::Timeout;
