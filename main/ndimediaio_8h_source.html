295 // ---- Source-mode state ----
296 //
297 // The capture thread loops on @c NDIlib_recv_capture_v3
298 // with a short timeout, demultiplexes the returned
299 // frames into the queues / ring below, and frees the
300 // SDK buffers via the matching recv_free_* call. All
301 // queues are mutex-protected internally; the strand
302 // drains them in @c executeCmd(MediaIOCommandRead).
303 std::thread _captureThread;
304 std::atomic<bool> _stopFlag{false};
305 std::atomic<bool> _readCancelled{false};
306 // Reader-side video queue. Capacity is small (matches
307 // V4L2's "drop oldest, count drop") because NDI pushes
308 // at the source's true frame rate and we want timing-
309 // accurate delivery rather than a long backlog.
310 static constexpr int VideoQueueDepth = 2;
311 Queue<UncompressedVideoPayload::Ptr> _videoQueue;
312 AudioBuffer _audioRing;
313 Mutex _audioMutex;
314 // Sender-anchored timeline state for the audio ring.
315 // Both fields are NDI 100ns ticks, guarded by
316 // @ref _audioMutex (the same mutex that guards the
317 // ring). The capture thread updates them on every
318 // received audio frame; the strand drains and resets
319 // them on every executeCmd(Read).
320 //
321 // @c _audioFirstSampleTicks is the timestamp of the
322 // first sample currently sitting in the ring. It's
323 // the canonical PTS for whatever the next drain emits
324 // — anchored on the sender's first-sample time, not
325 // on the most-recent NDI frame's timestamp (which
326 // would be the *last* sample's anchor and therefore
327 // lie about a coalesced multi-frame drain). Zero
328 // when no anchor is latched (ring empty + no prior
329 // frames since the last reset).
330 //
331 // @c _audioNextSampleTicks is the timestamp the next
332 // arriving sample is expected to land on, computed as
333 // @c lastFrame.timestamp + samples * 1e7 / rate.
334 // The capture thread compares each new frame's
335 // timestamp against this value to detect gaps; the
336 // gap (if positive and within sanity bounds) is
337 // bridged with @ref AudioBuffer::pushSilence so the
338 // ring's sample count and the sender's media
339 // timeline stay in sync. Zero when no prior frame
340 // has been pushed since the last reset.
341 int64_t _audioFirstSampleTicks = 0;
342 int64_t _audioNextSampleTicks = 0;
343 // Markers accumulated for the next drained payload.
344 // Each entry's @c offset is the sample index within
345 // the eventual payload (i.e. the value of
346 // @c _audioRing.available() at the moment the marker
347 // was appended); @c length is the sample count
348 // covered. The list is stamped onto the drained
349 // payload's @ref Metadata::AudioMarkers in
350 // executeCmd(Read) and then cleared.
351 AudioMarkerList _audioMarkersSinceDrain;
352 // Total samples of silence injected by the gap-fill
353 // logic since the last open. Atomic so the strand's
354 // executeCmd(Stats) can publish it without taking
355 // @ref _audioMutex. Mirrored as
356 // @ref StatsAudioSilenceFilled.
357 std::atomic<int64_t> _audioSilenceSamples{0};
358 // Total contiguous silence-fill events since the
359 // last open. Mirrored as @ref StatsAudioGapEvents.
360 std::atomic<int64_t> _audioGapEvents{0};
361 Mutex _metadataMutex;
362 Metadata _pendingMetadata;
363 bool _hasPendingMetadata = false;
364
365 // Reader configuration captured at openSource time.
366 // The bit-depth tag controls how P216 frames are tagged
367 // when emitted (see @ref NdiReceiveBitDepth).
368 int _captureTimeoutMs = 100;
369 int _bitDepthHint = 0; // 0 = Auto (16-bit).

371 // Reader-side telemetry.

374 std::atomic<int64_t> _metadataReceived{0};

377 // Source-mode clock — driven by NDI per-frame

384 // External sink-mode pacing — one PacingGate per

393 // internal clock_video / clock_audio flags remain in
394 // control.
