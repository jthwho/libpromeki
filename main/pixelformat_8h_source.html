342 // -- Full-range uncompressed YCbCr --
343 //
344 // The library-wide YCbCr naming convention is that
345 // the unsuffixed form (e.g. YUV8_422_Rec709) is
346 // limited-range (16..235 Y, 16..240 Cb/Cr) to match
347 // broadcast / SDI / ST 2110 defaults. The explicit
348 // "_Full" suffix opts in to full-range 0..255 Y and
349 // 0..255 Cb/Cr, which is what JPEG / JFIF uses and
350 // what most consumer video decoders (ffplay,
351 // browsers, libjpeg-turbo) expect. These
352 // PixelFormats exist both as general-purpose
353 // full-range YCbCr storage and as encode-source
354 // intermediates for the full-range JPEG variants
355 // below.
356 YUV8_422_Rec709_Full = 143,
357 YUV8_422_Rec601_Full = 144,
358 YUV8_420_Planar_Rec709_Full = 145,

361 // -- Full complement of JPEG YCbCr variants (matrix × range) --
362 //
363 // Following the library-wide YCbCr convention: the
364 // unsuffixed JPEG names (existing
365 // JPEG_YUV8_422_Rec709 / JPEG_YUV8_420_Rec709) are
366 // limited-range to stay consistent with the
367 // uncompressed YCbCr defaults. Full-range variants
368 // take the explicit "_Full" suffix. All eight
369 // combinations (matrix × range × subsampling) are
370 // provided; pick the one that matches what the
371 // downstream decoder expects to interpret:
372 //
373 // - Rec.601 + full range (strict JFIF) → ffplay,
374 // browsers, libjpeg-turbo, most consumer apps
375 // - Rec.709 + full range → modern cameras with
376 // an ICC profile; good for high-quality still
377 // storage that a colour-managed viewer honours
378 // - Rec.709 + limited range (existing default) →
379 // broadcast / SDI JPEG pipelines (ST 2110 JPEG XS
380 // style)
381 // - Rec.601 + limited range → legacy broadcast
382 JPEG_YUV8_422_Rec601 =
383 147,

385 148,
