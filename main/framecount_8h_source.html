299inline FrameNumber operator+(FrameNumber a, const FrameCount &c) {

305inline FrameNumber operator-(FrameNumber a, const FrameCount &c) {
306 if (a.isUnknown() || c.isUnknown() || c.isInfinite()) return FrameNumber::unknown();
