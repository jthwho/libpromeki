625 String format(const String &spec, Error *err = nullptr) const {
626 if(err != nullptr) *err = Error::Ok;
627 String defaultStr = get<String>();
628 if(spec.isEmpty()) return defaultStr;
629
630 std::string fmtStr;
631 fmtStr.reserve(spec.byteCount() + 3);
632 fmtStr += "{:";
633 fmtStr.append(spec.cstr(), spec.byteCount());
634 fmtStr += '}';
635
636 try {
637 return std::visit([&fmtStr, &defaultStr](auto &&arg) -> String {
638 using T = std::decay_t<decltype(arg)>;
639 if constexpr (std::is_same_v<T, std::monostate>) {

641 return String();

643 const auto &val = arg;

706 Enum asEnum(Enum::Type enumType, Error *err = nullptr) const {
707 if(!enumType.isValid()) {
708 if(err != nullptr) *err = Error::InvalidArgument;
709 return Enum();
710 }
711 auto setOk = [err]() { if(err != nullptr) *err = Error::Ok; };
712 auto setErr = [err]() { if(err != nullptr) *err = Error::Invalid; };
713 switch(type()) {
714 case TypeInvalid:
715 setOk();
716 return Enum(enumType);
717 case TypeEnum: {
718 Enum e = get<Enum>();
719 if(e.type() != enumType) {
720 setErr();
721 return Enum();
722 }
723 setOk();
724 return e;
725 }
726 case TypeString: {
727 String s = get<String>();
728 // Qualified "TypeName::ValueName"?
729 if(s.contains("::")) {
730 Error parseErr;
731 Enum e = Enum::lookup(s, &parseErr);
732 if(parseErr.isOk() && e.type() == enumType) {
733 setOk();
734 return e;
735 }
736 setErr();
737 return Enum();
738 }
739 // Unqualified value name against the target type.
740 Enum byName(enumType, s);
741 if(byName.hasListedValue()) {
742 setOk();
743 return byName;
744 }
745 // Last resort: signed decimal integer.
746 Error intErr;
747 int iv = s.template to<int>(&intErr);
748 if(intErr.isOk()) {
749 setOk();
750 return Enum(enumType, iv);
751 }
752 setErr();
753 return Enum();
754 }
755 case TypeBool:
756 case TypeU8:
757 case TypeS8:
758 case TypeU16:
759 case TypeS16:
760 case TypeU32:
761 case TypeS32:
762 case TypeU64:
763 case TypeS64: {
764 Error ge;
765 int iv = get<int32_t>(&ge);
766 if(ge.isError()) {
767 setErr();
768 return Enum();
769 }
770 setOk();
771 return Enum(enumType, iv);
772 }
773 default:
774 break;
775 }
776 setErr();
777 return Enum();
778 }

808 bool operator==(const VariantImpl &other) const {
809 return std::visit([this, &other](auto &&a, auto &&b) -> bool {
810 using A = std::decay_t<decltype(a)>;
811 using B = std::decay_t<decltype(b)>;
812 if constexpr (std::is_same_v<A, B>) {
813 return a == b;
814 } else if constexpr (std::is_arithmetic_v<A> && std::is_arithmetic_v<B>) {
815 if constexpr (std::is_floating_point_v<A> || std::is_floating_point_v<B>) {
816 return static_cast<double>(a) == static_cast<double>(b);
817 } else if constexpr (std::is_signed_v<A> && !std::is_signed_v<B>) {
818 if(a < 0) return false;
819 return static_cast<uint64_t>(a) == static_cast<uint64_t>(b);
820 } else if constexpr (!std::is_signed_v<A> && std::is_signed_v<B>) {
821 if(b < 0) return false;
822 return static_cast<uint64_t>(a) == static_cast<uint64_t>(b);
823 } else {
824 using Common = std::common_type_t<A, B>;
825 return static_cast<Common>(a) == static_cast<Common>(b);
826 }
827 } else {
828 Error err;
829 A ca = other.template get<A>(&err);
830 if(err.isOk() && a == ca) return true;
831 B cb = get<B>(&err);
832 if(err.isOk() && cb == b) return true;
833 return false;
834 }
835 }, v, other.v);
836 }
