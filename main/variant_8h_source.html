440 if (err != nullptr) *err = Error::Ok;
441 String defaultStr = get<String>();
442 if (spec.isEmpty()) return defaultStr;
443

523 Enum asEnum(Enum::Type enumType, Error *err = nullptr) const {
524 if (!enumType.isValid()) {
525 if (err != nullptr) *err = Error::InvalidArgument;
526 return Enum();
527 }
528 auto setOk = [err]() {
529 if (err != nullptr) *err = Error::Ok;
530 };
531 auto setErr = [err]() {
532 if (err != nullptr) *err = Error::Invalid;
533 };
534 switch (type()) {
535 case TypeInvalid: setOk(); return Enum(enumType);
536 case TypeEnum: {
537 Enum e = get<Enum>();
538 if (e.type() != enumType) {
539 setErr();
540 return Enum();
541 }
542 setOk();
543 return e;
544 }
545 case TypeString: {
546 String s = get<String>();
547 // Qualified "TypeName::ValueName"?
548 if (s.contains("::")) {
549 Error parseErr;
550 Enum e = Enum::lookup(s, &parseErr);
551 if (parseErr.isOk() && e.type() == enumType) {
552 setOk();
553 return e;
554 }
555 setErr();
556 return Enum();
557 }
558 // Unqualified value name against the target type.
559 Enum byName(enumType, s);
560 if (byName.hasListedValue()) {
561 setOk();
562 return byName;
563 }
564 // Last resort: signed decimal integer.
565 Error intErr;
566 int iv = s.template to<int>(&intErr);
567 if (intErr.isOk()) {
568 setOk();
569 return Enum(enumType, iv);
570 }
571 setErr();
572 return Enum();
573 }
574 case TypeBool:
575 case TypeU8:
576 case TypeS8:
577 case TypeU16:
578 case TypeS16:
579 case TypeU32:
580 case TypeS32:
581 case TypeU64:
582 case TypeS64: {
583 Error ge;
584 int iv = get<int32_t>(&ge);
585 if (ge.isError()) {
586 setErr();
587 return Enum();
588 }
589 setOk();
590 return Enum(enumType, iv);
591 }
592 default: break;
593 }
594 setErr();
595 return Enum();
596 }

950// Out-of-line VariantImpl::fromJson — must come after VariantList /
951// VariantMap class definitions so the body can construct them.
952template <typename... Types>
953VariantImpl<Types...> VariantImpl<Types...>::fromJson(const nlohmann::json &val) {
954 if (val.is_null()) return VariantImpl();
955 if (val.is_boolean()) return val.get<bool>();
956 if (val.is_number_integer()) {
957 if (val.is_number_unsigned()) return val.get<uint64_t>();
958 return val.get<int64_t>();
959 }
960 if (val.is_number_float()) return val.get<double>();
961 if (val.is_string()) return String(val.get<std::string>());
962 if (val.is_array()) {
963 VariantList list;
964 list.reserve(val.size());
965 for (const auto &item : val) {
966 list.pushToBack(Variant(VariantImpl::fromJson(item)));
967 }
968 return list;
969 }
970 if (val.is_object()) {
971 VariantMap map;
972 for (auto it = val.begin(); it != val.end(); ++it) {
973 map.insert(String(it.key()), Variant(VariantImpl::fromJson(it.value())));
974 }
975 return map;
976 }
977 return String(val.dump());
978}
