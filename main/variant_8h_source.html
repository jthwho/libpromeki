209 static VariantImpl fromJson(const nlohmann::json &val) {
210 if(val.is_null()) return VariantImpl();
211 if(val.is_boolean()) return val.get<bool>();
212 if(val.is_number_integer()) {
213 if(val.is_number_unsigned()) return val.get<uint64_t>();

567 Enum asEnum(Enum::Type enumType, Error *err = nullptr) const {
568 if(!enumType.isValid()) {
569 if(err != nullptr) *err = Error::InvalidArgument;
570 return Enum();
571 }
572 auto setOk = [err]() { if(err != nullptr) *err = Error::Ok; };
573 auto setErr = [err]() { if(err != nullptr) *err = Error::Invalid; };
574 switch(type()) {
575 case TypeInvalid:
576 setOk();
577 return Enum(enumType);
578 case TypeEnum: {
579 Enum e = get<Enum>();
580 if(e.type() != enumType) {
581 setErr();
582 return Enum();
583 }
584 setOk();
585 return e;
586 }
587 case TypeString: {
588 String s = get<String>();
589 // Qualified "TypeName::ValueName"?
590 if(s.contains("::")) {
591 Error parseErr;
592 Enum e = Enum::lookup(s, &parseErr);
593 if(parseErr.isOk() && e.type() == enumType) {
594 setOk();
595 return e;
596 }
597 setErr();
598 return Enum();
599 }
600 // Unqualified value name against the target type.
601 Enum byName(enumType, s);
602 if(byName.hasListedValue()) {
603 setOk();
604 return byName;
605 }
606 // Last resort: signed decimal integer.
607 Error intErr;
608 int iv = s.template to<int>(&intErr);
609 if(intErr.isOk()) {
610 setOk();
611 return Enum(enumType, iv);
612 }
613 setErr();
614 return Enum();
615 }
616 case TypeBool:
617 case TypeU8:
618 case TypeS8:
619 case TypeU16:
620 case TypeS16:
621 case TypeU32:
622 case TypeS32:
623 case TypeU64:
624 case TypeS64: {
625 Error ge;
626 int iv = get<int32_t>(&ge);
627 if(ge.isError()) {
628 setErr();
629 return Enum();
630 }
631 setOk();
632 return Enum(enumType, iv);
633 }
634 default:
635 break;
636 }
637 setErr();
638 return Enum();
639 }

669 bool operator==(const VariantImpl &other) const {
670 return std::visit([this, &other](auto &&a, auto &&b) -> bool {
671 using A = std::decay_t<decltype(a)>;
672 using B = std::decay_t<decltype(b)>;
673 if constexpr (std::is_same_v<A, B>) {
674 return a == b;
675 } else if constexpr (std::is_arithmetic_v<A> && std::is_arithmetic_v<B>) {
676 if constexpr (std::is_floating_point_v<A> || std::is_floating_point_v<B>) {
677 return static_cast<double>(a) == static_cast<double>(b);
678 } else if constexpr (std::is_signed_v<A> && !std::is_signed_v<B>) {
679 if(a < 0) return false;
680 return static_cast<uint64_t>(a) == static_cast<uint64_t>(b);
681 } else if constexpr (!std::is_signed_v<A> && std::is_signed_v<B>) {
682 if(b < 0) return false;
683 return static_cast<uint64_t>(a) == static_cast<uint64_t>(b);
684 } else {
685 using Common = std::common_type_t<A, B>;
686 return static_cast<Common>(a) == static_cast<Common>(b);
687 }
688 } else {
689 Error err;
690 A ca = other.template get<A>(&err);
691 if(err.isOk() && a == ca) return true;
692 B cb = get<B>(&err);
693 if(err.isOk() && cb == b) return true;
694 return false;
695 }
696 }, v, other.v);
697 }
