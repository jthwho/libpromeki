627 Enum asEnum(Enum::Type enumType, Error *err = nullptr) const {
628 if(!enumType.isValid()) {
629 if(err != nullptr) *err = Error::InvalidArgument;
630 return Enum();
631 }
632 auto setOk = [err]() { if(err != nullptr) *err = Error::Ok; };
633 auto setErr = [err]() { if(err != nullptr) *err = Error::Invalid; };
634 switch(type()) {
635 case TypeInvalid:
636 setOk();
637 return Enum(enumType);
638 case TypeEnum: {
639 Enum e = get<Enum>();
640 if(e.type() != enumType) {
641 setErr();
642 return Enum();
643 }
644 setOk();
645 return e;
646 }
647 case TypeString: {
648 String s = get<String>();
649 // Qualified "TypeName::ValueName"?
650 if(s.contains("::")) {
651 Error parseErr;
652 Enum e = Enum::lookup(s, &parseErr);
653 if(parseErr.isOk() && e.type() == enumType) {
654 setOk();
655 return e;
656 }
657 setErr();
658 return Enum();
659 }
660 // Unqualified value name against the target type.
661 Enum byName(enumType, s);
662 if(byName.hasListedValue()) {
663 setOk();
664 return byName;
665 }
666 // Last resort: signed decimal integer.
667 Error intErr;
668 int iv = s.template to<int>(&intErr);
669 if(intErr.isOk()) {
670 setOk();
671 return Enum(enumType, iv);
672 }
673 setErr();
674 return Enum();
675 }
676 case TypeBool:
677 case TypeU8:
678 case TypeS8:
679 case TypeU16:
680 case TypeS16:
681 case TypeU32:
682 case TypeS32:
683 case TypeU64:
684 case TypeS64: {
685 Error ge;
686 int iv = get<int32_t>(&ge);
687 if(ge.isError()) {
688 setErr();
689 return Enum();
690 }
691 setOk();
692 return Enum(enumType, iv);
693 }
694 default:
695 break;
696 }
697 setErr();
698 return Enum();
699 }

729 bool operator==(const VariantImpl &other) const {
730 return std::visit([this, &other](auto &&a, auto &&b) -> bool {
731 using A = std::decay_t<decltype(a)>;
732 using B = std::decay_t<decltype(b)>;
733 if constexpr (std::is_same_v<A, B>) {
734 return a == b;
735 } else if constexpr (std::is_arithmetic_v<A> && std::is_arithmetic_v<B>) {
736 if constexpr (std::is_floating_point_v<A> || std::is_floating_point_v<B>) {
737 return static_cast<double>(a) == static_cast<double>(b);
738 } else if constexpr (std::is_signed_v<A> && !std::is_signed_v<B>) {
739 if(a < 0) return false;
740 return static_cast<uint64_t>(a) == static_cast<uint64_t>(b);
741 } else if constexpr (!std::is_signed_v<A> && std::is_signed_v<B>) {
742 if(b < 0) return false;
743 return static_cast<uint64_t>(a) == static_cast<uint64_t>(b);
744 } else {
745 using Common = std::common_type_t<A, B>;
746 return static_cast<Common>(a) == static_cast<Common>(b);
747 }
748 } else {
749 Error err;
750 A ca = other.template get<A>(&err);
751 if(err.isOk() && a == ca) return true;
752 B cb = get<B>(&err);
753 if(err.isOk() && cb == b) return true;
754 return false;
755 }
756 }, v, other.v);
757 }
