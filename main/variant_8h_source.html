275 static VariantImpl fromJson(const nlohmann::json &val) {
276 if (val.is_null()) return VariantImpl();
277 if (val.is_boolean()) return val.get<bool>();
278 if (val.is_number_integer()) {
279 if (val.is_number_unsigned()) return val.get<uint64_t>();
280 return val.get<int64_t>();

282 if (val.is_number_float()) return val.get<double>();
283 if (val.is_string()) return String(val.get<std::string>());

288 list.pushToBack(String(item.get<std::string>()));

291 }
292 return list;
293 }
294 return String(val.dump());

417 String format(const String &spec, Error *err = nullptr) const {
418 if (err != nullptr) *err = Error::Ok;
419 String defaultStr = get<String>();
420 if (spec.isEmpty()) return defaultStr;
421
422 std::string fmtStr;
423 fmtStr.reserve(spec.byteCount() + 3);
424 fmtStr += "{:";
425 fmtStr.append(spec.cstr(), spec.byteCount());
426 fmtStr += '}';
427
428 try {
429 return std::visit(
430 [&fmtStr, &defaultStr](auto &&arg) -> String {
431 using T = std::decay_t<decltype(arg)>;
432 if constexpr (std::is_same_v<T, std::monostate>) {
433 (void)fmtStr;
434 return String();
435 } else if constexpr (std::is_default_constructible_v<
436 std::formatter<T, char>>) {
437 const auto &val = arg;
438 return String(std::vformat(fmtStr, std::make_format_args(val)));
439 } else {
440 std::string_view sv(defaultStr.cstr(), defaultStr.byteCount());
441 return String(std::vformat(fmtStr, std::make_format_args(sv)));
442 }
443 },
444 v);
445 } catch (const std::format_error &) {
446 if (err != nullptr) *err = Error::Invalid;
447 return defaultStr;
448 }
449 }

501 Enum asEnum(Enum::Type enumType, Error *err = nullptr) const {
502 if (!enumType.isValid()) {
503 if (err != nullptr) *err = Error::InvalidArgument;
504 return Enum();
505 }
506 auto setOk = [err]() {
507 if (err != nullptr) *err = Error::Ok;
508 };
509 auto setErr = [err]() {
510 if (err != nullptr) *err = Error::Invalid;
511 };
512 switch (type()) {
513 case TypeInvalid: setOk(); return Enum(enumType);
514 case TypeEnum: {
515 Enum e = get<Enum>();
516 if (e.type() != enumType) {
517 setErr();
518 return Enum();
519 }
520 setOk();
521 return e;
522 }
523 case TypeString: {
524 String s = get<String>();
525 // Qualified "TypeName::ValueName"?
526 if (s.contains("::")) {
527 Error parseErr;
528 Enum e = Enum::lookup(s, &parseErr);
529 if (parseErr.isOk() && e.type() == enumType) {
530 setOk();
531 return e;
532 }
533 setErr();
534 return Enum();
535 }
536 // Unqualified value name against the target type.
537 Enum byName(enumType, s);
538 if (byName.hasListedValue()) {
539 setOk();
540 return byName;
541 }
542 // Last resort: signed decimal integer.
543 Error intErr;
544 int iv = s.template to<int>(&intErr);
545 if (intErr.isOk()) {
546 setOk();
547 return Enum(enumType, iv);
548 }
549 setErr();
550 return Enum();
551 }
552 case TypeBool:
553 case TypeU8:
554 case TypeS8:
555 case TypeU16:
556 case TypeS16:
557 case TypeU32:
558 case TypeS32:
559 case TypeU64:
560 case TypeS64: {
561 Error ge;
562 int iv = get<int32_t>(&ge);
563 if (ge.isError()) {
564 setErr();
565 return Enum();
566 }
567 setOk();
568 return Enum(enumType, iv);
569 }
570 default: break;
571 }
572 setErr();
573 return Enum();
574 }
