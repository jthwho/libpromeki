401 String format(const String &spec, Error *err = nullptr) const {
402 if(err != nullptr) *err = Error::Ok;
403 String defaultStr = get<String>();
404 if(spec.isEmpty()) return defaultStr;
405
406 std::string fmtStr;
407 fmtStr.reserve(spec.byteCount() + 3);
408 fmtStr += "{:";
409 fmtStr.append(spec.cstr(), spec.byteCount());
410 fmtStr += '}';
411
412 try {
413 return std::visit([&fmtStr, &defaultStr](auto &&arg) -> String {
414 using T = std::decay_t<decltype(arg)>;
415 if constexpr (std::is_same_v<T, std::monostate>) {
416 (void)fmtStr;
417 return String();
418 } else if constexpr (std::is_default_constructible_v<std::formatter<T, char>>) {
419 const auto &val = arg;
420 return String(std::vformat(fmtStr, std::make_format_args(val)));
421 } else {
422 std::string_view sv(defaultStr.cstr(), defaultStr.byteCount());
423 return String(std::vformat(fmtStr, std::make_format_args(sv)));
424 }
425 }, v);
426 } catch(const std::format_error &) {
427 if(err != nullptr) *err = Error::Invalid;
428 return defaultStr;
429 }
430 }

482 Enum asEnum(Enum::Type enumType, Error *err = nullptr) const {
483 if(!enumType.isValid()) {
484 if(err != nullptr) *err = Error::InvalidArgument;
485 return Enum();
486 }
487 auto setOk = [err]() { if(err != nullptr) *err = Error::Ok; };
488 auto setErr = [err]() { if(err != nullptr) *err = Error::Invalid; };
489 switch(type()) {
490 case TypeInvalid:
491 setOk();
492 return Enum(enumType);
493 case TypeEnum: {
494 Enum e = get<Enum>();
495 if(e.type() != enumType) {
496 setErr();
497 return Enum();
498 }
499 setOk();
500 return e;
501 }
502 case TypeString: {
503 String s = get<String>();
504 // Qualified "TypeName::ValueName"?
505 if(s.contains("::")) {
506 Error parseErr;
507 Enum e = Enum::lookup(s, &parseErr);
508 if(parseErr.isOk() && e.type() == enumType) {
509 setOk();
510 return e;
511 }
512 setErr();
513 return Enum();
514 }
515 // Unqualified value name against the target type.
516 Enum byName(enumType, s);
517 if(byName.hasListedValue()) {
518 setOk();
519 return byName;
520 }
521 // Last resort: signed decimal integer.
522 Error intErr;
523 int iv = s.template to<int>(&intErr);
524 if(intErr.isOk()) {
525 setOk();
526 return Enum(enumType, iv);
527 }
528 setErr();
529 return Enum();
530 }
531 case TypeBool:
532 case TypeU8:
533 case TypeS8:
534 case TypeU16:
535 case TypeS16:
536 case TypeU32:
537 case TypeS32:
538 case TypeU64:
539 case TypeS64: {
540 Error ge;
541 int iv = get<int32_t>(&ge);
542 if(ge.isError()) {
543 setErr();
544 return Enum();
545 }
546 setOk();
547 return Enum(enumType, iv);
548 }
549 default:
550 break;
551 }
552 setErr();
553 return Enum();
554 }
