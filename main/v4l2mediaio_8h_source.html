230 // -- Push-time tracking for audio MediaTimeStamps --
231 //
232 // ALSA delivers samples in small chunks; we need to
233 // stamp each outgoing audio chunk (one per video
234 // frame) with the wall-time the first sample was
235 // pushed, so downstream drift correction can derive
236 // the true source audio rate from timestamp deltas.
237 //
238 // The list holds one record per outstanding push;
239 // the front is the oldest unpopped sample. When a
240 // record's samplesRemaining reaches zero it is
241 // removed; when a pop consumes only part of a record
242 // its wallNs is advanced by (consumed / sampleRate).
243 struct AudioPushRecord {
244 int64_t wallNs;
245 size_t samplesRemaining;
246 };

248 List<AudioPushRecord> _audioPushRecords;

280 //
281 // Sequence tracking uses @c vbuf.sequence from

283 // (where the camera produced a frame but no buffer
