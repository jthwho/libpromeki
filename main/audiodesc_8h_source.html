91 template <typename IntegerType, IntegerType Min, IntegerType Max>
92 static IntegerType floatToInteger(float value) {
93 static_assert(std::is_integral<IntegerType>::value, "IntegerType must be an integer.");
94 const float min = static_cast<float>(Min);
95 const float max = static_cast<float>(Max);
96 if(value <= -1.0f) return Min;

98 return static_cast<IntegerType>((value + 1.0f) * 0.5f * (max - min) + min);

126 template <typename IntegerType, bool InputIsBigEndian>
127 static void samplesToFloat(float *out, const uint8_t *inbuf, size_t samples) {
128 static_assert(std::is_integral<IntegerType>::value, "IntegerType must be an integer.");
129 const IntegerType *in = reinterpret_cast<const IntegerType *>(inbuf);
130 for(size_t i = 0; i < samples; ++i) {
131 IntegerType val = *in++;
132 if constexpr (InputIsBigEndian != System::isBigEndian()) System::swapEndian(val);
133 *out++ = integerToFloat<IntegerType>(val);
134 }
135 return;
136 }

150 static void floatToSamples(uint8_t *outbuf, const float *in, size_t samples) {
151 static_assert(std::is_integral<IntegerType>::value, "IntegerType must be an integer.");
152 IntegerType *out = reinterpret_cast<IntegerType *>(outbuf);

573inline DataStream &operator>>(DataStream &stream, AudioDesc &desc) {
574 if(!stream.readTag(DataStream::TypeAudioDesc)) { desc = AudioDesc(); return stream; }
575 int32_t dt = 0;
576 float sr = 0.0f;
577 uint32_t ch = 0;
578 Metadata meta;
579 stream >> dt >> sr >> ch >> meta;
580 if(stream.status() != DataStream::Ok) { desc = AudioDesc(); return stream; }
581 desc = AudioDesc(static_cast<AudioDesc::DataType>(dt), sr, ch);
582 desc.metadata() = std::move(meta);
583 return stream;
584}
