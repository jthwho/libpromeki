97 template <typename IntegerType, IntegerType Min, IntegerType Max>
98 static IntegerType floatToInteger(float value) {
99 static_assert(std::is_integral<IntegerType>::value, "IntegerType must be an integer.");
100 const float min = static_cast<float>(Min);
101 const float max = static_cast<float>(Max);
102 if(value <= -1.0f) return Min;
103 else if(value >= 1.0f) return Max;
104 return static_cast<IntegerType>((value + 1.0f) * 0.5f * (max - min) + min);

132 template <typename IntegerType, bool InputIsBigEndian>
133 static void samplesToFloat(float *out, const uint8_t *inbuf, size_t samples) {
134 static_assert(std::is_integral<IntegerType>::value, "IntegerType must be an integer.");
135 const IntegerType *in = reinterpret_cast<const IntegerType *>(inbuf);
136 for(size_t i = 0; i < samples; ++i) {
137 IntegerType val = *in++;
138 if constexpr (InputIsBigEndian != System::isBigEndian()) System::swapEndian(val);
139 *out++ = integerToFloat<IntegerType>(val);
140 }
141 return;
142 }

155 template <typename IntegerType, bool OutputIsBigEndian>
156 static void floatToSamples(uint8_t *outbuf, const float *in, size_t samples) {
157 static_assert(std::is_integral<IntegerType>::value, "IntegerType must be an integer.");
158 IntegerType *out = reinterpret_cast<IntegerType *>(outbuf);
159 for(size_t i = 0; i < samples; ++i) {
160 IntegerType val = floatToInteger<IntegerType>(*in++);
161 if constexpr (OutputIsBigEndian != System::isBigEndian()) System::swapEndian(val);
162 *out++ = val;

596inline DataStream &operator>>(DataStream &stream, AudioDesc &desc) {
597 if(!stream.readTag(DataStream::TypeAudioDesc)) { desc = AudioDesc(); return stream; }
598 int32_t dt = 0;
599 float sr = 0.0f;
600 uint32_t ch = 0;
601 Metadata meta;
602 stream >> dt >> sr >> ch >> meta;
603 if(stream.status() != DataStream::Ok) { desc = AudioDesc(); return stream; }
604 desc = AudioDesc(static_cast<AudioDesc::DataType>(dt), sr, ch);
605 desc.metadata() = std::move(meta);
606 return stream;
607}
