317inline DataStream &operator>>(DataStream &stream, AudioDesc &desc) {
318 if(!stream.readTag(DataStream::TypeAudioDesc)) { desc = AudioDesc(); return stream; }
319 AudioFormat fmt;
320 float sr = 0.0f;
321 uint32_t ch = 0;
322 Metadata meta;
323 stream >> fmt >> sr >> ch >> meta;
324 if(stream.status() != DataStream::Ok) { desc = AudioDesc(); return stream; }
