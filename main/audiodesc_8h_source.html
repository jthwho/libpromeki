367inline DataStream &operator>>(DataStream &stream, AudioDesc &desc) {
368 if (!stream.readTag(DataStream::TypeAudioDesc)) {
369 desc = AudioDesc();
370 return stream;
371 }
372 AudioFormat fmt;
373 float sr = 0.0f;
374 uint32_t ch = 0;
375 AudioChannelMap map;
376 Metadata meta;
377 stream >> fmt >> sr >> ch >> map >> meta;
378 if (stream.status() != DataStream::Ok) {
379 desc = AudioDesc();
380 return stream;
381 }
382 desc = AudioDesc(fmt, sr, ch);
383 // Restore the explicit channel map if it parsed validly; otherwise keep
384 // the default that the (fmt, sr, ch) constructor installs.
385 if (map.channels() == ch) desc.setChannelMap(map);
386 desc.metadata() = std::move(meta);
387 return stream;
388}
