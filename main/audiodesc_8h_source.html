89 template <typename IntegerType, IntegerType Min, IntegerType Max>
90 static IntegerType floatToInteger(float value) {
91 static_assert(std::is_integral<IntegerType>::value, "IntegerType must be an integer.");
92 const float min = static_cast<float>(Min);
93 const float max = static_cast<float>(Max);
94 if(value <= -1.0f) return Min;
95 else if(value >= 1.0f) return Max;
96 return static_cast<IntegerType>((value + 1.0f) * 0.5f * (max - min) + min);

124 template <typename IntegerType, bool InputIsBigEndian>
125 static void samplesToFloat(float *out, const uint8_t *inbuf, size_t samples) {
126 static_assert(std::is_integral<IntegerType>::value, "IntegerType must be an integer.");
127 const IntegerType *in = reinterpret_cast<const IntegerType *>(inbuf);
128 for(size_t i = 0; i < samples; ++i) {
129 IntegerType val = *in++;
130 if constexpr (InputIsBigEndian != System::isBigEndian()) System::swapEndian(val);
131 *out++ = integerToFloat<IntegerType>(val);
132 }
133 return;
134 }

147 template <typename IntegerType, bool OutputIsBigEndian>
148 static void floatToSamples(uint8_t *outbuf, const float *in, size_t samples) {
149 static_assert(std::is_integral<IntegerType>::value, "IntegerType must be an integer.");
150 IntegerType *out = reinterpret_cast<IntegerType *>(outbuf);
151 for(size_t i = 0; i < samples; ++i) {
152 IntegerType val = floatToInteger<IntegerType>(*in++);
153 if constexpr (OutputIsBigEndian != System::isBigEndian()) System::swapEndian(val);

530inline DataStream &operator>>(DataStream &stream, AudioDesc &desc) {
531 if(!stream.readTag(DataStream::TypeAudioDesc)) { desc = AudioDesc(); return stream; }
532 int32_t dt = 0;
533 float sr = 0.0f;
534 uint32_t ch = 0;
535 Metadata meta;
536 stream >> dt >> sr >> ch >> meta;
537 if(stream.status() != DataStream::Ok) { desc = AudioDesc(); return stream; }
538 desc = AudioDesc(static_cast<AudioDesc::DataType>(dt), sr, ch);
539 desc.metadata() = std::move(meta);
540 return stream;
