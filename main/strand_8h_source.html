224 auto submitImpl(F &&callable, TaskFunc onCancel, bool urgent) -> Future<std::invoke_result_t<F>> {
225 using R = std::invoke_result_t<F>;
226 auto promise = std::make_shared<Promise<R>>();
227 Future<R> future = promise->future();
228
229 Entry entry;
230 entry.run = [promise, f = std::forward<F>(callable)]() mutable {
231 if constexpr (std::is_void_v<R>) {
232 f();
233 promise->setValue();
234 } else {
235 promise->setValue(f());
236 }
237 };
238 entry.cancel = [promise, onCancel = std::move(onCancel)]() mutable {
239 if (onCancel) onCancel();
240 promise->setError(Error::Cancelled);
241 };
242
243 bool needSpawn = false;
244 {
245 Mutex::Locker lock(_mutex);
246 if (urgent) {
247 _queue.emplace_front(std::move(entry));
248 } else {
249 _queue.emplace_back(std::move(entry));
250 }
251 if (!_running) {
252 _running = true;
253 needSpawn = true;
254 }
255 }
256 if (needSpawn) {
257 _pool.submit([this] { runNext(); });
258 }
259 return future;
260 }

267 void runNext() {
268 Entry entry;
269 bool haveTask = false;
270 {
271 Mutex::Locker lock(_mutex);
272 if (!_queue.empty()) {
273 entry = std::move(_queue.front());
274 _queue.pop_front();
275 haveTask = true;
276 }
277 }
278
279 // Possible if cancelPending() emptied the queue
280 // between spawn and execution.
281 if (haveTask) entry.run();
282
283 // Either re-submit (more work pending) or mark idle.
284 bool reSpawn = false;
285 {
286 Mutex::Locker lock(_mutex);
287 if (_queue.empty()) {
288 _running = false;
289 _idleCv.wakeAll();
290 } else {
291 reSpawn = true;
292 }
293 }
294
295 if (reSpawn) {
296 _pool.submit([this] { runNext(); });
297 }
298 }
