341 // AvSync tone burst cache: built lazily per requested
342 // sample count so the cadenced rates (29.97, 59.94)
343 // which alternate between two sizes don't thrash the
344 // cache. Each entry is a one-shot tone burst (phase
345 // reset to zero so every marker frame is byte-identical).
346 mutable Map<size_t, Audio> _avSyncToneCache;
347
348 // Cached noise buffers. One-dimensional arrays of
349 // samples that the per-channel noise dispatch indexes
350 // with a per-channel offset so different WhiteNoise /
351 // PinkNoise channels stay decorrelated while sharing
352 // the same underlying samples. Generated at most once
353 // per configure() call; empty until a noise channel
354 // requests them.
355 List<float> _whiteNoiseBuffer;
356 List<float> _pinkNoiseBuffer;
357
358 // Continuous-sweep cursor/phase for the Chirp
359 // generator. The cursor is the sample position inside
360 // the current sweep period (wraps at
361 // _chirpDurationSec * sampleRate). The phase is the
362 // running sin-argument; we accumulate it sample-by-
363 // sample instead of recomputing it from the closed-
364 // form integral so the waveform stays sample-exact
365 // continuous across both chunk boundaries and period
366 // wraps (the closed form resets to 0 at each period
367 // wrap, which causes an audible click).
368 mutable double _chirpSampleCursor = 0.0;
369 mutable double _chirpPhase = 0.0;
370
371 // Per-tone phase accumulators for the DualTone
372 // generator. Tracking these across create() calls is
373 // the only way to avoid a once-per-chunk phase-reset
374 // click; a 30 fps pipeline restarts the pattern 30
375 // times a second otherwise, which is plainly audible.
376 mutable double _dualTonePhase1 = 0.0;
377 mutable double _dualTonePhase2 = 0.0;
378
379 // Cumulative sample cursor for the cached noise
380 // buffers. Advanced once per create() call so
381 // successive frames read from successive regions of
382 // the buffer instead of re-reading the same slice —
383 // otherwise the noise would modulate at the frame
384 // rate and sound like a tone, not noise. All noise
385 // channels share this cursor, with a per-channel
386 // offset layered on top so multiple noise channels
387 // on the same stream stay decorrelated.
388 mutable size_t _noiseSampleCursor = 0;
389
