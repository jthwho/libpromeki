56 static Matrix3x3 rotationMatrix(float angle, char axis) {
57 Matrix3x3 result;
58 float cos_angle = std::cos(angle);
59 float sin_angle = std::sin(angle);
60 switch (axis) {
61 case 'x':
62 result.data[0][0] = 1.0f;
63 result.data[1][1] = cos_angle;
64 result.data[1][2] = -sin_angle;
65 result.data[2][1] = sin_angle;
66 result.data[2][2] = cos_angle;
67 break;
68 case 'y':
69 result.data[0][0] = cos_angle;
70 result.data[0][2] = sin_angle;
71 result.data[1][1] = 1.0f;
72 result.data[2][0] = -sin_angle;
73 result.data[2][2] = cos_angle;
74 break;
75 case 'z':
76 result.data[0][0] = cos_angle;
77 result.data[0][1] = -sin_angle;
78 result.data[1][0] = sin_angle;
79 result.data[1][1] = cos_angle;

219 Matrix3x3 inverse() const {
220 Matrix3x3 result;
221 float det = determinant();
222 if (det == 0.0f) {
223 // Singular matrix, inverse does not exist
224 return result; // Return zero matrix
225 }
226 float inv_det = 1.0f / det;
227 result.data[0][0] = (data[1][1] * data[2][2] - data[1][2] * data[2][1]) * inv_det;
228 result.data[0][1] = (data[0][2] * data[2][1] - data[0][1] * data[2][2]) * inv_det;
229 result.data[0][2] = (data[0][1] * data[1][2] - data[0][2] * data[1][1]) * inv_det;
230 result.data[1][0] = (data[1][2] * data[2][0] - data[1][0] * data[2][2]) * inv_det;

344 Matrix3x3 elementDivide(const Matrix3x3 &other) const {
345 Matrix3x3 result;
346 for (int i = 0; i < 3; ++i) {
347 __m128 row = _mm_set_ps(0.0f, data[i][2], data[i][1], data[i][0]);
348 __m128 other_row =
349 _mm_set_ps(0.0f, other.data[i][2], other.data[i][1], other.data[i][0]);
350 // Create a mask to handle division by zero
351 __m128 mask = _mm_cmpneq_ps(other_row, _mm_set1_ps(0.0f));
352 __m128 div = _mm_div_ps(row, other_row);
353 // Apply the mask to avoid NaN values due to division by zero
354 div = _mm_and_ps(div, mask);
355 _mm_storeu_ps(result.data[i], div);
356 }
357 return result;

364 void vectorTransform(float vector[3]) const {
