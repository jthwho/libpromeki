66 static Matrix3x3 rotationMatrix(float angle, char axis) {
67 Matrix3x3 result;
68 float cos_angle = std::cos(angle);
69 float sin_angle = std::sin(angle);
70 switch (axis) {
71 case 'x':
72 result.data[0][0] = 1.0f;
73 result.data[1][1] = cos_angle;
74 result.data[1][2] = -sin_angle;
75 result.data[2][1] = sin_angle;
76 result.data[2][2] = cos_angle;
77 break;
78 case 'y':
79 result.data[0][0] = cos_angle;
80 result.data[0][2] = sin_angle;
81 result.data[1][1] = 1.0f;
82 result.data[2][0] = -sin_angle;
83 result.data[2][2] = cos_angle;
84 break;

232 Matrix3x3 inverse() const {
233 Matrix3x3 result;
234 float det = determinant();
235 if (det == 0.0f) {
236 // Singular matrix, inverse does not exist
237 return result; // Return zero matrix
238 }
239 float inv_det = 1.0f / det;
240 result.data[0][0] = (data[1][1] * data[2][2] - data[1][2] * data[2][1]) * inv_det;
241 result.data[0][1] = (data[0][2] * data[2][1] - data[0][1] * data[2][2]) * inv_det;
242 result.data[0][2] = (data[0][1] * data[1][2] - data[0][2] * data[1][1]) * inv_det;
243 result.data[1][0] = (data[1][2] * data[2][0] - data[1][0] * data[2][2]) * inv_det;
244 result.data[1][1] = (data[0][0] * data[2][2] - data[0][2] * data[2][0]) * inv_det;

361 Matrix3x3 elementDivide(const Matrix3x3& other) const {
362 Matrix3x3 result;
363 for (int i = 0; i < 3; ++i) {
364 __m128 row = _mm_set_ps(0.0f, data[i][2], data[i][1], data[i][0]);
365 __m128 other_row = _mm_set_ps(0.0f, other.data[i][2], other.data[i][1], other.data[i][0]);
366 // Create a mask to handle division by zero
367 __m128 mask = _mm_cmpneq_ps(other_row, _mm_set1_ps(0.0f));
368 __m128 div = _mm_div_ps(row, other_row);
369 // Apply the mask to avoid NaN values due to division by zero
370 div = _mm_and_ps(div, mask);
371 _mm_storeu_ps(result.data[i], div);
