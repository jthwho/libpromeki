68 float sin_angle = std::sin(angle);
69 switch (axis) {
70 case 'x':
71 result.data[0][0] = 1.0f;
72 result.data[1][1] = cos_angle;
73 result.data[1][2] = -sin_angle;
74 result.data[2][1] = sin_angle;
75 result.data[2][2] = cos_angle;
76 break;
77 case 'y':
78 result.data[0][0] = cos_angle;
79 result.data[0][2] = sin_angle;

231 Matrix3x3 inverse() const {
232 Matrix3x3 result;
233 float det = determinant();
234 if (det == 0.0f) {
235 // Singular matrix, inverse does not exist
236 return result; // Return zero matrix
237 }
238 float inv_det = 1.0f / det;
239 result.data[0][0] = (data[1][1] * data[2][2] - data[1][2] * data[2][1]) * inv_det;
240 result.data[0][1] = (data[0][2] * data[2][1] - data[0][1] * data[2][2]) * inv_det;
241 result.data[0][2] = (data[0][1] * data[1][2] - data[0][2] * data[1][1]) * inv_det;
242 result.data[1][0] = (data[1][2] * data[2][0] - data[1][0] * data[2][2]) * inv_det;
