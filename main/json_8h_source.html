272 template <typename T>
273 static T getVal(const nlohmann::json &val, Error *err) {
274 T ret{};
275 bool good = false;
276 try {
277 if constexpr (std::is_same_v<T, bool>) {
278 if(val.is_boolean()) { ret = val.get<bool>(); good = true; }
279 else if(val.is_number_integer()) { ret = val.get<int64_t>() != 0; good = true; }
280 } else if constexpr (std::is_integral_v<T>) {
281 if(val.is_number()) { ret = static_cast<T>(val.get<int64_t>()); good = true; }
282 else if(val.is_boolean()) { ret = val.get<bool>() ? 1 : 0; good = true; }
283 } else if constexpr (std::is_floating_point_v<T>) {
284 if(val.is_number()) { ret = val.get<T>(); good = true; }
285 } else if constexpr (std::is_same_v<T, std::string>) {
286 if(val.is_string()) { ret = val.get<std::string>(); good = true; }
287 else if(!val.is_null()) { ret = val.dump(); good = true; }
288 }
289 } catch(...) {}
290 if(err) *err = good ? Error::Ok : Error::Invalid;
291 return ret;
292 }
