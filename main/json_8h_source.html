291 template <typename T>
292 static T getVal(const nlohmann::json &val, Error *err) {
293 T ret{};
294 bool good = false;
295 try {
296 if constexpr (std::is_same_v<T, bool>) {
297 if(val.is_boolean()) { ret = val.get<bool>(); good = true; }
298 else if(val.is_number_integer()) { ret = val.get<int64_t>() != 0; good = true; }
299 } else if constexpr (std::is_integral_v<T>) {
300 if(val.is_number()) { ret = static_cast<T>(val.get<int64_t>()); good = true; }
301 else if(val.is_boolean()) { ret = val.get<bool>() ? 1 : 0; good = true; }
302 } else if constexpr (std::is_floating_point_v<T>) {
303 if(val.is_number()) { ret = val.get<T>(); good = true; }
304 } else if constexpr (std::is_same_v<T, std::string>) {
305 if(val.is_string()) { ret = val.get<std::string>(); good = true; }
306 else if(!val.is_null()) { ret = val.dump(); good = true; }
307 }
308 } catch(...) {}
309 if(err) *err = good ? Error::Ok : Error::Invalid;
310 return ret;
311 }

553// ============================================================================
554// DataStream serialization
555// ============================================================================
556//
557// JsonObject and JsonArray are serialized as their compact JSON text form.
558// This keeps the wire format stable across nlohmann::json upgrades and makes
559// cross-reading with json text-based tools trivial. Readers that see a
560// truncated or malformed payload are marked as ReadCorruptData.
561
568inline DataStream &operator<<(DataStream &stream, const JsonObject &obj) {
569 stream.writeTag(DataStream::TypeJsonObject);
570 stream << obj.toString(0);
571 return stream;
572}
573
580inline DataStream &operator>>(DataStream &stream, JsonObject &obj) {
581 if(!stream.readTag(DataStream::TypeJsonObject)) { obj = JsonObject(); return stream; }
582 String text;
583 stream >> text;
584 if(stream.status() != DataStream::Ok) { obj = JsonObject(); return stream; }
585 Error err;
586 obj = JsonObject::parse(text, &err);
587 if(err.isError()) {
588 stream.setError(DataStream::ReadCorruptData,
589 String("JsonObject::parse failed"));
590 }
591 return stream;
592}

597inline DataStream &operator<<(DataStream &stream, const JsonArray &arr) {
598 stream.writeTag(DataStream::TypeJsonArray);

601}
