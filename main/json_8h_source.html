273 template <typename T> static T getVal(const nlohmann::json &val, Error *err) {
274 T ret{};
275 bool good = false;
276 try {
277 if constexpr (std::is_same_v<T, bool>) {
278 if (val.is_boolean()) {
279 ret = val.get<bool>();
280 good = true;
281 } else if (val.is_number_integer()) {
282 ret = val.get<int64_t>() != 0;
283 good = true;
284 }
285 } else if constexpr (std::is_integral_v<T>) {
286 if (val.is_number()) {
287 ret = static_cast<T>(val.get<int64_t>());
288 good = true;
289 } else if (val.is_boolean()) {
290 ret = val.get<bool>() ? 1 : 0;
291 good = true;
292 }
293 } else if constexpr (std::is_floating_point_v<T>) {
294 if (val.is_number()) {
295 ret = val.get<T>();
296 good = true;
297 }
298 } else if constexpr (std::is_same_v<T, std::string>) {
299 if (val.is_string()) {
300 ret = val.get<std::string>();
301 good = true;
302 } else if (!val.is_null()) {
303 ret = val.dump();
304 good = true;
305 }
306 }
307 } catch (...) {}
308 if (err) *err = good ? Error::Ok : Error::Invalid;
309 return ret;
310 }

529// ============================================================================
530// DataStream serialization
531// ============================================================================
532//
533// JsonObject and JsonArray are serialized as their compact JSON text form.
534// This keeps the wire format stable across nlohmann::json upgrades and makes
535// cross-reading with json text-based tools trivial. Readers that see a
536// truncated or malformed payload are marked as ReadCorruptData.
537
544inline DataStream &operator<<(DataStream &stream, const JsonObject &obj) {
545 stream.writeTag(DataStream::TypeJsonObject);
546 stream << obj.toString(0);
547 return stream;
548}
549
556inline DataStream &operator>>(DataStream &stream, JsonObject &obj) {
557 if (!stream.readTag(DataStream::TypeJsonObject)) {
558 obj = JsonObject();
559 return stream;
560 }
561 String text;
562 stream >> text;
563 if (stream.status() != DataStream::Ok) {
564 obj = JsonObject();
565 return stream;
566 }
567 Error err;
568 obj = JsonObject::parse(text, &err);
569 if (err.isError()) {
570 stream.setError(DataStream::ReadCorruptData, String("JsonObject::parse failed"));
571 }
572 return stream;
573}
