239 static int writeSpecMapHelp(TextStream &stream, const SpecMap &specs,
240 const StringList &skipKeys = StringList()) {
241 // Collect the visible set first so the column
242 // width pass doesn't count skipped keys.
243 StringList names;
244 for (auto it = specs.cbegin(); it != specs.cend(); ++it) {
245 const String &n = it->first.name();
246 if (skipKeys.contains(n)) continue;
247 names.pushToBack(n);
248 }
249 names = names.sort();
250 if (names.isEmpty()) return 0;
251
252 // Three-column layout: name | details | description.
253 // We cache the per-row details string so the width
254 // pass and the emit pass don't rebuild it twice.
255 List<String> details;
256 details.resize(names.size());
257 int nameWidth = 0;
258 int detailsWidth = 0;
259 for (size_t i = 0; i < names.size(); ++i) {
260 ID id(names[i]);
261 auto it = specs.find(id);
262 if (it == specs.end()) continue;
263 int nw = static_cast<int>(names[i].size());
264 if (nw > nameWidth) nameWidth = nw;
265 details[i] = it->second.detailsString();
266 int dw = static_cast<int>(details[i].size());
267 if (dw > detailsWidth) detailsWidth = dw;
268 }
269
270 // Fixed structure: " <name> <details> <desc>".
271 // Everything up to the description is a known
272 // size; the description's real width varies per
273 // row, so we track the widest actual line as we
274 // go and return that so the caller can draw a
275 // border matching the block.
276 const int prefixWidth = 2 + nameWidth + 2 + detailsWidth;
277 int maxLineWidth = 0;
278 for (size_t i = 0; i < names.size(); ++i) {
279 ID id(names[i]);
280 auto it = specs.find(id);
281 if (it == specs.end()) continue;
282
283 String nameCol = names[i];
284 while (static_cast<int>(nameCol.size()) < nameWidth) {
285 nameCol += ' ';
286 }
287 String detailsCol = details[i];
288 while (static_cast<int>(detailsCol.size()) < detailsWidth) {
289 detailsCol += ' ';
290 }
291 stream << " " << nameCol << " " << detailsCol;
292 const String &desc = it->second.description();
293 int lineWidth = prefixWidth;
294 if (!desc.isEmpty()) {
295 stream << " " << desc;
296 lineWidth += 2 + static_cast<int>(desc.size());
297 }
298 stream << endl;
299 if (lineWidth > maxLineWidth) maxLineWidth = lineWidth;
300 }
301 return maxLineWidth;
302 }

349 Error setFromJson(ID id, const Variant &value) {
350 const VariantSpec *sp = spec(id);
351 if (sp != nullptr) {
352 // Recursive coercion: top-level Strings get
353 // parsed via VariantSpec::parseString, and
354 // nested VariantList / VariantMap entries get
355 // walked with their element / value sub-spec
356 // so a {"color": "red"} JSON object whose
357 // declared spec is TypeVariantMap with a
358 // TypeColor valueSpec lands as
359 // VariantMap{"color": Color::Red} rather than
360 // VariantMap{"color": String("red")}.
361 Error ce;
362 Variant coerced = sp->coerce(value, &ce);
363 if (ce.isError()) {
364 // The leaf parse / recursion failed
365 // — surface that error rather than
366 // storing a value that would fail
367 // spec validation in Strict mode
368 // anyway.
369 return ce;
370 }
371 Error err;
372 set(id, std::move(coerced), &err);
373 return err;
374 }
375 Error err;
376 set(id, value, &err);
377 return err;
378 }

604 template <typename Resolver>
605 String format(const String &tmpl, Resolver &&resolver, Error *err = nullptr) const {
606 if (err != nullptr) *err = Error::Ok;
607 std::string out;
608 out.reserve(tmpl.byteCount());
609 const char *src = tmpl.cstr();
610 const size_t len = tmpl.byteCount();
611 bool sawUnresolved = false;
612 size_t i = 0;
613 while (i < len) {
614 char c = src[i];
615 if (c == '{') {
616 if (i + 1 < len && src[i + 1] == '{') {
617 out.push_back('{');
618 i += 2;
619 continue;
620 }
621 size_t end = i + 1;
622 while (end < len && src[end] != '}') ++end;
623 if (end >= len) {
624 out.append(src + i, len - i);
625 break;
626 }
627 std::string_view body(src + i + 1, end - (i + 1));
628 size_t colon = body.find(':');
629 std::string_view keyView =
630 (colon == std::string_view::npos) ? body : body.substr(0, colon);
631 std::string_view specView = (colon == std::string_view::npos)
632 ? std::string_view()
633 : body.substr(colon + 1);
634 String keyName(keyView.data(), keyView.size());
635 String specStr(specView.data(), specView.size());
636 // Split on the first '.' (or '[') to allow
637 // nested keys like "Foo.bar" or "Foo[0]"
638 // when the value at "Foo" is a VariantMap
639 // / VariantList. The leading ID
640 // component still must be a registered key.
641 size_t sep = keyName.byteCount();
642 const char *kn = keyName.cstr();
643 for (size_t p = 0; p < keyName.byteCount(); ++p) {
644 if (kn[p] == '.' || kn[p] == '[') {
645 sep = p;
646 break;
647 }
648 }
649 String headKey = (sep == keyName.byteCount()) ? keyName
650 : String(kn, sep);
651 String tailPath;
652 if (sep < keyName.byteCount()) {
653 // Trim a leading '.' but keep '['.
654 size_t start = (kn[sep] == '.') ? sep + 1 : sep;
655 tailPath = String(kn + start, keyName.byteCount() - start);
656 }
657 ID id = ID::find(headKey);
658 auto it = id.isValid() ? _d->data.find(id.id()) : _d->data.end();
659 if (it != _d->data.end()) {
660 Variant target = it->second;
661 if (!tailPath.isEmpty()) {
662 Error pe;
663 target = promekiResolveVariantPath(target, tailPath, &pe);
664 if (pe.isError() || !target.isValid()) {
665 sawUnresolved = true;
666 out += "[UNKNOWN KEY: ";
667 out.append(keyView.data(), keyView.size());
668 out += ']';
669 i = end + 1;
670 continue;
671 }
672 }
673 String rendered = target.format(specStr);
674 out.append(rendered.cstr(), rendered.byteCount());
675 } else {
676 std::optional<String> resolved;
677 if constexpr (!std::is_same_v<std::decay_t<Resolver>, std::nullptr_t>) {
678 resolved = resolver(keyName, specStr);
679 }
680 if (resolved.has_value()) {
681 const String &r = *resolved;
682 out.append(r.cstr(), r.byteCount());
683 } else {
684 sawUnresolved = true;
685 out += "[UNKNOWN KEY: ";
686 out.append(keyView.data(), keyView.size());
687 out += ']';
688 }
689 }
690 i = end + 1;
691 } else if (c == '}') {
692 if (i + 1 < len && src[i + 1] == '}') {
693 out.push_back('}');
694 i += 2;
695 } else {
696 out.push_back('}');
697 ++i;
698 }
699 } else {
700 out.push_back(c);
701 ++i;
702 }
703 }
704 if (sawUnresolved && err != nullptr) *err = Error::IdNotFound;
705 return String(std::move(out));
706 }

1036 bool validateOnSet(ID id, const Variant &value, Error *err = nullptr) {
1037 if (err) *err = Error::Ok;
1038 const SpecValidation mode = _d->validation;
1039 if (mode == SpecValidation::None) return true;
1040 const VariantSpec *s = specRegistry().find(id.id());
1041 if (!s) return true;

1049 if (err) *err = verr;
