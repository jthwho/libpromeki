198 static int writeSpecMapHelp(TextStream &stream, const SpecMap &specs,
199 const StringList &skipKeys = StringList()) {
200 // Collect the visible set first so the column
201 // width pass doesn't count skipped keys.
202 StringList names;
203 for(auto it = specs.cbegin(); it != specs.cend(); ++it) {
204 const String &n = it->first.name();
205 if(skipKeys.contains(n)) continue;
206 names.pushToBack(n);
207 }
208 names = names.sort();
209 if(names.isEmpty()) return 0;
210
211 // Three-column layout: name | details | description.
212 // We cache the per-row details string so the width
213 // pass and the emit pass don't rebuild it twice.
214 List<String> details;
215 details.resize(names.size());
216 int nameWidth = 0;
217 int detailsWidth = 0;
218 for(size_t i = 0; i < names.size(); ++i) {
219 ID id(names[i]);
220 auto it = specs.find(id);
221 if(it == specs.end()) continue;
222 int nw = static_cast<int>(names[i].size());
223 if(nw > nameWidth) nameWidth = nw;
224 details[i] = it->second.detailsString();
225 int dw = static_cast<int>(details[i].size());
226 if(dw > detailsWidth) detailsWidth = dw;
227 }
228
229 // Fixed structure: " <name> <details> <desc>".
230 // Everything up to the description is a known
231 // size; the description's real width varies per
232 // row, so we track the widest actual line as we
233 // go and return that so the caller can draw a
234 // border matching the block.
235 const int prefixWidth = 2 + nameWidth + 2 + detailsWidth;
236 int maxLineWidth = 0;
237 for(size_t i = 0; i < names.size(); ++i) {
238 ID id(names[i]);
239 auto it = specs.find(id);
240 if(it == specs.end()) continue;
241
242 String nameCol = names[i];
243 while(static_cast<int>(nameCol.size()) < nameWidth) {
244 nameCol += ' ';
245 }
246 String detailsCol = details[i];
247 while(static_cast<int>(detailsCol.size()) < detailsWidth) {
248 detailsCol += ' ';
249 }
250 stream << " " << nameCol << " " << detailsCol;
251 const String &desc = it->second.description();
252 int lineWidth = prefixWidth;
253 if(!desc.isEmpty()) {
254 stream << " " << desc;
255 lineWidth += 2 + static_cast<int>(desc.size());
256 }
257 stream << endl;
258 if(lineWidth > maxLineWidth) maxLineWidth = lineWidth;
259 }
260 return maxLineWidth;
261 }

536 template <typename Resolver>
537 String format(const String &tmpl, Resolver &&resolver, Error *err = nullptr) const {
538 if(err != nullptr) *err = Error::Ok;
539 std::string out;
540 out.reserve(tmpl.byteCount());
541 const char *src = tmpl.cstr();
542 const size_t len = tmpl.byteCount();
543 bool sawUnresolved = false;
544 size_t i = 0;
545 while(i < len) {
546 char c = src[i];
547 if(c == '{') {
548 if(i + 1 < len && src[i + 1] == '{') {
549 out.push_back('{');
550 i += 2;
551 continue;
552 }
553 size_t end = i + 1;
554 while(end < len && src[end] != '}') ++end;
555 if(end >= len) {
556 out.append(src + i, len - i);
557 break;
558 }
559 std::string_view body(src + i + 1, end - (i + 1));
560 size_t colon = body.find(':');
561 std::string_view keyView = (colon == std::string_view::npos)
562 ? body : body.substr(0, colon);
563 std::string_view specView = (colon == std::string_view::npos)
564 ? std::string_view() : body.substr(colon + 1);
565 String keyName(keyView.data(), keyView.size());
566 String specStr(specView.data(), specView.size());
567 ID id = ID::find(keyName);
568 auto it = id.isValid() ? _data.find(id.id()) : _data.end();
569 if(it != _data.end()) {
570 String rendered = it->second.format(specStr);
571 out.append(rendered.cstr(), rendered.byteCount());
572 } else {
573 std::optional<String> resolved;
574 if constexpr (!std::is_same_v<std::decay_t<Resolver>, std::nullptr_t>) {
575 resolved = resolver(keyName, specStr);
576 }
577 if(resolved.has_value()) {
578 const String &r = *resolved;
579 out.append(r.cstr(), r.byteCount());
580 } else {
581 sawUnresolved = true;
582 out += "[UNKNOWN KEY: ";
583 out.append(keyView.data(), keyView.size());
584 out += ']';
585 }
586 }
587 i = end + 1;
588 } else if(c == '}') {
589 if(i + 1 < len && src[i + 1] == '}') {
590 out.push_back('}');
591 i += 2;
592 } else {
593 out.push_back('}');
594 ++i;
595 }
596 } else {
597 out.push_back(c);
598 ++i;
599 }
600 }
601 if(sawUnresolved && err != nullptr) *err = Error::IdNotFound;
602 return String(std::move(out));
603 }

914 bool validateOnSet(ID id, const Variant &value) {
915 if(_validation == SpecValidation::None) return true;
916 const VariantSpec *s = specRegistry().find(id.id());
917 if(!s) return true;
918 Error err;
919 if(!s->validate(value, &err)) {
920 if(_validation == SpecValidation::Warn) {
921 promekiWarn("VariantDatabase: value for '%s' fails spec (%s)",
922 id.name().cstr(), err.name().cstr());
923 return true;
924 }
925 return false; // Strict: reject
926 }
927 return true;
