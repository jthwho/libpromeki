196 static int writeSpecMapHelp(TextStream &stream, const SpecMap &specs,
197 const StringList &skipKeys = StringList()) {
198 // Collect the visible set first so the column
199 // width pass doesn't count skipped keys.
200 StringList names;
201 for(auto it = specs.cbegin(); it != specs.cend(); ++it) {
202 const String &n = it->first.name();
203 if(skipKeys.contains(n)) continue;
204 names.pushToBack(n);
205 }
206 names = names.sort();
207 if(names.isEmpty()) return 0;
208
209 // Three-column layout: name | details | description.
210 // We cache the per-row details string so the width
211 // pass and the emit pass don't rebuild it twice.
212 List<String> details;
213 details.resize(names.size());
214 int nameWidth = 0;
215 int detailsWidth = 0;
216 for(size_t i = 0; i < names.size(); ++i) {
217 ID id(names[i]);
218 auto it = specs.find(id);
219 if(it == specs.end()) continue;
220 int nw = static_cast<int>(names[i].size());
221 if(nw > nameWidth) nameWidth = nw;
222 details[i] = it->second.detailsString();
223 int dw = static_cast<int>(details[i].size());
224 if(dw > detailsWidth) detailsWidth = dw;
225 }
226
227 // Fixed structure: " <name> <details> <desc>".
228 // Everything up to the description is a known
229 // size; the description's real width varies per
230 // row, so we track the widest actual line as we
231 // go and return that so the caller can draw a
232 // border matching the block.
233 const int prefixWidth = 2 + nameWidth + 2 + detailsWidth;
234 int maxLineWidth = 0;
235 for(size_t i = 0; i < names.size(); ++i) {
236 ID id(names[i]);
237 auto it = specs.find(id);
238 if(it == specs.end()) continue;
239
240 String nameCol = names[i];
241 while(static_cast<int>(nameCol.size()) < nameWidth) {
242 nameCol += ' ';
243 }
244 String detailsCol = details[i];
245 while(static_cast<int>(detailsCol.size()) < detailsWidth) {
246 detailsCol += ' ';
247 }
248 stream << " " << nameCol << " " << detailsCol;
249 const String &desc = it->second.description();
250 int lineWidth = prefixWidth;
251 if(!desc.isEmpty()) {
252 stream << " " << desc;
253 lineWidth += 2 + static_cast<int>(desc.size());
254 }
255 stream << endl;
256 if(lineWidth > maxLineWidth) maxLineWidth = lineWidth;
257 }
258 return maxLineWidth;
259 }

709 bool validateOnSet(ID id, const Variant &value) {
710 if(_validation == SpecValidation::None) return true;
711 const VariantSpec *s = specRegistry().find(id.id());
712 if(!s) return true;
713 Error err;
714 if(!s->validate(value, &err)) {
715 if(_validation == SpecValidation::Warn) {
716 promekiWarn("VariantDatabase: value for '%s' fails spec (%s)",
717 id.name().cstr(), err.name().cstr());
718 return true;
719 }
720 return false; // Strict: reject
721 }
722 return true;
723 }
