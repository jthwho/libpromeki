226 static int writeSpecMapHelp(TextStream &stream, const SpecMap &specs,
227 const StringList &skipKeys = StringList()) {
228 // Collect the visible set first so the column
229 // width pass doesn't count skipped keys.
230 StringList names;
231 for (auto it = specs.cbegin(); it != specs.cend(); ++it) {
232 const String &n = it->first.name();
233 if (skipKeys.contains(n)) continue;
234 names.pushToBack(n);
235 }
236 names = names.sort();
237 if (names.isEmpty()) return 0;
238
239 // Three-column layout: name | details | description.
240 // We cache the per-row details string so the width
241 // pass and the emit pass don't rebuild it twice.
242 List<String> details;
243 details.resize(names.size());
244 int nameWidth = 0;
245 int detailsWidth = 0;
246 for (size_t i = 0; i < names.size(); ++i) {
247 ID id(names[i]);
248 auto it = specs.find(id);
249 if (it == specs.end()) continue;
250 int nw = static_cast<int>(names[i].size());
251 if (nw > nameWidth) nameWidth = nw;
252 details[i] = it->second.detailsString();
253 int dw = static_cast<int>(details[i].size());
254 if (dw > detailsWidth) detailsWidth = dw;
255 }
256
257 // Fixed structure: " <name> <details> <desc>".
258 // Everything up to the description is a known
259 // size; the description's real width varies per
260 // row, so we track the widest actual line as we
261 // go and return that so the caller can draw a
262 // border matching the block.
263 const int prefixWidth = 2 + nameWidth + 2 + detailsWidth;
264 int maxLineWidth = 0;
265 for (size_t i = 0; i < names.size(); ++i) {
266 ID id(names[i]);
267 auto it = specs.find(id);
268 if (it == specs.end()) continue;
269
270 String nameCol = names[i];
271 while (static_cast<int>(nameCol.size()) < nameWidth) {
272 nameCol += ' ';
273 }
274 String detailsCol = details[i];
275 while (static_cast<int>(detailsCol.size()) < detailsWidth) {
276 detailsCol += ' ';
277 }
278 stream << " " << nameCol << " " << detailsCol;
279 const String &desc = it->second.description();
280 int lineWidth = prefixWidth;
281 if (!desc.isEmpty()) {
282 stream << " " << desc;
283 lineWidth += 2 + static_cast<int>(desc.size());
284 }
285 stream << endl;
286 if (lineWidth > maxLineWidth) maxLineWidth = lineWidth;
287 }
288 return maxLineWidth;
289 }

336 Error setFromJson(ID id, const Variant &value) {
337 const VariantSpec *sp = spec(id);
338 if (sp != nullptr) {
339 // Recursive coercion: top-level Strings get
340 // parsed via VariantSpec::parseString, and
341 // nested VariantList / VariantMap entries get
342 // walked with their element / value sub-spec
343 // so a {"color": "red"} JSON object whose
344 // declared spec is TypeVariantMap with a
345 // TypeColor valueSpec lands as
346 // VariantMap{"color": Color::Red} rather than
347 // VariantMap{"color": String("red")}.
348 Error ce;
349 Variant coerced = sp->coerce(value, &ce);
350 if (ce.isError()) {
351 // The leaf parse / recursion failed
352 // — surface that error rather than
353 // storing a value that would fail
354 // spec validation in Strict mode
355 // anyway.
356 return ce;
357 }
358 Error err;
359 set(id, std::move(coerced), &err);
360 return err;
361 }
362 Error err;
363 set(id, value, &err);
364 return err;
365 }

591 template <typename Resolver>
592 String format(const String &tmpl, Resolver &&resolver, Error *err = nullptr) const {
593 if (err != nullptr) *err = Error::Ok;
594 std::string out;
595 out.reserve(tmpl.byteCount());
596 const char *src = tmpl.cstr();
597 const size_t len = tmpl.byteCount();
598 bool sawUnresolved = false;
599 size_t i = 0;
600 while (i < len) {
601 char c = src[i];
602 if (c == '{') {
603 if (i + 1 < len && src[i + 1] == '{') {
604 out.push_back('{');
605 i += 2;
606 continue;
607 }
608 size_t end = i + 1;
609 while (end < len && src[end] != '}') ++end;
610 if (end >= len) {
611 out.append(src + i, len - i);
612 break;
613 }
614 std::string_view body(src + i + 1, end - (i + 1));
615 size_t colon = body.find(':');
616 std::string_view keyView =
617 (colon == std::string_view::npos) ? body : body.substr(0, colon);
618 std::string_view specView = (colon == std::string_view::npos)
619 ? std::string_view()
620 : body.substr(colon + 1);
621 String keyName(keyView.data(), keyView.size());
622 String specStr(specView.data(), specView.size());
623 // Split on the first '.' (or '[') to allow
624 // nested keys like "Foo.bar" or "Foo[0]"
625 // when the value at "Foo" is a VariantMap
626 // / VariantList. The leading ID
627 // component still must be a registered key.
628 size_t sep = keyName.byteCount();
629 const char *kn = keyName.cstr();
630 for (size_t p = 0; p < keyName.byteCount(); ++p) {
631 if (kn[p] == '.' || kn[p] == '[') {
632 sep = p;
633 break;
634 }
635 }
636 String headKey = (sep == keyName.byteCount()) ? keyName
637 : String(kn, sep);
638 String tailPath;
639 if (sep < keyName.byteCount()) {
640 // Trim a leading '.' but keep '['.
641 size_t start = (kn[sep] == '.') ? sep + 1 : sep;
642 tailPath = String(kn + start, keyName.byteCount() - start);
643 }
644 ID id = ID::find(headKey);
645 auto it = id.isValid() ? _data.find(id.id()) : _data.end();
646 if (it != _data.end()) {
647 Variant target = it->second;
648 if (!tailPath.isEmpty()) {
649 Error pe;
650 target = promekiResolveVariantPath(target, tailPath, &pe);
651 if (pe.isError() || !target.isValid()) {
652 sawUnresolved = true;
653 out += "[UNKNOWN KEY: ";
654 out.append(keyView.data(), keyView.size());
655 out += ']';
656 i = end + 1;
657 continue;
658 }
659 }
660 String rendered = target.format(specStr);
661 out.append(rendered.cstr(), rendered.byteCount());
662 } else {
663 std::optional<String> resolved;
664 if constexpr (!std::is_same_v<std::decay_t<Resolver>, std::nullptr_t>) {
665 resolved = resolver(keyName, specStr);
666 }
667 if (resolved.has_value()) {
668 const String &r = *resolved;
669 out.append(r.cstr(), r.byteCount());
670 } else {
671 sawUnresolved = true;
672 out += "[UNKNOWN KEY: ";
673 out.append(keyView.data(), keyView.size());
674 out += ']';
675 }
676 }
677 i = end + 1;
678 } else if (c == '}') {
679 if (i + 1 < len && src[i + 1] == '}') {
680 out.push_back('}');
681 i += 2;
682 } else {
683 out.push_back('}');
684 ++i;
685 }
686 } else {
687 out.push_back(c);
688 ++i;
689 }
690 }
691 if (sawUnresolved && err != nullptr) *err = Error::IdNotFound;
692 return String(std::move(out));
693 }

999 bool validateOnSet(ID id, const Variant &value, Error *err = nullptr) {
1000 if (err) *err = Error::Ok;
1001 if (_validation == SpecValidation::None) return true;
1002 const VariantSpec *s = specRegistry().find(id.id());
1003 if (!s) return true;
1004 Error verr;
1005 if (!s->validate(value, &verr)) {
1006 if (_validation == SpecValidation::Warn) {
1007 promekiWarn("VariantDatabase: value for '%s' fails spec (%s)", id.name().cstr(),
1008 verr.name().cstr());
1009 return true;
1010 }
1011 if (err) *err = verr;
1012 return false; // Strict: reject
1013 }
1014 return true;
1015 }
