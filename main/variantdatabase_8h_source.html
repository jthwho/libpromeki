593 bool validateOnSet(ID id, const Variant &value) {
594 if(_validation == SpecValidation::None) return true;
595 const VariantSpec *s = specRegistry().find(id.id());
596 if(!s) return true;
597 Error err;
598 if(!s->validate(value, &err)) {
599 if(_validation == SpecValidation::Warn) {
600 promekiWarn("VariantDatabase: value for '%s' fails spec (%s)",
601 id.name().cstr(), err.name().cstr());
602 return true;
603 }
604 return false; // Strict: reject
605 }
606 return true;
607 }
