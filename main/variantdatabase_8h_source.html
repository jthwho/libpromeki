177 static int writeSpecMapHelp(TextStream &stream, const SpecMap &specs,
178 const StringList &skipKeys = StringList()) {
179 // Collect the visible set first so the column
180 // width pass doesn't count skipped keys.
181 StringList names;
182 for(auto it = specs.cbegin(); it != specs.cend(); ++it) {
183 const String &n = it->first.name();
184 if(skipKeys.contains(n)) continue;
185 names.pushToBack(n);
186 }
187 names = names.sort();
188 if(names.isEmpty()) return 0;
189
190 // Three-column layout: name | details | description.
191 // We cache the per-row details string so the width
192 // pass and the emit pass don't rebuild it twice.
193 List<String> details;
194 details.resize(names.size());
195 int nameWidth = 0;
196 int detailsWidth = 0;
197 for(size_t i = 0; i < names.size(); ++i) {
198 ID id(names[i]);
199 auto it = specs.find(id);
200 if(it == specs.end()) continue;
201 int nw = static_cast<int>(names[i].size());
202 if(nw > nameWidth) nameWidth = nw;
203 details[i] = it->second.detailsString();
204 int dw = static_cast<int>(details[i].size());
205 if(dw > detailsWidth) detailsWidth = dw;
206 }
207
208 // Fixed structure: " <name> <details> <desc>".
209 // Everything up to the description is a known
210 // size; the description's real width varies per
211 // row, so we track the widest actual line as we
212 // go and return that so the caller can draw a
213 // border matching the block.
214 const int prefixWidth = 2 + nameWidth + 2 + detailsWidth;
215 int maxLineWidth = 0;
216 for(size_t i = 0; i < names.size(); ++i) {
217 ID id(names[i]);
218 auto it = specs.find(id);
219 if(it == specs.end()) continue;
220
221 String nameCol = names[i];
222 while(static_cast<int>(nameCol.size()) < nameWidth) {
223 nameCol += ' ';
224 }
225 String detailsCol = details[i];
226 while(static_cast<int>(detailsCol.size()) < detailsWidth) {
227 detailsCol += ' ';
228 }
229 stream << " " << nameCol << " " << detailsCol;
230 const String &desc = it->second.description();
231 int lineWidth = prefixWidth;
232 if(!desc.isEmpty()) {
233 stream << " " << desc;
234 lineWidth += 2 + static_cast<int>(desc.size());
235 }
236 stream << endl;
237 if(lineWidth > maxLineWidth) maxLineWidth = lineWidth;
238 }
239 return maxLineWidth;
240 }

690 bool validateOnSet(ID id, const Variant &value) {
691 if(_validation == SpecValidation::None) return true;
692 const VariantSpec *s = specRegistry().find(id.id());
693 if(!s) return true;
694 Error err;
695 if(!s->validate(value, &err)) {
696 if(_validation == SpecValidation::Warn) {
697 promekiWarn("VariantDatabase: value for '%s' fails spec (%s)",
698 id.name().cstr(), err.name().cstr());
699 return true;
700 }
701 return false; // Strict: reject
702 }
703 return true;
704 }
