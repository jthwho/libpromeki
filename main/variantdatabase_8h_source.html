220 static int writeSpecMapHelp(TextStream &stream, const SpecMap &specs,
221 const StringList &skipKeys = StringList()) {
222 // Collect the visible set first so the column
223 // width pass doesn't count skipped keys.
224 StringList names;
225 for(auto it = specs.cbegin(); it != specs.cend(); ++it) {
226 const String &n = it->first.name();
227 if(skipKeys.contains(n)) continue;
228 names.pushToBack(n);
229 }
230 names = names.sort();
231 if(names.isEmpty()) return 0;
232
233 // Three-column layout: name | details | description.
234 // We cache the per-row details string so the width
235 // pass and the emit pass don't rebuild it twice.
236 List<String> details;
237 details.resize(names.size());
238 int nameWidth = 0;
239 int detailsWidth = 0;
240 for(size_t i = 0; i < names.size(); ++i) {
241 ID id(names[i]);
242 auto it = specs.find(id);
243 if(it == specs.end()) continue;
244 int nw = static_cast<int>(names[i].size());
245 if(nw > nameWidth) nameWidth = nw;
246 details[i] = it->second.detailsString();
247 int dw = static_cast<int>(details[i].size());
248 if(dw > detailsWidth) detailsWidth = dw;
249 }
250
251 // Fixed structure: " <name> <details> <desc>".
252 // Everything up to the description is a known
253 // size; the description's real width varies per
254 // row, so we track the widest actual line as we
255 // go and return that so the caller can draw a
256 // border matching the block.
257 const int prefixWidth = 2 + nameWidth + 2 + detailsWidth;
258 int maxLineWidth = 0;
259 for(size_t i = 0; i < names.size(); ++i) {
260 ID id(names[i]);
261 auto it = specs.find(id);
262 if(it == specs.end()) continue;
263
264 String nameCol = names[i];
265 while(static_cast<int>(nameCol.size()) < nameWidth) {
266 nameCol += ' ';
267 }
268 String detailsCol = details[i];
269 while(static_cast<int>(detailsCol.size()) < detailsWidth) {
270 detailsCol += ' ';
271 }
272 stream << " " << nameCol << " " << detailsCol;
273 const String &desc = it->second.description();
274 int lineWidth = prefixWidth;
275 if(!desc.isEmpty()) {
276 stream << " " << desc;
277 lineWidth += 2 + static_cast<int>(desc.size());
278 }
279 stream << endl;
280 if(lineWidth > maxLineWidth) maxLineWidth = lineWidth;
281 }
282 return maxLineWidth;
283 }

558 template <typename Resolver>
559 String format(const String &tmpl, Resolver &&resolver, Error *err = nullptr) const {
560 if(err != nullptr) *err = Error::Ok;
561 std::string out;
562 out.reserve(tmpl.byteCount());
563 const char *src = tmpl.cstr();
564 const size_t len = tmpl.byteCount();
565 bool sawUnresolved = false;
566 size_t i = 0;
567 while(i < len) {
568 char c = src[i];
569 if(c == '{') {
570 if(i + 1 < len && src[i + 1] == '{') {
571 out.push_back('{');
572 i += 2;
573 continue;
574 }
575 size_t end = i + 1;
576 while(end < len && src[end] != '}') ++end;
577 if(end >= len) {
578 out.append(src + i, len - i);
579 break;
580 }
581 std::string_view body(src + i + 1, end - (i + 1));
582 size_t colon = body.find(':');
583 std::string_view keyView = (colon == std::string_view::npos)
584 ? body : body.substr(0, colon);
585 std::string_view specView = (colon == std::string_view::npos)
586 ? std::string_view() : body.substr(colon + 1);
587 String keyName(keyView.data(), keyView.size());
588 String specStr(specView.data(), specView.size());
589 ID id = ID::find(keyName);
590 auto it = id.isValid() ? _data.find(id.id()) : _data.end();
591 if(it != _data.end()) {
592 String rendered = it->second.format(specStr);
593 out.append(rendered.cstr(), rendered.byteCount());
594 } else {
595 std::optional<String> resolved;
596 if constexpr (!std::is_same_v<std::decay_t<Resolver>, std::nullptr_t>) {
597 resolved = resolver(keyName, specStr);
598 }
599 if(resolved.has_value()) {
600 const String &r = *resolved;
601 out.append(r.cstr(), r.byteCount());
602 } else {
603 sawUnresolved = true;
604 out += "[UNKNOWN KEY: ";
605 out.append(keyView.data(), keyView.size());
606 out += ']';
607 }
608 }
609 i = end + 1;
610 } else if(c == '}') {
611 if(i + 1 < len && src[i + 1] == '}') {
612 out.push_back('}');
613 i += 2;
614 } else {
615 out.push_back('}');
616 ++i;
617 }
618 } else {
619 out.push_back(c);
620 ++i;
621 }
622 }
623 if(sawUnresolved && err != nullptr) *err = Error::IdNotFound;
624 return String(std::move(out));
625 }

936 bool validateOnSet(ID id, const Variant &value) {
937 if(_validation == SpecValidation::None) return true;
938 const VariantSpec *s = specRegistry().find(id.id());
939 if(!s) return true;
940 Error err;
