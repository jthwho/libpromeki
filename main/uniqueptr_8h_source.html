261template<typename Derived, typename Base>
262UniquePtr<Derived> uniquePointerCast(UniquePtr<Base> &&up) {
263 static_assert(std::is_base_of_v<Base, Derived>,
264 "Derived must publicly derive from Base");
265 if(up.isNull()) return UniquePtr<Derived>();
266 Derived *d = dynamic_cast<Derived *>(up.ptr());
267 if(d == nullptr) return UniquePtr<Derived>();
268 up.release();
269 return UniquePtr<Derived>::takeOwnership(d);
