1896template <typename Fn>
1897Error MediaIO::copyFrames(MediaIO *src, MediaIO *dst,
1898 int64_t fromFrame, int64_t count,
1899 Fn &&mutate, int64_t *copied) {
1900 if(copied != nullptr) *copied = 0;
1901 if(src == nullptr || dst == nullptr) return Error::InvalidArgument;
1902
1903 if(fromFrame > 0) {
1904 Error se = copyFramesSeekTo(src, fromFrame);
1905 if(se == Error::EndOfFile) return Error::Ok;
1906 if(se.isError()) return se;
1907 }
1908
1909 // remaining < 0 encodes the infinite-copy case; the
1910 // "if(remaining > 0) --remaining" branch below never fires in
1911 // that mode, so the loop runs until the source returns EOF.
1912 int64_t remaining = (count == FrameCountInfinite) ? -1 : count;
1913 int64_t written = 0;
1914 int64_t index = fromFrame;
1915
1916 while(remaining != 0) {
1917 Frame::Ptr frame;
1918 Error rerr = src->readFrame(frame);
1919 if(rerr == Error::EndOfFile) break;
1920 if(rerr.isError()) return rerr;
1921 if(!frame.isValid()) {
1922 if(remaining > 0) --remaining;
1923 ++index;
1924 continue;
1925 }
1926
1927 Frame::Ptr outFrame = mutate(frame, index);
1928 if(outFrame.isValid()) {
1929 Error werr = dst->writeFrame(outFrame);
1930 if(werr.isError()) return werr;
1931 ++written;
1932 }
1933 ++index;
1934 if(remaining > 0) --remaining;
1935 }
1936
1937 if(copied != nullptr) *copied = written;
1938 return Error::Ok;
1939}
