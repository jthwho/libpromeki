2076template <typename Fn>
2077Result<FrameCount> MediaIO::copyFrames(MediaIO *src, MediaIO *dst,
2078 const FrameNumber &fromFrame, const FrameCount &count,
2079 Fn &&mutate) {
2080 if(src == nullptr || dst == nullptr) return makeError<FrameCount>(Error::InvalidArgument);
2081
2082 if(fromFrame.isValid() && fromFrame.value() > 0) {
2083 Error se = copyFramesSeekTo(src, fromFrame);
2084 if(se == Error::EndOfFile) return makeResult(FrameCount(0));
2085 if(se.isError()) return makeError<FrameCount>(se);
2086 }
2087
2088 // remaining < 0 encodes the infinite-copy case; the
2089 // "if(remaining > 0) --remaining" branch below never fires in
2090 // that mode, so the loop runs until the source returns EOF.
2091 int64_t remaining = count.isInfinite() ? -1 : count.value();
2092 int64_t written = 0;
2093 int64_t index = fromFrame.isValid() ? fromFrame.value() : 0;
2094
2095 while(remaining != 0) {
2096 Frame::Ptr frame;
2097 Error rerr = src->readFrame(frame);
2098 if(rerr == Error::EndOfFile) break;
2099 if(rerr.isError()) return makeError<FrameCount>(rerr);
2100 if(!frame.isValid()) {
2101 if(remaining > 0) --remaining;
2102 ++index;
2103 continue;
2104 }
2105
2106 Frame::Ptr outFrame = mutate(frame, index);
2107 if(outFrame.isValid()) {
2108 Error werr = dst->writeFrame(outFrame);
2109 if(werr.isError()) {
2110 Result<FrameCount> r = makeResult(FrameCount(written));
2111 r.second() = werr;
2112 return r;
2113 }
2114 ++written;
2115 }
2116 ++index;
2117 if(remaining > 0) --remaining;
2118 }
2119
2120 return makeResult(FrameCount(written));
2121}
