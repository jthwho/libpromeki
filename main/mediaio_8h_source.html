1919template <typename Fn>
1920Result<FrameCount> MediaIO::copyFrames(MediaIO *src, MediaIO *dst,
1921 const FrameNumber &fromFrame, const FrameCount &count,
1922 Fn &&mutate) {
1923 if(src == nullptr || dst == nullptr) return makeError<FrameCount>(Error::InvalidArgument);
1924
1925 if(fromFrame.isValid() && fromFrame.value() > 0) {
1926 Error se = copyFramesSeekTo(src, fromFrame);
1927 if(se == Error::EndOfFile) return makeResult(FrameCount(0));
1928 if(se.isError()) return makeError<FrameCount>(se);
1929 }
1930
1931 // remaining < 0 encodes the infinite-copy case; the
1932 // "if(remaining > 0) --remaining" branch below never fires in
1933 // that mode, so the loop runs until the source returns EOF.
1934 int64_t remaining = count.isInfinite() ? -1 : count.value();
1935 int64_t written = 0;
1936 int64_t index = fromFrame.isValid() ? fromFrame.value() : 0;
1937
1938 while(remaining != 0) {
1939 Frame::Ptr frame;
1940 Error rerr = src->readFrame(frame);
1941 if(rerr == Error::EndOfFile) break;
1942 if(rerr.isError()) return makeError<FrameCount>(rerr);
1943 if(!frame.isValid()) {
1944 if(remaining > 0) --remaining;
1945 ++index;
1946 continue;
1947 }
1948
1949 Frame::Ptr outFrame = mutate(frame, index);
1950 if(outFrame.isValid()) {
1951 Error werr = dst->writeFrame(outFrame);
1952 if(werr.isError()) {
1953 Result<FrameCount> r = makeResult(FrameCount(written));
1954 r.second() = werr;
1955 return r;
1956 }
1957 ++written;
1958 }
1959 ++index;
1960 if(remaining > 0) --remaining;
1961 }
1962
1963 return makeResult(FrameCount(written));
1964}
