2080template <typename Fn>
2081Result<FrameCount> MediaIO::copyFrames(MediaIO *src, MediaIO *dst,
2082 const FrameNumber &fromFrame, const FrameCount &count,
2083 Fn &&mutate) {
2084 if(src == nullptr || dst == nullptr) return makeError<FrameCount>(Error::InvalidArgument);
2085
2086 if(fromFrame.isValid() && fromFrame.value() > 0) {
2087 Error se = copyFramesSeekTo(src, fromFrame);
2088 if(se == Error::EndOfFile) return makeResult(FrameCount(0));
2089 if(se.isError()) return makeError<FrameCount>(se);
2090 }
2091
2092 // remaining < 0 encodes the infinite-copy case; the
2093 // "if(remaining > 0) --remaining" branch below never fires in
2094 // that mode, so the loop runs until the source returns EOF.
2095 int64_t remaining = count.isInfinite() ? -1 : count.value();
2096 int64_t written = 0;
2097 int64_t index = fromFrame.isValid() ? fromFrame.value() : 0;
2098
2099 while(remaining != 0) {
2100 Frame::Ptr frame;
2101 Error rerr = src->readFrame(frame);
2102 if(rerr == Error::EndOfFile) break;
2103 if(rerr.isError()) return makeError<FrameCount>(rerr);
2104 if(!frame.isValid()) {
2105 if(remaining > 0) --remaining;
2106 ++index;
2107 continue;
2108 }
2109
2110 Frame::Ptr outFrame = mutate(frame, index);
2111 if(outFrame.isValid()) {
2112 Error werr = dst->writeFrame(outFrame);
2113 if(werr.isError()) {
2114 Result<FrameCount> r = makeResult(FrameCount(written));
2115 r.second() = werr;
2116 return r;
2117 }
2118 ++written;
2119 }
2120 ++index;
2121 if(remaining > 0) --remaining;
2122 }
2123
2124 return makeResult(FrameCount(written));
2125}
