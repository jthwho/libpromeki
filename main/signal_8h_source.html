260 void emit(Args... args) const {
261 // Snapshot under the lock so iteration runs
262 // against a stable copy. Concurrent
263 // connect / disconnect calls only mutate the
264 // live @c _slots — the snapshot is unaffected,
265 // which keeps reentrant connect / disconnect
266 // from a slot safe and matches Boost.Signals2
267 // semantics.
268 List<Info> snapshot;
269 {
270 Mutex::Locker lock(_slotsMutex);
271 snapshot = _slots;
272 }
273 for (const auto &slot : snapshot) slot.func(args...);
