281 uint64_t findOrCreateProbe(const String &str) {
282 uint64_t start = fnv1a(str.cstr());
283 // Fast path: read-locked probe.
284 {
285 ReadWriteLock::ReadLocker lock(_lock);
286 uint64_t slot = start;
287 while(true) {
288 auto it = _names.find(slot);
289 if(it == _names.end()) break;
290 if(it->second == str) return slot;
291 ++slot;
292 }
293 }
294 // Slow path: acquire write lock and retry so we can
295 // safely insert if still missing.
296 ReadWriteLock::WriteLocker lock(_lock);
297 uint64_t slot = start;
298 while(true) {
299 auto it = _names.find(slot);
300 if(it == _names.end()) {
301 _names.insert(slot, str);
302 return slot;
303 }
304 if(it->second == str) return slot;
305 ++slot;
306 }
307 }
