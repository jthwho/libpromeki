263 uint64_t findOrCreateStrict(const String &str) {
264 uint64_t h = fnv1a(str.cstr());
265 // Fast path: check under a read lock first.
266 {
267 ReadWriteLock::ReadLocker lock(_lock);
268 auto it = _names.find(h);

296 uint64_t findOrCreateProbe(const String &str) {
297 uint64_t start = fnv1a(str.cstr());
298 // Fast path: read-locked probe.
299 {
300 ReadWriteLock::ReadLocker lock(_lock);
301 uint64_t slot = start;
302 while (true) {
303 auto it = _names.find(slot);
304 if (it == _names.end()) break;
305 if (it->second == str) return slot;
306 ++slot;
307 }
308 }
309 // Slow path: acquire write lock and retry so we can
310 // safely insert if still missing.
311 ReadWriteLock::WriteLocker lock(_lock);
312 uint64_t slot = start;
313 while (true) {
314 auto it = _names.find(slot);
315 if (it == _names.end()) {
316 _names.insert(slot, str);
317 return slot;
318 }
319 if (it->second == str) return slot;
320 ++slot;
321 }
322 }
