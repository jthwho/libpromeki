314template <CompiledString N>
315void HttpServer::exposeDatabase(const String &mountPath, VariantDatabase<N> &db, bool readOnly) {
316 using DB = VariantDatabase<N>;
317 using ID = typename DB::ID;
318
319 // Full snapshot.
320 route(mountPath, HttpMethod::Get, [&db](const HttpRequest &, HttpResponse &res) { res.setJson(db.toJson()); });
321
322 // Schema introspection. Keys not yet present in the
323 // database still appear here when a spec was declared at
324 // static-init time.
325 const String schemaPath = mountPath + "/_schema";
326 route(schemaPath, HttpMethod::Get, [](const HttpRequest &, HttpResponse &res) {
327 JsonObject out;
328 const auto specs = DB::registeredSpecs();
329 for (auto it = specs.cbegin(); it != specs.cend(); ++it) {
330 const String name = ID::fromId(it->first).name();
331 out.set(name, HttpServer::specToJson(it->second));
332 }
333 res.setJson(out);
334 });
335
336 // Per-key get.
337 const String keyPath = mountPath + "/{key}";
338 route(keyPath, HttpMethod::Get, [&db](const HttpRequest &req, HttpResponse &res) {
339 const String name = req.pathParam("key");
340 ID id = ID::find(name);
341 if (!id.isValid() || !db.contains(id)) {
342 res = HttpResponse::notFound(String("Unknown key: ") + name);
343 return;
344 }
345 JsonObject out;
346 out.setFromVariant(name, db.get(id));
347 const VariantSpec *sp = DB::spec(id);
348 if (sp != nullptr) {
349 out.set("_spec", HttpServer::specToJson(*sp));
350 }
351 res.setJson(out);
352 });
353
354 if (readOnly) return;
355
356 // Per-key set.
357 route(keyPath, HttpMethod::Put, [&db](const HttpRequest &req, HttpResponse &res) {
358 const String name = req.pathParam("key");
359 Error perr;
360 JsonObject body = req.bodyAsJson(&perr);
361 if (perr.isError() || !body.contains("value")) {
362 res = HttpResponse::badRequest("Body must be JSON object with a \"value\" field");
363 return;
364 }
365 // Snapshot the value via JsonObject::forEach,
366 // which converts the underlying nlohmann json
367 // into a typed @ref Variant — that's the path
368 // that setFromJson knows how to coerce
369 // against the registered spec.
370 ID id(name);
371 Variant captured;
372 body.forEach([&](const String &k, const Variant &val) {
373 if (k == "value") captured = val;
374 });
375 Error serr = db.setFromJson(id, captured);
376 if (serr.isError()) {
377 res = HttpResponse::badRequest(String("Validation failed for key '") + name +
378 "': " + serr.desc());
379 return;
380 }
381 JsonObject out;
382 out.setFromVariant(name, db.get(id));
383 res.setJson(out);
384 });
385
386 // Per-key delete.
387 route(keyPath, HttpMethod::Delete, [&db](const HttpRequest &req, HttpResponse &res) {
388 const String name = req.pathParam("key");
389 ID id = ID::find(name);
390 if (!id.isValid() || !db.contains(id)) {
391 res = HttpResponse::notFound(String("Unknown key: ") + name);
392 return;
393 }
394 db.remove(id);
395 res = HttpResponse::noContent();
396 });
397}

399template <typename T> void HttpServer::exposeLookup(const String &mountPath, T &target) {
400 const String greedy = mountPath + "/{path:*}";
401 route(greedy, HttpMethod::Get, [&target](const HttpRequest &req, HttpResponse &res) {
402 const String key = HttpServer::lookupPathToKey(req.pathParam("path"));
403 if (key.isEmpty()) {
404 res = HttpResponse::badRequest("Empty lookup key");
405 return;
406 }
407 Error err;
408 auto v = VariantLookup<T>::resolve(target, key, &err);
409 if (!v.has_value() || err.isError()) {
410 res = HttpResponse::notFound(String("Unknown lookup: ") + key);
411 return;
412 }
413 JsonObject out;
414 out.setFromVariant("value", *v);
415 res.setJson(out);
416 });
417}
