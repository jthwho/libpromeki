495template <> struct std::formatter<promeki::Timecode> {
496 enum class Style {
497 Smpte,
498 SmpteWithFps,
499 SmpteSpaceFps,
500 Field
501 };
502
503 Style _style = Style::Smpte;
504 std::formatter<std::string_view> _base;
505
506 constexpr auto parse(std::format_parse_context &ctx) {
507 auto it = ctx.begin();
508 auto end = ctx.end();
509
510 // Try to match a recognised style keyword at the start of
511 // the spec. Each candidate is checked in longest-first
512 // order so prefixes do not collide ("smpte-fps" must beat
513 // "smpte"). On match, advance past the keyword.
514 auto tryKeyword = [&](const char *kw, Style s) {
515 auto p = it;
516 while (*kw && p != end && *p == *kw) {
517 ++p;
518 ++kw;
519 }
520 if (*kw == 0 && (p == end || *p == '}' || *p == ':')) {
521 it = p;
522 _style = s;
523 return true;
524 }
525 return false;
526 };
527
528 if (!tryKeyword("smpte-fps", Style::SmpteWithFps) &&
529 !tryKeyword("smpte-space", Style::SmpteSpaceFps) && !tryKeyword("smpte", Style::Smpte) &&
530 !tryKeyword("field", Style::Field)) {
531 // No keyword — leave _style at its default and let
532 // the base parser consume the entire remaining spec.
533 }
534
535 // A separating ':' between the hint and a standard string
536 // format spec is consumed here so the base parser sees a
537 // bare ">16" rather than ":>16".
538 if (it != end && *it == ':') ++it;
539
540 // Forward whatever remains to the standard string format
541 // parser so width / fill / alignment / precision still work.
542 ctx.advance_to(it);
543 return _base.parse(ctx);
544 }
545
546 template <typename FormatContext> auto format(const promeki::Timecode &tc, FormatContext &ctx) const {
547 const VtcStringFormat *fmt = &VTC_STR_FMT_SMPTE;
548 switch (_style) {
549 case Style::Smpte: fmt = &VTC_STR_FMT_SMPTE; break;
550 case Style::SmpteWithFps: fmt = &VTC_STR_FMT_SMPTE_WITH_FPS; break;
551 case Style::SmpteSpaceFps: fmt = &VTC_STR_FMT_SMPTE_SPACE_FPS; break;
552 case Style::Field: fmt = &VTC_STR_FMT_FIELD; break;
553 }
554 auto [s, err] = tc.toString(fmt);
555 (void)err; // Formatter consumers do not get the error path.
556 return _base.format(std::string_view(s.cstr(), s.byteCount()), ctx);
557 }
