462 Style _style = Style::Smpte;
463 std::formatter<std::string_view> _base;
464
465 constexpr auto parse(std::format_parse_context &ctx) {
466 auto it = ctx.begin();
467 auto end = ctx.end();
468
469 // Try to match a recognised style keyword at the start of
470 // the spec. Each candidate is checked in longest-first
471 // order so prefixes do not collide ("smpte-fps" must beat
472 // "smpte"). On match, advance past the keyword.
473 auto tryKeyword = [&](const char *kw, Style s) {
474 auto p = it;
475 while(*kw && p != end && *p == *kw) { ++p; ++kw; }
476 if(*kw == 0 && (p == end || *p == '}' || *p == ':')) {
477 it = p;
478 _style = s;
479 return true;
480 }
481 return false;
482 };
483
484 if(!tryKeyword("smpte-fps", Style::SmpteWithFps)
485 && !tryKeyword("smpte-space", Style::SmpteSpaceFps)
486 && !tryKeyword("smpte", Style::Smpte)
487 && !tryKeyword("field", Style::Field)) {
488 // No keyword — leave _style at its default and let
489 // the base parser consume the entire remaining spec.
490 }
491
492 // A separating ':' between the hint and a standard string
493 // format spec is consumed here so the base parser sees a
494 // bare ">16" rather than ":>16".
495 if(it != end && *it == ':') ++it;
496
497 // Forward whatever remains to the standard string format
498 // parser so width / fill / alignment / precision still work.
499 ctx.advance_to(it);
500 return _base.parse(ctx);
501 }

503 template <typename FormatContext>
504 auto format(const promeki::Timecode &tc, FormatContext &ctx) const {
505 const VtcStringFormat *fmt = &VTC_STR_FMT_SMPTE;
506 switch(_style) {
507 case Style::Smpte: fmt = &VTC_STR_FMT_SMPTE; break;
508 case Style::SmpteWithFps: fmt = &VTC_STR_FMT_SMPTE_WITH_FPS; break;
509 case Style::SmpteSpaceFps: fmt = &VTC_STR_FMT_SMPTE_SPACE_FPS; break;
510 case Style::Field: fmt = &VTC_STR_FMT_FIELD; break;
511 }
512 auto [s, err] = tc.toString(fmt);
513 (void)err; // Formatter consumers do not get the error path.
514 return _base.format(std::string_view(s.cstr(), s.byteCount()), ctx);
515 }
