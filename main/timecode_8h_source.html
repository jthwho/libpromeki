502 enum class Style {
503 Smpte,
504 SmpteWithFps,
505 SmpteSpaceFps,
506 Field
507 };
508
509 Style _style = Style::Smpte;
510 std::formatter<std::string_view> _base;
511
512 constexpr auto parse(std::format_parse_context &ctx) {
513 auto it = ctx.begin();
514 auto end = ctx.end();
515
516 // Try to match a recognised style keyword at the start of
517 // the spec. Each candidate is checked in longest-first
518 // order so prefixes do not collide ("smpte-fps" must beat
519 // "smpte"). On match, advance past the keyword.
520 auto tryKeyword = [&](const char *kw, Style s) {
521 auto p = it;
522 while(*kw && p != end && *p == *kw) { ++p; ++kw; }
523 if(*kw == 0 && (p == end || *p == '}' || *p == ':')) {
524 it = p;
525 _style = s;
526 return true;
527 }
528 return false;
529 };
530
531 if(!tryKeyword("smpte-fps", Style::SmpteWithFps)
532 && !tryKeyword("smpte-space", Style::SmpteSpaceFps)
533 && !tryKeyword("smpte", Style::Smpte)
534 && !tryKeyword("field", Style::Field)) {
535 // No keyword — leave _style at its default and let
536 // the base parser consume the entire remaining spec.
537 }
538
539 // A separating ':' between the hint and a standard string
540 // format spec is consumed here so the base parser sees a
541 // bare ">16" rather than ":>16".
542 if(it != end && *it == ':') ++it;
543
544 // Forward whatever remains to the standard string format
545 // parser so width / fill / alignment / precision still work.
546 ctx.advance_to(it);
547 return _base.parse(ctx);
548 }
549
550 template <typename FormatContext>
551 auto format(const promeki::Timecode &tc, FormatContext &ctx) const {
552 const VtcStringFormat *fmt = &VTC_STR_FMT_SMPTE;
553 switch(_style) {
554 case Style::Smpte: fmt = &VTC_STR_FMT_SMPTE; break;
555 case Style::SmpteWithFps: fmt = &VTC_STR_FMT_SMPTE_WITH_FPS; break;
556 case Style::SmpteSpaceFps: fmt = &VTC_STR_FMT_SMPTE_SPACE_FPS; break;
557 case Style::Field: fmt = &VTC_STR_FMT_FIELD; break;
558 }
559 auto [s, err] = tc.toString(fmt);
560 (void)err; // Formatter consumers do not get the error path.
561 return _base.format(std::string_view(s.cstr(), s.byteCount()), ctx);
562 }
