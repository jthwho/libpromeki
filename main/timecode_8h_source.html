333
334 constexpr auto parse(std::format_parse_context &ctx) {
335 auto it = ctx.begin();
336 auto end = ctx.end();
337
338 // Try to match a recognised style keyword at the start of
339 // the spec. Each candidate is checked in longest-first
340 // order so prefixes do not collide ("smpte-fps" must beat
341 // "smpte"). On match, advance past the keyword.
342 auto tryKeyword = [&](const char *kw, Style s) {
343 auto p = it;
344 while(*kw && p != end && *p == *kw) { ++p; ++kw; }
345 if(*kw == 0 && (p == end || *p == '}' || *p == ':')) {
346 it = p;
347 _style = s;
348 return true;
349 }
350 return false;
351 };
352
353 if(!tryKeyword("smpte-fps", Style::SmpteWithFps)
354 && !tryKeyword("smpte-space", Style::SmpteSpaceFps)
355 && !tryKeyword("smpte", Style::Smpte)
356 && !tryKeyword("field", Style::Field)) {
357 // No keyword — leave _style at its default and let
358 // the base parser consume the entire remaining spec.
359 }
360
361 // A separating ':' between the hint and a standard string
362 // format spec is consumed here so the base parser sees a
363 // bare ">16" rather than ":>16".
364 if(it != end && *it == ':') ++it;
365
366 // Forward whatever remains to the standard string format
367 // parser so width / fill / alignment / precision still work.
368 ctx.advance_to(it);
369 return _base.parse(ctx);
370 }

372 template <typename FormatContext>
373 auto format(const promeki::Timecode &tc, FormatContext &ctx) const {
374 const VtcStringFormat *fmt = &VTC_STR_FMT_SMPTE;
375 switch(_style) {
376 case Style::Smpte: fmt = &VTC_STR_FMT_SMPTE; break;
377 case Style::SmpteWithFps: fmt = &VTC_STR_FMT_SMPTE_WITH_FPS; break;
378 case Style::SmpteSpaceFps: fmt = &VTC_STR_FMT_SMPTE_SPACE_FPS; break;
379 case Style::Field: fmt = &VTC_STR_FMT_FIELD; break;
380 }
381 auto [s, err] = tc.toString(fmt);
382 (void)err; // Formatter consumers do not get the error path.
383 return _base.format(std::string_view(s.cstr(), s.byteCount()), ctx);
384 }
