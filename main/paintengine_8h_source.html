299 Pixel createPixel(const Color &color) const {
300 const PixelDesc &pd = d->pixelDesc();
301 const ColorModel &targetModel = pd.isValid()
302 ? pd.colorModel() : ColorModel(ColorModel::sRGB);
303 Color c = (color.model() == targetModel)
304 ? color : color.convert(targetModel);
305 size_t count = pd.isValid() ? pd.compCount() : 4;
306 uint16_t data[PixelFormat::MaxComps] = {};
307 for(size_t i = 0; i < count && i < 3; i++) {
308 float v = c.comp(i) * 65535.0f;
309 if(v < 0.0f) v = 0.0f;
310 if(v > 65535.0f) v = 65535.0f;
311 data[i] = static_cast<uint16_t>(v);
312 }
313 if(count >= 4) {
314 float a = c.alpha() * 65535.0f;
315 if(a < 0.0f) a = 0.0f;
316 if(a > 65535.0f) a = 65535.0f;
317 data[3] = static_cast<uint16_t>(a);
318 }
319 return d->createPixel(data, count);
320 }
