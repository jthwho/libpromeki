308 Pixel createPixel(const Color &color) const {
309 const PixelFormat &pd = d->pixelFormat();
310 const ColorModel &targetModel = pd.isValid() ? pd.colorModel() : ColorModel(ColorModel::sRGB);
311 Color c = (color.model() == targetModel) ? color : color.convert(targetModel);
312 size_t count = pd.isValid() ? pd.compCount() : 4;
313 uint16_t data[PixelMemLayout::MaxComps] = {};
314 for (size_t i = 0; i < count && i < 3; i++) {
315 float v = c.comp(i) * 65535.0f;
316 if (v < 0.0f) v = 0.0f;
317 if (v > 65535.0f) v = 65535.0f;
318 data[i] = static_cast<uint16_t>(v);
319 }
320 if (count >= 4) {
321 float a = c.alpha() * 65535.0f;
322 if (a < 0.0f) a = 0.0f;
323 if (a > 65535.0f) a = 65535.0f;
324 data[3] = static_cast<uint16_t>(a);
325 }
326 return d->createPixel(data, count);
327 }
