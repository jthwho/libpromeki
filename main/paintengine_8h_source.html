300 Pixel createPixel(const Color &color) const {
301 const PixelFormat &pd = d->pixelFormat();
302 const ColorModel &targetModel = pd.isValid()
303 ? pd.colorModel() : ColorModel(ColorModel::sRGB);
304 Color c = (color.model() == targetModel)
305 ? color : color.convert(targetModel);
306 size_t count = pd.isValid() ? pd.compCount() : 4;
307 uint16_t data[PixelMemLayout::MaxComps] = {};
308 for(size_t i = 0; i < count && i < 3; i++) {
309 float v = c.comp(i) * 65535.0f;
310 if(v < 0.0f) v = 0.0f;
311 if(v > 65535.0f) v = 65535.0f;
312 data[i] = static_cast<uint16_t>(v);
313 }
314 if(count >= 4) {
315 float a = c.alpha() * 65535.0f;
316 if(a < 0.0f) a = 0.0f;
317 if(a > 65535.0f) a = 65535.0f;
318 data[3] = static_cast<uint16_t>(a);
319 }
320 return d->createPixel(data, count);
321 }
