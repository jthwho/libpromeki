569 enum class Style {
570 Smpte,
571 Named,
572 };

574 Style _style = Style::Smpte;
575 std::formatter<std::string_view> _base;
576
577 constexpr auto parse(std::format_parse_context &ctx) {
578 auto it = ctx.begin();
579 auto end = ctx.end();
580
581 // Match a style keyword at the start of the spec. Each
582 // candidate stops at end-of-spec or a ':' separator so
583 // trailing std-spec forwarding still works.
584 auto tryKeyword = [&](const char *kw, Style s) {
585 auto p = it;
586 while (*kw && p != end && *p == *kw) {
587 ++p;
588 ++kw;
589 }
590 if (*kw == 0 && (p == end || *p == '}' || *p == ':')) {
591 it = p;
592 _style = s;
593 return true;
594 }
595 return false;
596 };
597
598 if (!tryKeyword("smpte", Style::Smpte) && !tryKeyword("named", Style::Named)) {
599 // No keyword — leave _style at default and let
600 // the base parser consume the entire remaining spec.
601 }
602
603 // Separating ':' between the style hint and a standard
604 // string format spec is consumed here so the base parser
605 // sees a bare ">16" rather than ":>16".
606 if (it != end && *it == ':') ++it;
607
608 ctx.advance_to(it);
609 return _base.parse(ctx);
610 }
611
612 template <typename FormatContext>
613 auto format(const promeki::VideoFormat &vf, FormatContext &ctx) const {
614 promeki::VideoFormat::StringOptions opts;
615 opts.useNamedRaster = (_style == Style::Named);
616 const promeki::String s = vf.toString(opts);
617 return _base.format(std::string_view(s.cstr(), s.byteCount()), ctx);
618 }
