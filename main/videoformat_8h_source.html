576 enum class Style {
577 Smpte,
578 Named,
579 };
580
581 Style _style = Style::Smpte;
582 std::formatter<std::string_view> _base;
583
584 constexpr auto parse(std::format_parse_context &ctx) {
585 auto it = ctx.begin();
586 auto end = ctx.end();
587
588 // Match a style keyword at the start of the spec. Each
589 // candidate stops at end-of-spec or a ':' separator so
590 // trailing std-spec forwarding still works.
591 auto tryKeyword = [&](const char *kw, Style s) {
592 auto p = it;
593 while(*kw && p != end && *p == *kw) { ++p; ++kw; }
594 if(*kw == 0 && (p == end || *p == '}' || *p == ':')) {
595 it = p;
596 _style = s;
597 return true;
598 }
599 return false;
600 };
601
602 if(!tryKeyword("smpte", Style::Smpte)
603 && !tryKeyword("named", Style::Named)) {
604 // No keyword — leave _style at default and let
605 // the base parser consume the entire remaining spec.
606 }
607
608 // Separating ':' between the style hint and a standard
609 // string format spec is consumed here so the base parser
610 // sees a bare ">16" rather than ":>16".
611 if(it != end && *it == ':') ++it;
612
613 ctx.advance_to(it);
614 return _base.parse(ctx);
615 }
616
617 template <typename FormatContext>
618 auto format(const promeki::VideoFormat &vf, FormatContext &ctx) const {
619 promeki::VideoFormat::StringOptions opts;
620 opts.useNamedRaster = (_style == Style::Named);
621 const promeki::String s = vf.toString(opts);
622 return _base.format(std::string_view(s.cstr(), s.byteCount()), ctx);
623 }
