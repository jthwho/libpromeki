189
191 RefCount() : v(1) {}
192
194 RefCount(const RefCount &o) : v(1) {}
195
197 RefCount &operator=(const RefCount &) {
198 v.store(1, std::memory_order_relaxed);
199 return *this;
200 }
201
208 void inc() {
209 if (v.load(std::memory_order_relaxed) >= Immortal) return;
210 v.fetch_add(1, std::memory_order_relaxed);
211 return;
212 }
213
220 bool dec() {
221 if (v.load(std::memory_order_relaxed) >= Immortal) return false;
222 return v.fetch_sub(1, std::memory_order_acq_rel) == 1;
223 }

226 int value() const { return v.load(std::memory_order_relaxed); }
227
229 bool isImmortal() const { return v.load(std::memory_order_relaxed) >= Immortal; }

232 void setImmortal() { v.store(Immortal, std::memory_order_relaxed); }

518template <typename Derived, typename Base, bool CoW, typename ST>
519SharedPtr<Derived, CoW, Derived> sharedPointerCast(const SharedPtr<Base, CoW, ST> &sp) {
520 static_assert(IsSharedObject<Base>::value, "sharedPointerCast requires native shared objects");
521 static_assert(IsSharedObject<Derived>::value, "sharedPointerCast requires native shared objects");
522 static_assert(std::is_base_of_v<Base, Derived>, "Derived must publicly derive from Base");
523 SharedPtr<Derived, CoW, Derived> result;
524 if (sp.isNull()) return result;
525 Derived *d = dynamic_cast<Derived *>(sp._data);
526 if (d == nullptr) return result;
527 result._data = d;
528 result._data->_promeki_refct.inc();
529 return result;
530}
