210 RefCount() : v(1) {}
211
213 RefCount(const RefCount &o) : v(1) {}
214
216 RefCount &operator=(const RefCount &) {
217 v.store(1, std::memory_order_relaxed);
218 return *this;
219 }
220
227 void inc() {
228 if (v.load(std::memory_order_relaxed) >= Immortal) return;
229 v.fetch_add(1, std::memory_order_relaxed);
230 return;
231 }
232
239 bool dec() {
240 if (v.load(std::memory_order_relaxed) >= Immortal) return false;
241 return v.fetch_sub(1, std::memory_order_acq_rel) == 1;
242 }

248 bool isImmortal() const { return v.load(std::memory_order_relaxed) >= Immortal; }

537template <typename Derived, typename Base, bool CoW, typename ST>
538SharedPtr<Derived, CoW, Derived> sharedPointerCast(const SharedPtr<Base, CoW, ST> &sp) {
539 static_assert(IsSharedObject<Base>::value, "sharedPointerCast requires native shared objects");
540 static_assert(IsSharedObject<Derived>::value, "sharedPointerCast requires native shared objects");
541 static_assert(std::is_base_of_v<Base, Derived>, "Derived must publicly derive from Base");
542 SharedPtr<Derived, CoW, Derived> result;
543 if (sp.isNull()) return result;
544 Derived *d = dynamic_cast<Derived *>(sp._data);
545 if (d == nullptr) return result;
546 result._data = d;
547 result._data->_promeki_refct.inc();
548 return result;
549}
