147 void inc() {
148 if(v.load(std::memory_order_relaxed) >= Immortal) return;
149 v.fetch_add(1, std::memory_order_relaxed);

494SharedPtr<Derived, CoW, Derived>
495sharedPointerCast(const SharedPtr<Base, CoW, ST> &sp) {
496 static_assert(IsSharedObject<Base>::value,
497 "sharedPointerCast requires native shared objects");
498 static_assert(IsSharedObject<Derived>::value,
499 "sharedPointerCast requires native shared objects");
500 static_assert(std::is_base_of_v<Base, Derived>,
501 "Derived must publicly derive from Base");
502 SharedPtr<Derived, CoW, Derived> result;
503 if(sp.isNull()) return result;
504 Derived *d = dynamic_cast<Derived *>(sp._data);
505 if(d == nullptr) return result;
506 result._data = d;
507 result._data->_promeki_refct.inc();
508 return result;
