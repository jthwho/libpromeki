68 static int registerFactory(FileFormatFactory *factory) {
69 if (factory == nullptr) return -1;
70 auto &list = factoryList();
71 int ret = list.size();
72 promekiLog(Logger::LogLevel::Debug, "Registered FileFormatFactory %s", factory->name().cstr());
73 // Adopt the factory so the list owns it; otherwise
74 // every PROMEKI_REGISTER_FILE_FORMAT_FACTORY(new Foo)
75 // would leak a pointer for the lifetime of the process.
76 list.pushToBack(UniquePtr<FileFormatFactory>::takeOwnership(factory));

119 bool isExtensionSupported(const String &filename) const {
120 size_t dot = filename.rfind('.');
