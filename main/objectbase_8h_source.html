440template <typename... Args>
441void ObjectBase::connect(Signal<Args...> *signal, Slot<Args...> *slot) {
442 if(signal == nullptr || slot == nullptr) return;
443
444 signal->connect([signal, slot](Args... args) {
445 ObjectBase *signalObject = static_cast<ObjectBase *>(signal->owner());
446 ObjectBase *slotObject = static_cast<ObjectBase *>(slot->owner());
447
448 EventLoop *slotLoop = slotObject->_eventLoop;
449 EventLoop *currentLoop = EventLoop::current();
450
451 // Same thread or no event loop: direct call (zero overhead path)
452 if(slotLoop == nullptr || slotLoop == currentLoop) {
453 ObjectBase *prevSender = slotObject->_signalSender;
454 slotObject->_signalSender = signalObject;
455 slot->exec(args...);
456 slotObject->_signalSender = prevSender;
457 } else {
458 // Cross-thread: marshal via VariantList and post
459 VariantList packed = Signal<Args...>::pack(args...);
460 ObjectBasePtr senderTracker(signalObject);
461 slotLoop->postCallable(
462 [slot, slotObject, packed = std::move(packed),
463 senderTracker = std::move(senderTracker)]() {
464 ObjectBase *prevSender = slotObject->_signalSender;
465 slotObject->_signalSender =
466 senderTracker.isValid()
467 ? const_cast<ObjectBase *>(senderTracker.data())
468 : nullptr;
469 slot->exec(packed);
470 slotObject->_signalSender = prevSender;
471 }
472 );
473 }
474 }, slot->owner());
475
476 // Register a cleanup
477 ObjectBase *signalObject = static_cast<ObjectBase *>(signal->owner());
478 ObjectBase *slotObject = static_cast<ObjectBase *>(slot->owner());
479 slotObject->_cleanupList += Cleanup(
480 signalObject,
481 [signal](ObjectBase *ptr){ signal->disconnectFromObject(ptr); }
482 );
483}
