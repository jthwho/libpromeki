511 // A single process-wide mutex serializes every
512 // ObjectBasePtr link / unlink (linkTo, linkFromSource,
513 // unlink) with ObjectBase::runCleanup. A per-object
514 // mutex is unsafe here because unlink reads the obj
515 // pointer via the atomic and *then* needs to take a
516 // lock to manipulate the map — if we put that lock on
517 // the object itself, the object can be destroyed in
518 // between, leaving us with a use-after-free on the
519 // mutex member. Serializing through one global mutex
520 // closes that window: an unlink that observes obj
521 // != nullptr is guaranteed that runCleanup hasn't run
522 // yet (and won't until we release). The link path
523 // folds the source-pointer load and the @c _pointerMap
524 // registration into the same critical section so a
525 // newly constructed tracker is either fully registered
526 // before runCleanup nulls it, or never observes a
527 // doomed pointer at all.
528 static Mutex &objectBasePtrMutex();

551 void runCleanup() {
552 // Null out any ObjectBasePtr's that are currently pointing
553 // to this object. Hold the global mutex so a concurrent
554 // ObjectBasePtr link / unlink on another thread can't
555 // observe a half-destroyed object.
556 {
557 Mutex::Locker lock(objectBasePtrMutex());
558 for (auto item : _pointerMap) {
559 item.first->store(nullptr, std::memory_order_release);
560 }
561 _pointerMap.clear();
562 }
563
564 // Walk down the cleanup list and run any cleanup functions
565 for (auto &item : _cleanupList) {
566 if (!item.object.isValid()) continue;
567 item.func(this);
568 }
569 _cleanupList.clear();
570 return;
571 }

574template <typename T> inline void ObjectBasePtr<T>::linkTo(ObjectBase *obj) {
575 // Single critical section: store @c p and add to the tracked
576 // object's @c _pointerMap atomically. An interleaved
577 // runCleanup can either run entirely before us (it sees no
578 // entry, then we observe the cleared @c _pointerMap and
579 // implicitly do nothing meaningful — though @c obj is then a
580 // dangling pointer, see note below) or entirely after us (it
581 // observes the new entry and nulls @c p). Either way the
582 // invariant "non-null @c p implies registered in the tracked
583 // object's @c _pointerMap" holds for the duration of the
584 // tracker's life.
585 //
586 // Note on the dangling-input case: callers of this constructor
587 // pass a raw pointer they own. If the caller's object is
588 // destroyed before @c linkTo runs, that's a caller-side
589 // lifetime bug, not something this class can plug. The
590 // mutex-protected variant @ref linkFromSource exists for the
591 // copy paths where the source is itself an ObjectBasePtr and
592 // can be invalidated concurrently — that one re-checks the
593 // atomic under the lock.
594 Mutex::Locker lock(ObjectBase::objectBasePtrMutex());
595 p.store(obj, std::memory_order_relaxed);
596 if (obj != nullptr) {
597 obj->_pointerMap[&p] = &p;
598 }
599 return;
600}

602template <typename T>
603inline void ObjectBasePtr<T>::linkFromSource(const std::atomic<ObjectBase *> &source) {
604 // Lock first, then load the source: if a concurrent
605 // runCleanup is mid-flight on the tracked object it has
606 // already nulled the source's atomic before releasing the
607 // mutex, so once we hold the lock the source either still
608 // points at a live object (which can't be destroyed until we
609 // release) or has been nulled (we observe nullptr and link to
610 // nothing). Performing the source load *outside* the lock —
611 // as the previous mem-initializer pattern did — leaves a
612 // window where @c p is set to a doomed pointer, runCleanup
613 // misses us because we're not in @c _pointerMap yet, and the
614 // subsequent registration writes through a freed @c obj.
615 Mutex::Locker lock(ObjectBase::objectBasePtrMutex());
616 ObjectBase *obj = source.load(std::memory_order_relaxed);
617 p.store(obj, std::memory_order_relaxed);
618 if (obj != nullptr) {
619 obj->_pointerMap[&p] = &p;
620 }
621 return;
622}

624template <typename T> inline void ObjectBasePtr<T>::unlink() {
625 // Same ordering as the link path: acquire the global mutex
626 // first so we can't race ObjectBase::runCleanup, which holds
627 // the same lock while it walks _pointerMap and frees the
628 // trackers.
629 Mutex::Locker lock(ObjectBase::objectBasePtrMutex());
630 ObjectBase *obj = p.exchange(nullptr, std::memory_order_acq_rel);
631 if (obj != nullptr) {
632 auto it = obj->_pointerMap.find(&p);
633 if (it != obj->_pointerMap.end()) {
634 obj->_pointerMap.remove(it);
635 }
636 }
637}
