358 void runCleanup() {
359 // Null out any ObjectBasePtr's that are currently pointing
360 // to this object.
361 for(auto item : _pointerMap) item.first->p = nullptr;
362 _pointerMap.clear();
363
364 // Walk down the cleanup list and run any cleanup functions
365 for(auto &item : _cleanupList) {
366 if(!item.object.isValid()) continue;
