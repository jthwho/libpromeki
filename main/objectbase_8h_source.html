412 void runCleanup() {
413 // Null out any ObjectBasePtr's that are currently pointing
414 // to this object. Hold the mutex so concurrent unlink()
415 // on another thread won't modify _pointerMap mid-iteration.
416 {
417 Mutex::Locker lock(_pointerMapMutex);
418 for(auto item : _pointerMap) {
419 item.first->p.store(nullptr, std::memory_order_release);
420 }
421 _pointerMap.clear();
422 }
423
424 // Walk down the cleanup list and run any cleanup functions
425 for(auto &item : _cleanupList) {
426 if(!item.object.isValid()) continue;
427 item.func(this);

461template <typename... Args>
462void ObjectBase::connect(Signal<Args...> *signal, Slot<Args...> *slot) {
463 if(signal == nullptr || slot == nullptr) return;
464
465 signal->connect([signal, slot](Args... args) {
466 ObjectBase *signalObject = static_cast<ObjectBase *>(signal->owner());
467 ObjectBase *slotObject = static_cast<ObjectBase *>(slot->owner());
468
469 EventLoop *slotLoop = slotObject->_eventLoop;
470 EventLoop *currentLoop = EventLoop::current();
471
472 // Same thread or no event loop: direct call (zero overhead path)
473 if(slotLoop == nullptr || slotLoop == currentLoop) {
474 ObjectBase *prevSender = slotObject->_signalSender;
475 slotObject->_signalSender = signalObject;
476 slot->exec(args...);
477 slotObject->_signalSender = prevSender;
478 } else {
479 // Cross-thread: marshal via VariantList and post
480 VariantList packed = Signal<Args...>::pack(args...);
481 ObjectBasePtr senderTracker(signalObject);
482 slotLoop->postCallable(
483 [slot, slotObject, packed = std::move(packed),
484 senderTracker = std::move(senderTracker)]() {
485 ObjectBase *prevSender = slotObject->_signalSender;
486 slotObject->_signalSender =
487 senderTracker.isValid()
488 ? const_cast<ObjectBase *>(senderTracker.data())
489 : nullptr;
490 slot->exec(packed);
491 slotObject->_signalSender = prevSender;
492 }
493 );
494 }
495 }, slot->owner());
496
497 // Register a cleanup
498 ObjectBase *signalObject = static_cast<ObjectBase *>(signal->owner());
499 ObjectBase *slotObject = static_cast<ObjectBase *>(slot->owner());
500 slotObject->_cleanupList += Cleanup(
501 signalObject,
502 [signal](ObjectBase *ptr){ signal->disconnectFromObject(ptr); }
503 );
504}

506// Context-aware Signal::connect overload. Defined here because it
507// needs ObjectBase and EventLoop to be complete — signal.h forward
508// declares ObjectBase and only stores the declaration. The wrapping
509// bridge lambda captures @p owner and uses its @c eventLoop() at
510// emit time, matching Qt's @c Qt::AutoConnection: direct invocation
511// on the owner's own thread, @c postCallable marshalling otherwise.
512template <typename... Args>
513size_t Signal<Args...>::connect(Function slot, ObjectBase *owner) {
514 // Passing a null @ref ObjectBase context is a programmer error
515 // — the whole point of this overload is to anchor dispatch to
516 // the owner's EventLoop, and there is no loop to anchor to
517 // when @p owner is null. Callers that explicitly want the
518 // raw, same-thread-only behaviour should use the @c void*
519 // overload with @c nullptr instead.
520 PROMEKI_ASSERT(owner != nullptr);
521 return connect(
522 [slot = std::move(slot), owner](Args... args) {
523 EventLoop *ownerLoop = owner->eventLoop();
524 EventLoop *currentLoop = EventLoop::current();
525 if(ownerLoop == nullptr || ownerLoop == currentLoop) {
526 slot(args...);
527 return;
528 }
529 // Copy args into a tuple, hop to the owner's
530 // EventLoop, then unpack and invoke. Requires
531 // every Args type to be copy-constructible,
532 // which is the same restriction the Slot-based
533 // ObjectBase::connect path already carries via
534 // VariantList packing.
535 ownerLoop->postCallable(
536 [slot, tup = std::make_tuple(std::move(args)...)]() {
537 std::apply(slot, tup);
538 });
539 },
540 static_cast<void *>(owner));
541}
