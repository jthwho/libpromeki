432 void runCleanup() {
433 // Null out any ObjectBasePtr's that are currently pointing
434 // to this object. Hold the mutex so concurrent unlink()
435 // on another thread won't modify _pointerMap mid-iteration.
436 {
437 Mutex::Locker lock(_pointerMapMutex);
438 for(auto item : _pointerMap) {
439 item.first->store(nullptr, std::memory_order_release);
440 }
441 _pointerMap.clear();
442 }
443
444 // Walk down the cleanup list and run any cleanup functions
445 for(auto &item : _cleanupList) {
446 if(!item.object.isValid()) continue;
447 item.func(this);
448 }
449 _cleanupList.clear();
450 return;
451 }

454template <typename T>
455inline void ObjectBasePtr<T>::link() {
456 ObjectBase *obj = p.load(std::memory_order_relaxed);
457 if(obj != nullptr) {
458 Mutex::Locker lock(obj->_pointerMapMutex);
459 obj->_pointerMap[&p] = &p;
460 }
461 return;
462}

464template <typename T>
465inline void ObjectBasePtr<T>::unlink() {
466 ObjectBase *obj = p.exchange(nullptr, std::memory_order_acq_rel);
467 if(obj != nullptr) {
468 Mutex::Locker lock(obj->_pointerMapMutex);
469 auto it = obj->_pointerMap.find(&p);
470 if(it != obj->_pointerMap.end()) {
471 obj->_pointerMap.remove(it);
472 }
473 }
