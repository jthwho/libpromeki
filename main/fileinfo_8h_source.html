210 DateTime lastModified() const {
211 std::error_code ec;
212 auto t = std::filesystem::last_write_time(_path, ec);
213 if (ec) return DateTime();
214 // file_time_type is not directly convertible to
215 // system_clock::time_point in pre-C++20 stdlibs,
216 // and clock_cast (C++20) is not yet ubiquitous in
217 // the libstdc++/libc++ versions in our matrix.
218 // Shift the file_clock instant onto system_clock
219 // by anchoring both clocks at "now" and applying
220 // the delta — accurate to one tick of either
221 // clock, which is plenty for HTTP semantics.
222 const auto now_fc = decltype(t)::clock::now();
223 const auto now_sc = std::chrono::system_clock::now();
224 const auto sc =
225 std::chrono::time_point_cast<std::chrono::system_clock::duration>(t - now_fc + now_sc);
226 return DateTime(sc);
227 }
