134 template <typename N> bool pointIsInside(const Point<N, 2> &p) const {
135 // First gate the >=0 check so a signed point coordinate paired with
136 // an unsigned size dimension doesn't trip -Wsign-compare on the
137 // upper bound: after p.x() >= 0 the cast to T is always safe.
138 if (p.x() < N{0} || p.y() < N{0}) return false;

146 // Parses one width or height token from @p s, dispatching to
147 // strtoll for signed T and strtoull for unsigned T so that
148 // negative inputs are accepted by signed templates and
149 // rejected by unsigned templates. On success returns true,
150 // sets @p out, and points @p end at the first unparsed
151 // character. On parse failure returns false.
152 static bool parseDim(const char *s, T &out, const char *&end) {
153 char *parseEnd = nullptr;
154 errno = 0;
155 if constexpr (std::is_integral_v<T> && std::is_signed_v<T>) {
156 long long v = std::strtoll(s, &parseEnd, 10);
157 end = parseEnd;
158 if (parseEnd == s || errno != 0) return false;
159 out = static_cast<T>(v);
160 return true;
161 } else if constexpr (std::is_integral_v<T>) {
162 // strtoull silently accepts a leading '-' and
163 // returns the negation as unsigned. Reject up
164 // front so an unsigned size cannot be parsed
165 // from a negative-looking input.
166 const char *p = s;
167 while (*p == ' ' || *p == '\t') ++p;
168 if (*p == '-') return false;
169 unsigned long long v = std::strtoull(s, &parseEnd, 10);
170 end = parseEnd;
171 if (parseEnd == s || errno != 0) return false;
172 out = static_cast<T>(v);
173 return true;
174 } else {
175 double v = std::strtod(s, &parseEnd);
176 end = parseEnd;
177 if (parseEnd == s || errno != 0) return false;
178 out = static_cast<T>(v);
179 return true;
180 }
181 }
