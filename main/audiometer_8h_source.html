160 void process(const float *samples, size_t frames, size_t channels) override {
161 if (channels > _channels) channels = _channels;
162 if (frames == 0 || channels == 0) return;
163
164 const float decay = _peakDecay;
165 const float alpha = _rmsAlpha;
166
167 for (size_t c = 0; c < channels; ++c) {
168 float peak = _peak[c].load(std::memory_order_relaxed) * decay;
169 // Convert held RMS back to mean-square for accumulation.
170 float held = _rms[c].load(std::memory_order_relaxed);
171 float msq = held * held;
172 for (size_t f = 0; f < frames; ++f) {
173 float s = samples[f * channels + c];
174 float a = std::fabs(s);
175 if (a > peak) peak = a;
176 msq = msq + alpha * (s * s - msq);
177 }
178 _peak[c].store(peak, std::memory_order_relaxed);
179 _rms[c].store(std::sqrt(msq < 0.0f ? 0.0f : msq), std::memory_order_relaxed);
180 }
181 }
182
184 void reset() override {
185 for (size_t i = 0; i < _channels; ++i) {
186 _peak[i].store(0.0f, std::memory_order_relaxed);
187 _rms[i].store(0.0f, std::memory_order_relaxed);
188 }
189 }
