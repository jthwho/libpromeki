440inline Error MemSpace::copy(const MemAllocation &src, const MemAllocation &dst, size_t bytes) const {
441 if (src.ptr == nullptr || dst.ptr == nullptr) {
442 d->stats->copyFailCount.fetchAndAdd(1);
443 return Error::Invalid;
444 }
445 Error err = d->copy(src, dst, bytes);
446 if (err.isOk()) {
447 d->stats->copyCount.fetchAndAdd(1);
448 d->stats->copyBytes.fetchAndAdd(bytes);
449 } else {
450 d->stats->copyFailCount.fetchAndAdd(1);
451 }
452 return err;
453}

455inline MemAllocation MemSpace::alloc(size_t bytes, size_t align) const {
456 MemAllocation a;
457 a.size = bytes;
458 a.align = align;
459 a.ms = *this;
460 // A zero-byte allocation is valid (returns a.ptr == nullptr) but
461 // does not count as a failure. Skip stats so counters aren't
462 // polluted by the legitimate empty-buffer case.
463 if (bytes == 0) return a;
464 d->alloc(a);
465 if (a.ptr != nullptr) {
466 d->stats->recordAlloc(static_cast<uint64_t>(bytes));
467 } else {
468 d->stats->allocFailCount.fetchAndAdd(1);
469 }
470 return a;
471}

473inline void MemSpace::release(MemAllocation &alloc) const {
474 if (alloc.ptr == nullptr) return;
475 uint64_t bytes = static_cast<uint64_t>(alloc.size);
476 d->release(alloc);
477 d->stats->recordRelease(bytes);
478 alloc.ptr = nullptr;
479 alloc.priv = nullptr;
480}
