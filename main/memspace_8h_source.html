455inline Error MemSpace::copy(const MemAllocation &src, const MemAllocation &dst, size_t bytes) const {
456 if (src.ptr == nullptr || dst.ptr == nullptr) {
457 d->stats->copyFailCount.fetchAndAdd(1);
458 return Error::Invalid;
459 }
460 Error err = d->copy(src, dst, bytes);
461 if (err.isOk()) {
462 d->stats->copyCount.fetchAndAdd(1);
463 d->stats->copyBytes.fetchAndAdd(bytes);
464 } else {
465 d->stats->copyFailCount.fetchAndAdd(1);
466 }
467 return err;
468}

470inline MemAllocation MemSpace::alloc(size_t bytes, size_t align) const {
471 MemAllocation a;
472 a.size = bytes;
473 a.align = align;
474 a.ms = *this;
475 // A zero-byte allocation is valid (returns a.ptr == nullptr) but
476 // does not count as a failure. Skip stats so counters aren't
477 // polluted by the legitimate empty-buffer case.
478 if (bytes == 0) return a;
479 d->alloc(a);
480 if (a.ptr != nullptr) {
481 d->stats->recordAlloc(static_cast<uint64_t>(bytes));
482 } else {
483 d->stats->allocFailCount.fetchAndAdd(1);
484 }
485 return a;
486}
487
488inline void MemSpace::release(MemAllocation &alloc) const {
489 if (alloc.ptr == nullptr) return;
490 uint64_t bytes = static_cast<uint64_t>(alloc.size);
491 d->release(alloc);
492 d->stats->recordRelease(bytes);
493 alloc.ptr = nullptr;
494 alloc.priv = nullptr;
495}
