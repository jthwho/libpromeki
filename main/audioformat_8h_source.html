359 // -- Direct (no-float) format-to-format conversion --------
360 //
361 // The library tracks a per-(src,dst) registry of "direct"
362 // converters that bypass the via-float intermediate step.
363 // Direct converters are faster (one memory pass) and
364 // additionally permit @ref isBitAccurateTo, which matters
365 // when audio buffers carry non-PCM payloads (SMPTE 337M
366 // data bursts, AES3 user bits, …) — those bytes survive
367 // a direct integer-to-integer transform but are scrambled
368 // by an int → float → int round-trip.

518 template <typename IntegerType, IntegerType Min, IntegerType Max>
519 static float integerToFloat(IntegerType value) {
520 static_assert(std::is_integral<IntegerType>::value, "IntegerType must be an integer.");
521 constexpr float min = static_cast<float>(Min);
522 constexpr float max = static_cast<float>(Max);
523 return ((static_cast<float>(value) - min) * 2.0f / (max - min)) - 1.0f;
524 }

532 template <typename IntegerType> static float integerToFloat(IntegerType value) {
533 static_assert(std::is_integral<IntegerType>::value, "IntegerType must be an integer.");
534 return integerToFloat<IntegerType, std::numeric_limits<IntegerType>::min(),
535 std::numeric_limits<IntegerType>::max()>(value);
536 }

550 template <typename IntegerType, IntegerType Min, IntegerType Max>
551 static IntegerType floatToInteger(float value) {
552 static_assert(std::is_integral<IntegerType>::value, "IntegerType must be an integer.");
553 const float min = static_cast<float>(Min);
554 const float max = static_cast<float>(Max);
555 if (value <= -1.0f)
556 return Min;
557 else if (value >= 1.0f)
558 return Max;
559 return static_cast<IntegerType>((value + 1.0f) * 0.5f * (max - min) + min);
560 }

586 template <typename IntegerType, bool InputIsBigEndian>
587 static void samplesToFloatImpl(float *out, const uint8_t *inbuf, size_t samples) {
588 static_assert(std::is_integral<IntegerType>::value, "IntegerType must be an integer.");
589 const IntegerType *in = reinterpret_cast<const IntegerType *>(inbuf);
590 for (size_t i = 0; i < samples; ++i) {
591 IntegerType val = *in++;
592 if constexpr (InputIsBigEndian != System::isBigEndian()) System::swapEndian(val);
593 *out++ = integerToFloat<IntegerType>(val);
594 }
595 }

609 template <typename IntegerType, bool OutputIsBigEndian>
610 static void floatToSamplesImpl(uint8_t *outbuf, const float *in, size_t samples) {
611 static_assert(std::is_integral<IntegerType>::value, "IntegerType must be an integer.");
612 IntegerType *out = reinterpret_cast<IntegerType *>(outbuf);
613 for (size_t i = 0; i < samples; ++i) {
614 IntegerType val = floatToInteger<IntegerType>(*in++);
615 if constexpr (OutputIsBigEndian != System::isBigEndian()) System::swapEndian(val);
616 *out++ = val;
617 }
