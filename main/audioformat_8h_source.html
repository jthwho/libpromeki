365 template <typename IntegerType, IntegerType Min, IntegerType Max>
366 static float integerToFloat(IntegerType value) {
367 static_assert(std::is_integral<IntegerType>::value, "IntegerType must be an integer.");
368 constexpr float min = static_cast<float>(Min);
369 constexpr float max = static_cast<float>(Max);
370 return((static_cast<float>(value) - min) * 2.0f / (max - min)) - 1.0f;
371 }

399 template <typename IntegerType, IntegerType Min, IntegerType Max>
400 static IntegerType floatToInteger(float value) {
401 static_assert(std::is_integral<IntegerType>::value, "IntegerType must be an integer.");
402 const float min = static_cast<float>(Min);
403 const float max = static_cast<float>(Max);
404 if(value <= -1.0f) return Min;
405 else if(value >= 1.0f) return Max;
406 return static_cast<IntegerType>((value + 1.0f) * 0.5f * (max - min) + min);
407 }

435 template <typename IntegerType, bool InputIsBigEndian>
436 static void samplesToFloatImpl(float *out, const uint8_t *inbuf, size_t samples) {
437 static_assert(std::is_integral<IntegerType>::value, "IntegerType must be an integer.");
438 const IntegerType *in = reinterpret_cast<const IntegerType *>(inbuf);
439 for(size_t i = 0; i < samples; ++i) {
440 IntegerType val = *in++;
441 if constexpr (InputIsBigEndian != System::isBigEndian()) System::swapEndian(val);
442 *out++ = integerToFloat<IntegerType>(val);
443 }
444 }
445
458 template <typename IntegerType, bool OutputIsBigEndian>
459 static void floatToSamplesImpl(uint8_t *outbuf, const float *in, size_t samples) {
460 static_assert(std::is_integral<IntegerType>::value, "IntegerType must be an integer.");
461 IntegerType *out = reinterpret_cast<IntegerType *>(outbuf);
462 for(size_t i = 0; i < samples; ++i) {
463 IntegerType val = floatToInteger<IntegerType>(*in++);
464 if constexpr (OutputIsBigEndian != System::isBigEndian()) System::swapEndian(val);
465 *out++ = val;
466 }
467 }
