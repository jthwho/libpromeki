947template<size_t N>
948class CompiledString {
949 public:
950 consteval CompiledString(const char (&str)[N])
951 : _bytes{}, _codepoints{}, _charCount(0), _isAscii(true) {
952 for(size_t i = 0; i < N; ++i) _bytes[i] = str[i];
953 size_t pos = 0;
954 while(pos < N - 1) {
955 unsigned char b = static_cast<unsigned char>(str[pos]);
956 if(b > 0x7F) _isAscii = false;
957 char32_t cp;
958 size_t seqLen;
959 if(b < 0x80) { cp = b; seqLen = 1; }
960 else if(b < 0xE0) { cp = b & 0x1F; seqLen = 2; }
961 else if(b < 0xF0) { cp = b & 0x0F; seqLen = 3; }
962 else { cp = b & 0x07; seqLen = 4; }
963 for(size_t j = 1; j < seqLen && pos + j < N - 1; ++j)
964 cp = (cp << 6) | (static_cast<unsigned char>(str[pos + j]) & 0x3F);
965 _codepoints[_charCount++] = cp;
966 pos += seqLen;
967 }
968 }
969
970 constexpr bool isAscii() const { return _isAscii; }
971 constexpr size_t charCount() const { return _charCount; }
972 constexpr size_t byteCount() const { return N - 1; }
973 constexpr const char *bytes() const { return _bytes; }
974 constexpr const char32_t *codepoints() const { return _codepoints; }
975
977 constexpr uint64_t hash() const {
978 if(_isAscii) return fnv1aData(_bytes, N - 1);
979 return fnv1aData(_codepoints, _charCount * sizeof(char32_t));
980 }
981
982 private:
983 char _bytes[N];
984 char32_t _codepoints[N]; // worst case: N-1 codepoints
985 size_t _charCount;
986 bool _isAscii;
987};
