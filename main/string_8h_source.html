938template<size_t N>
939class CompiledString {
940 public:
941 consteval CompiledString(const char (&str)[N])
942 : _bytes{}, _codepoints{}, _charCount(0), _isAscii(true) {
943 for(size_t i = 0; i < N; ++i) _bytes[i] = str[i];
944 size_t pos = 0;
945 while(pos < N - 1) {
946 unsigned char b = static_cast<unsigned char>(str[pos]);
947 if(b > 0x7F) _isAscii = false;
948 char32_t cp;
949 size_t seqLen;
950 if(b < 0x80) { cp = b; seqLen = 1; }
951 else if(b < 0xE0) { cp = b & 0x1F; seqLen = 2; }
952 else if(b < 0xF0) { cp = b & 0x0F; seqLen = 3; }
953 else { cp = b & 0x07; seqLen = 4; }
954 for(size_t j = 1; j < seqLen && pos + j < N - 1; ++j)
955 cp = (cp << 6) | (static_cast<unsigned char>(str[pos + j]) & 0x3F);
956 _codepoints[_charCount++] = cp;
957 pos += seqLen;
958 }
959 }
960
961 constexpr bool isAscii() const { return _isAscii; }
962 constexpr size_t charCount() const { return _charCount; }
963 constexpr size_t byteCount() const { return N - 1; }
964 constexpr const char *bytes() const { return _bytes; }
965 constexpr const char32_t *codepoints() const { return _codepoints; }
966
968 constexpr uint64_t hash() const {
969 if(_isAscii) return fnv1aData(_bytes, N - 1);
970 return fnv1aData(_codepoints, _charCount * sizeof(char32_t));
971 }
972
973 private:
974 char _bytes[N];
975 char32_t _codepoints[N]; // worst case: N-1 codepoints
976 size_t _charCount;
977 bool _isAscii;
978};
