782 friend bool operator<(const String &lhs, const String &rhs) {
783 if (lhs.d->isLatin1() == rhs.d->isLatin1()) {
784 // Same encoding: byte-level comparison is correct.
785 // - Latin1 vs Latin1: byte order == codepoint order.
786 // - Unicode vs Unicode: UTF-8 byte order == codepoint order.
787 return lhs.d->str() < rhs.d->str();
788 }
789 // Mixed encodings: walk codepoint-by-codepoint so that
790 // logically equal strings (e.g. Latin1 0xE9 and Unicode
791 // U+00E9) compare as equal under operator<.
792 const size_t llen = lhs.d->length();
793 const size_t rlen = rhs.d->length();
794 const size_t n = std::min(llen, rlen);
795 for (size_t i = 0; i < n; ++i) {
796 char32_t a = lhs.d->charAt(i).codepoint();
797 char32_t b = rhs.d->charAt(i).codepoint();
798 if (a != b) return a < b;
799 }
800 return llen < rlen;
801 }

1086 const char *s = cleaned.cstr();

1089 long long v = std::strtoll(s, &end, base);
1090 if (end == s || *end != '\0') {
1091 if (err != nullptr) *err = Error::Invalid;
1092 return OutputType{};
1093 }

1354template <size_t N> class CompiledString {
1355 public:
1356 consteval CompiledString(const char (&str)[N])
1357 : _bytes{}, _codepoints{}, _charCount(0), _isAscii(true) {
1358 for (size_t i = 0; i < N; ++i) _bytes[i] = str[i];
1359 size_t pos = 0;
1360 while (pos < N - 1) {
1361 unsigned char b = static_cast<unsigned char>(str[pos]);
1362 if (b > 0x7F) _isAscii = false;
1363 char32_t cp;
1364 size_t seqLen;
1365 if (b < 0x80) {
1366 cp = b;
1367 seqLen = 1;
1368 } else if (b < 0xE0) {
1369 cp = b & 0x1F;
1370 seqLen = 2;
1371 } else if (b < 0xF0) {
1372 cp = b & 0x0F;
1373 seqLen = 3;
1374 } else {
1375 cp = b & 0x07;
1376 seqLen = 4;
1377 }
1378 for (size_t j = 1; j < seqLen && pos + j < N - 1; ++j)
1379 cp = (cp << 6) | (static_cast<unsigned char>(str[pos + j]) & 0x3F);
1380 _codepoints[_charCount++] = cp;
1381 pos += seqLen;
1382 }
1383 }
1384
1385 constexpr bool isAscii() const { return _isAscii; }
1386 constexpr size_t charCount() const { return _charCount; }
1387 constexpr size_t byteCount() const { return N - 1; }
1388 constexpr const char *bytes() const { return _bytes; }
1389 constexpr const char32_t *codepoints() const { return _codepoints; }
1390
1399 constexpr uint64_t hash() const {
1400 if (_isAscii) return fnv1aLatin1AsCodepoints(_bytes, N - 1);
1401 return fnv1aCodepoints(_codepoints, _charCount);
1402 }
1403
1404 // Data members are public so the class qualifies as a
1405 // structural type (C++20 [temp.param]/7) and can therefore
1406 // be used as a class-type non-type template parameter —
1407 // e.g. `template <CompiledString Name>` in StringRegistry
1408 // and VariantDatabase. Treat them as private.
1409 char _bytes[N];
1410 char32_t _codepoints[N]; // worst case: N-1 codepoints
1411 size_t _charCount;
1412 bool _isAscii;
1413};
