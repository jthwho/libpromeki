734 friend bool operator<(const String &lhs, const String &rhs) {
735 if(lhs.d->isLatin1() == rhs.d->isLatin1()) {
736 // Same encoding: byte-level comparison is correct.
737 // - Latin1 vs Latin1: byte order == codepoint order.
738 // - Unicode vs Unicode: UTF-8 byte order == codepoint order.
739 return lhs.d->str() < rhs.d->str();
740 }
741 // Mixed encodings: walk codepoint-by-codepoint so that
742 // logically equal strings (e.g. Latin1 0xE9 and Unicode
743 // U+00E9) compare as equal under operator<.
744 const size_t llen = lhs.d->length();
745 const size_t rlen = rhs.d->length();
746 const size_t n = std::min(llen, rlen);
747 for(size_t i = 0; i < n; ++i) {
748 char32_t a = lhs.d->charAt(i).codepoint();
749 char32_t b = rhs.d->charAt(i).codepoint();
750 if(a != b) return a < b;
751 }
752 return llen < rlen;
753 }

781 String toUpper() const {

1034 unsigned long long v = std::strtoull(s, &end, 10);
1035 ret = static_cast<OutputType>(v);

1037 double v = std::strtod(s, &end);
1038 ret = static_cast<OutputType>(v);

1233class CompiledString {
1234 public:
1235 consteval CompiledString(const char (&str)[N])
1236 : _bytes{}, _codepoints{}, _charCount(0), _isAscii(true) {
1237 for(size_t i = 0; i < N; ++i) _bytes[i] = str[i];
1238 size_t pos = 0;
1239 while(pos < N - 1) {
1240 unsigned char b = static_cast<unsigned char>(str[pos]);
1241 if(b > 0x7F) _isAscii = false;
1242 char32_t cp;
1243 size_t seqLen;
1244 if(b < 0x80) { cp = b; seqLen = 1; }
1245 else if(b < 0xE0) { cp = b & 0x1F; seqLen = 2; }
1246 else if(b < 0xF0) { cp = b & 0x0F; seqLen = 3; }
1247 else { cp = b & 0x07; seqLen = 4; }
1248 for(size_t j = 1; j < seqLen && pos + j < N - 1; ++j)
1249 cp = (cp << 6) | (static_cast<unsigned char>(str[pos + j]) & 0x3F);
1250 _codepoints[_charCount++] = cp;
1251 pos += seqLen;
1252 }
1253 }
1254
1255 constexpr bool isAscii() const { return _isAscii; }
1256 constexpr size_t charCount() const { return _charCount; }
1257 constexpr size_t byteCount() const { return N - 1; }
1258 constexpr const char *bytes() const { return _bytes; }
1259 constexpr const char32_t *codepoints() const { return _codepoints; }
1260
1269 constexpr uint64_t hash() const {
1270 if(_isAscii) return fnv1aLatin1AsCodepoints(_bytes, N - 1);
1271 return fnv1aCodepoints(_codepoints, _charCount);
1272 }
1273
1274 private:
1275 char _bytes[N];
1276 char32_t _codepoints[N]; // worst case: N-1 codepoints
1277 size_t _charCount;
1278 bool _isAscii;
