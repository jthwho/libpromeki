948template<size_t N>
949class CompiledString {
950 public:
951 consteval CompiledString(const char (&str)[N])
952 : _bytes{}, _codepoints{}, _charCount(0), _isAscii(true) {
953 for(size_t i = 0; i < N; ++i) _bytes[i] = str[i];
954 size_t pos = 0;
955 while(pos < N - 1) {
956 unsigned char b = static_cast<unsigned char>(str[pos]);
957 if(b > 0x7F) _isAscii = false;
958 char32_t cp;
959 size_t seqLen;
960 if(b < 0x80) { cp = b; seqLen = 1; }
961 else if(b < 0xE0) { cp = b & 0x1F; seqLen = 2; }
962 else if(b < 0xF0) { cp = b & 0x0F; seqLen = 3; }
963 else { cp = b & 0x07; seqLen = 4; }
964 for(size_t j = 1; j < seqLen && pos + j < N - 1; ++j)
965 cp = (cp << 6) | (static_cast<unsigned char>(str[pos + j]) & 0x3F);
966 _codepoints[_charCount++] = cp;
967 pos += seqLen;
968 }
969 }
970
971 constexpr bool isAscii() const { return _isAscii; }
972 constexpr size_t charCount() const { return _charCount; }
973 constexpr size_t byteCount() const { return N - 1; }
974 constexpr const char *bytes() const { return _bytes; }
975 constexpr const char32_t *codepoints() const { return _codepoints; }
976
978 constexpr uint64_t hash() const {
979 if(_isAscii) return fnv1aData(_bytes, N - 1);
980 return fnv1aData(_codepoints, _charCount * sizeof(char32_t));
981 }
982
983 private:
984 char _bytes[N];
985 char32_t _codepoints[N]; // worst case: N-1 codepoints
986 size_t _charCount;
987 bool _isAscii;
988};
