795 friend bool operator<(const String &lhs, const String &rhs) {
796 if(lhs.d->isLatin1() == rhs.d->isLatin1()) {
797 // Same encoding: byte-level comparison is correct.
798 // - Latin1 vs Latin1: byte order == codepoint order.
799 // - Unicode vs Unicode: UTF-8 byte order == codepoint order.
800 return lhs.d->str() < rhs.d->str();
801 }
802 // Mixed encodings: walk codepoint-by-codepoint so that
803 // logically equal strings (e.g. Latin1 0xE9 and Unicode
804 // U+00E9) compare as equal under operator<.
805 const size_t llen = lhs.d->length();
806 const size_t rlen = rhs.d->length();
807 const size_t n = std::min(llen, rlen);
808 for(size_t i = 0; i < n; ++i) {
809 char32_t a = lhs.d->charAt(i).codepoint();
810 char32_t b = rhs.d->charAt(i).codepoint();
811 if(a != b) return a < b;
812 }
813 return llen < rlen;
814 }

1087 template <typename OutputType> OutputType to(Error *err = nullptr) const {
1088 const char *s = cstr();
1089 char *end = nullptr;
1090 OutputType ret{};
1091 if constexpr (std::is_integral_v<OutputType> && std::is_signed_v<OutputType>) {
1092 long long v = std::strtoll(s, &end, 10);
1093 ret = static_cast<OutputType>(v);
1094 } else if constexpr (std::is_integral_v<OutputType> && std::is_unsigned_v<OutputType>) {
1095 unsigned long long v = std::strtoull(s, &end, 10);
1096 ret = static_cast<OutputType>(v);
1097 } else if constexpr (std::is_floating_point_v<OutputType>) {
1098 double v = std::strtod(s, &end);
1099 ret = static_cast<OutputType>(v);
1100 } else {
1101 if(err != nullptr) *err = Error::Invalid;
1102 return ret;
1103 }
1104 if(end == s || *end != '\0') {
1105 if(err != nullptr) *err = Error::Invalid;
1106 return OutputType{};
1107 }
1108 if(err != nullptr) *err = Error::Ok;
1109 return ret;
1110 }

1293template<size_t N>
1294class CompiledString {
1295 public:
1296 consteval CompiledString(const char (&str)[N])
1297 : _bytes{}, _codepoints{}, _charCount(0), _isAscii(true) {
1298 for(size_t i = 0; i < N; ++i) _bytes[i] = str[i];
1299 size_t pos = 0;
1300 while(pos < N - 1) {
1301 unsigned char b = static_cast<unsigned char>(str[pos]);
1302 if(b > 0x7F) _isAscii = false;
1303 char32_t cp;
1304 size_t seqLen;
1305 if(b < 0x80) { cp = b; seqLen = 1; }
1306 else if(b < 0xE0) { cp = b & 0x1F; seqLen = 2; }
1307 else if(b < 0xF0) { cp = b & 0x0F; seqLen = 3; }
1308 else { cp = b & 0x07; seqLen = 4; }
1309 for(size_t j = 1; j < seqLen && pos + j < N - 1; ++j)
1310 cp = (cp << 6) | (static_cast<unsigned char>(str[pos + j]) & 0x3F);
1311 _codepoints[_charCount++] = cp;
1312 pos += seqLen;
1313 }
1314 }
1315
1316 constexpr bool isAscii() const { return _isAscii; }
1317 constexpr size_t charCount() const { return _charCount; }
1318 constexpr size_t byteCount() const { return N - 1; }
1319 constexpr const char *bytes() const { return _bytes; }
1320 constexpr const char32_t *codepoints() const { return _codepoints; }
1321
1330 constexpr uint64_t hash() const {
1331 if(_isAscii) return fnv1aLatin1AsCodepoints(_bytes, N - 1);
1332 return fnv1aCodepoints(_codepoints, _charCount);
1333 }
1334
1335 private:
1336 char _bytes[N];
1337 char32_t _codepoints[N]; // worst case: N-1 codepoints
1338 size_t _charCount;
1339 bool _isAscii;
1340};
