187 template <typename T>
188 static String bin(const T &val, int digits = 32, bool addPrefix = true) {
189 std::string ret;
190 if(addPrefix) ret += "0b";
191 T mask = static_cast<T>(1) << (digits - 1);
192 for(int i = 0; i < digits; i++) {
193 ret.push_back(val & mask ? '1' : '0');
194 mask >>= 1;
195 }
196 return ret;
197 }
