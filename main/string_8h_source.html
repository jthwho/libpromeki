824 friend bool operator<(const String &lhs, const String &rhs) {
825 if(lhs.d->isLatin1() == rhs.d->isLatin1()) {
826 // Same encoding: byte-level comparison is correct.
827 // - Latin1 vs Latin1: byte order == codepoint order.
828 // - Unicode vs Unicode: UTF-8 byte order == codepoint order.
829 return lhs.d->str() < rhs.d->str();
830 }
831 // Mixed encodings: walk codepoint-by-codepoint so that
832 // logically equal strings (e.g. Latin1 0xE9 and Unicode
833 // U+00E9) compare as equal under operator<.
834 const size_t llen = lhs.d->length();
835 const size_t rlen = rhs.d->length();
836 const size_t n = std::min(llen, rlen);
837 for(size_t i = 0; i < n; ++i) {
838 char32_t a = lhs.d->charAt(i).codepoint();
839 char32_t b = rhs.d->charAt(i).codepoint();
840 if(a != b) return a < b;
841 }
842 return llen < rlen;
843 }

1382template<size_t N>
1383class CompiledString {
1384 public:
1385 consteval CompiledString(const char (&str)[N])
1386 : _bytes{}, _codepoints{}, _charCount(0), _isAscii(true) {
1387 for(size_t i = 0; i < N; ++i) _bytes[i] = str[i];
1388 size_t pos = 0;
1389 while(pos < N - 1) {
1390 unsigned char b = static_cast<unsigned char>(str[pos]);
1391 if(b > 0x7F) _isAscii = false;
1392 char32_t cp;
1393 size_t seqLen;
1394 if(b < 0x80) { cp = b; seqLen = 1; }
1395 else if(b < 0xE0) { cp = b & 0x1F; seqLen = 2; }
1396 else if(b < 0xF0) { cp = b & 0x0F; seqLen = 3; }
1397 else { cp = b & 0x07; seqLen = 4; }
1398 for(size_t j = 1; j < seqLen && pos + j < N - 1; ++j)
1399 cp = (cp << 6) | (static_cast<unsigned char>(str[pos + j]) & 0x3F);
1400 _codepoints[_charCount++] = cp;
1401 pos += seqLen;
1402 }
1403 }
1404

1406 constexpr size_t charCount() const { return _charCount; }
1407 constexpr size_t byteCount() const { return N - 1; }
1408 constexpr const char *bytes() const { return _bytes; }
1409 constexpr const char32_t *codepoints() const { return _codepoints; }
1410
1419 constexpr uint64_t hash() const {
1420 if(_isAscii) return fnv1aLatin1AsCodepoints(_bytes, N - 1);
1421 return fnv1aCodepoints(_codepoints, _charCount);
1422 }
1423

1425 char _bytes[N];
