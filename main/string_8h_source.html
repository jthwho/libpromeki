763 friend bool operator<(const String &lhs, const String &rhs) {
764 if(lhs.d->isLatin1() == rhs.d->isLatin1()) {
765 // Same encoding: byte-level comparison is correct.
766 // - Latin1 vs Latin1: byte order == codepoint order.
767 // - Unicode vs Unicode: UTF-8 byte order == codepoint order.
768 return lhs.d->str() < rhs.d->str();
769 }
770 // Mixed encodings: walk codepoint-by-codepoint so that
771 // logically equal strings (e.g. Latin1 0xE9 and Unicode
772 // U+00E9) compare as equal under operator<.
773 const size_t llen = lhs.d->length();
774 const size_t rlen = rhs.d->length();
775 const size_t n = std::min(llen, rlen);
776 for(size_t i = 0; i < n; ++i) {
777 char32_t a = lhs.d->charAt(i).codepoint();
778 char32_t b = rhs.d->charAt(i).codepoint();
779 if(a != b) return a < b;
780 }
781 return llen < rlen;
782 }

810 String toUpper() const {
811 if(d->isLatin1()) {
812 const std::string &src = d->str();
813 std::string s;
814 s.resize(src.size());
815 for(size_t i = 0; i < src.size(); ++i) {
816 char32_t cp = Char(src[i]).toUpper().codepoint();
817 // Latin1 toUpper stays within the
818 // 0x00–0xFF range, so this is lossless.
819 s[i] = static_cast<char>(cp);
820 }
821 return String(std::move(s));
822 }
823 promeki::List<Char> chars;
824 chars.reserve(d->length());
825 for(size_t i = 0; i < d->length(); ++i)
826 chars.pushToBack(d->charAt(i).toUpper());
827 return String(new StringUnicodeData(std::move(chars)));
828 }

1055 template <typename OutputType> OutputType to(Error *err = nullptr) const {
1056 if constexpr (std::is_same_v<OutputType, bool>) {
1057 return toBool(err);
1058 } else if constexpr (std::is_integral_v<OutputType> && std::is_signed_v<OutputType>) {
1059 int base = 10;
1060 String cleaned = prepareIntParse(cstr(), &base);
1061 const char *s = cleaned.cstr();
1062 char *end = nullptr;
1063 errno = 0;
1064 long long v = std::strtoll(s, &end, base);
1065 if(end == s || *end != '\0') {
1066 if(err != nullptr) *err = Error::Invalid;
1067 return OutputType{};
1068 }
1069 if(errno == ERANGE) {
1070 if(err != nullptr) *err = Error::OutOfRange;
1071 return OutputType{};

1321template<size_t N>
1322class CompiledString {
1323 public:
1324 consteval CompiledString(const char (&str)[N])
1325 : _bytes{}, _codepoints{}, _charCount(0), _isAscii(true) {
1326 for(size_t i = 0; i < N; ++i) _bytes[i] = str[i];
1327 size_t pos = 0;
1328 while(pos < N - 1) {
1329 unsigned char b = static_cast<unsigned char>(str[pos]);
1330 if(b > 0x7F) _isAscii = false;
1331 char32_t cp;
1332 size_t seqLen;
1333 if(b < 0x80) { cp = b; seqLen = 1; }
1334 else if(b < 0xE0) { cp = b & 0x1F; seqLen = 2; }
1335 else if(b < 0xF0) { cp = b & 0x0F; seqLen = 3; }
1336 else { cp = b & 0x07; seqLen = 4; }
1337 for(size_t j = 1; j < seqLen && pos + j < N - 1; ++j)
1338 cp = (cp << 6) | (static_cast<unsigned char>(str[pos + j]) & 0x3F);
1339 _codepoints[_charCount++] = cp;
1340 pos += seqLen;
1341 }
1342 }
1343
1344 constexpr bool isAscii() const { return _isAscii; }
1345 constexpr size_t charCount() const { return _charCount; }
1346 constexpr size_t byteCount() const { return N - 1; }
1347 constexpr const char *bytes() const { return _bytes; }
1348 constexpr const char32_t *codepoints() const { return _codepoints; }
1349
1358 constexpr uint64_t hash() const {
1359 if(_isAscii) return fnv1aLatin1AsCodepoints(_bytes, N - 1);
1360 return fnv1aCodepoints(_codepoints, _charCount);
1361 }

1364 // structural type (C++20 [temp.param]/7) and can therefore
1365 // be used as a class-type non-type template parameter —
1366 // e.g. `template <CompiledString Name>` in StringRegistry
1367 // and VariantDatabase. Treat them as private.
1368 char _bytes[N];
1369 char32_t _codepoints[N]; // worst case: N-1 codepoints
1370 size_t _charCount;
1371 bool _isAscii;
