284inline void MediaIORequest::then(Callback cb) {
285 if (_hasOverride) {
286 // Sentinel path: fire immediately (marshal through
287 // the current loop when one is active so the
288 // callback observes the same thread semantics as
289 // the post-completion case).
290 EventLoop *loop = EventLoop::current();
291 Error err = _overrideError;
292 if (loop != nullptr) {
293 loop->postCallable([cb = std::move(cb), err]() mutable { cb(err); });
294 } else {
295 cb(err);
296 }
297 return;
298 }
299 if (_cmd.isNull()) return;
300 _cmd.modify()->setCompletionCallback(std::move(cb), EventLoop::current());
301}
