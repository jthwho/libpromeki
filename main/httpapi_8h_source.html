634template <CompiledString N>
635Error HttpApi::exposeDatabase(const String &mountPath, const String &title, VariantDatabase<N> &db, bool readOnly) {
636 using DB = VariantDatabase<N>;
637 using ID = typename DB::ID;
638
639 // mountPath is relative to this api's prefix. Resolve it
640 // before handing it to HttpServer (which doesn't know about
641 // the api's prefix) and to the catalog (which stores
642 // absolute paths).
643 const String absMount = resolve(mountPath);
644
645 // Hand the actual route registration off to HttpServer; the
646 // catalog publication is the only thing layered on top.
647 _server.exposeDatabase(absMount, db, readOnly);
648
649 // Collect declared keys for the {key} param description so the
650 // explorer's free-form input has a hint of valid values.
651 StringList knownKeys;
652 const auto specs = DB::registeredSpecs();
653 for (auto it = specs.cbegin(); it != specs.cend(); ++it) {
654 knownKeys.pushToBack(ID::fromId(it->first).name());
655 }
656
657 // GET <absMount> — full snapshot.
658 Endpoint epAll;
659 epAll.path = absMount;
660 epAll.method = HttpMethod::Get;
661 epAll.title = title + ": snapshot";
662 epAll.summary = String("Returns every key/value pair in ") + title + " as a single JSON object.";
663 epAll.tags = {title};
664 epAll.response = VariantSpec().setDescription("JSON object whose keys are the database IDs.");
665 if (Error err = addEndpointDescriptor(epAll); err.isError()) return err;
666
667 // GET <absMount>/_schema — registered specs.
668 Endpoint epSchema;
669 epSchema.path = absMount + "/_schema";
670 epSchema.method = HttpMethod::Get;
671 epSchema.title = title + ": schema";
672 epSchema.summary = "Returns the registered VariantSpec for every "
673 "declared key (type, default, range, description).";
674 epSchema.tags = {title};
675 epSchema.response = VariantSpec().setDescription("JSON object keyed by ID name; each value carries the spec.");
676 if (Error err = addEndpointDescriptor(epSchema); err.isError()) return err;
677
678 // GET <absMount>/{key} — single value.
679 Endpoint epGet;
680 epGet.path = absMount + "/{key}";
681 epGet.method = HttpMethod::Get;
682 epGet.title = title + ": get key";
683 epGet.summary = "Returns a single value plus its spec.";
684 epGet.tags = {title};
685 epGet.params = {Param{
686 .name = "key",
687 .in = ParamIn::Path,
688 .required = true,
689 .spec = keyParamSpec(knownKeys),
690 }};
691 epGet.response = VariantSpec().setDescription("JSON object with the value under its name and an "
692 "optional `_spec` companion.");
693 if (Error err = addEndpointDescriptor(epGet); err.isError()) return err;
694
695 if (readOnly) return Error::Ok;
696
697 // PUT <absMount>/{key} — set.
698 Endpoint epPut;
699 epPut.path = absMount + "/{key}";
700 epPut.method = HttpMethod::Put;
701 epPut.title = title + ": set key";
702 epPut.summary = "Updates a single value, validating it against the "
703 "registered spec.";
704 epPut.tags = {title};
705 epPut.params = {
706 Param{
707 .name = "key",
708 .in = ParamIn::Path,
709 .required = true,
710 .spec = keyParamSpec(knownKeys),
711 },
712 Param{
713 .name = "value",
714 .in = ParamIn::Body,
715 .required = true,
716 .spec = VariantSpec().setDescription("New value for the key (shape depends on spec)."),
717 },
718 };
719 epPut.response = VariantSpec().setDescription("JSON object with the stored value under its name.");
720 if (Error err = addEndpointDescriptor(epPut); err.isError()) return err;
721
722 // DELETE <absMount>/{key} — clear.
723 Endpoint epDel;
724 epDel.path = absMount + "/{key}";
725 epDel.method = HttpMethod::Delete;
726 epDel.title = title + ": delete key";
727 epDel.summary = "Clears the entry for a single key.";
728 epDel.tags = {title};
729 epDel.params = {Param{
730 .name = "key",
731 .in = ParamIn::Path,
732 .required = true,
733 .spec = keyParamSpec(knownKeys),
734 }};
735 return addEndpointDescriptor(epDel);
736}

738template <typename T> Error HttpApi::exposeLookup(const String &mountPath, const String &title, T &target) {
739 const String absMount = resolve(mountPath);
740 _server.exposeLookup(absMount, target);
741
742 Endpoint ep;
743 ep.path = absMount + "/{path}";
744 ep.method = HttpMethod::Get;
745 ep.title = title;
746 ep.summary = "Resolves a path-style key against the live object "
747 "tree (slashes mapped to dots; bare integer segments "
748 "to [N] index suffixes).";
749 ep.tags = {title};
750 ep.params = {Param{
751 .name = "path",
752 .in = ParamIn::Path,
753 .required = true,
754 .spec = VariantSpec().setType(Variant::TypeString).setDescription("Greedy lookup path."),
755 }};
756 ep.response = VariantSpec().setDescription("JSON object {\"value\": <variant>}.");
757 return addEndpointDescriptor(ep);
758}
