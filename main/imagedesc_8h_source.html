263inline DataStream &operator>>(DataStream &stream, ImageDesc &desc) {
264 if(!stream.readTag(DataStream::TypeImageDesc)) { desc = ImageDesc(); return stream; }
265 Size2Du32 size;
266 PixelDesc pd;
267 uint64_t linePad = 0, lineAlign = 1;
268 bool interlaced = false;
269 Metadata meta;
270 stream >> size >> pd >> linePad >> lineAlign >> interlaced >> meta;
271 if(stream.status() != DataStream::Ok) { desc = ImageDesc(); return stream; }
272 desc = ImageDesc(size, pd);
273 desc.setLinePad(static_cast<size_t>(linePad));
274 desc.setLineAlign(static_cast<size_t>(lineAlign));
275 desc.setInterlaced(interlaced);
276 desc.metadata() = std::move(meta);
277 return stream;
