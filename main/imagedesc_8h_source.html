404inline DataStream &operator>>(DataStream &stream, ImageDesc &desc) {
405 if(!stream.readTag(DataStream::TypeImageDesc)) { desc = ImageDesc(); return stream; }
406 Size2Du32 size;
407 PixelFormat pd;
408 uint64_t linePad = 0, lineAlign = 1;
409 uint32_t scanValue = 0;
410 Metadata meta;
411 stream >> size >> pd >> linePad >> lineAlign >> scanValue >> meta;
412 if(stream.status() != DataStream::Ok) { desc = ImageDesc(); return stream; }
413 desc = ImageDesc(size, pd);
414 desc.setLinePad(static_cast<size_t>(linePad));
415 desc.setLineAlign(static_cast<size_t>(lineAlign));
416 desc.setVideoScanMode(VideoScanMode{static_cast<int>(scanValue)});
417 desc.metadata() = std::move(meta);
418 return stream;
