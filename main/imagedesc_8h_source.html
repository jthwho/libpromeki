367inline DataStream &operator>>(DataStream &stream, ImageDesc &desc) {
368 if(!stream.readTag(DataStream::TypeImageDesc)) { desc = ImageDesc(); return stream; }
369 Size2Du32 size;
370 PixelDesc pd;
371 uint64_t linePad = 0, lineAlign = 1;
372 uint32_t interlaceValue = 0;
373 Metadata meta;
374 stream >> size >> pd >> linePad >> lineAlign >> interlaceValue >> meta;
375 if(stream.status() != DataStream::Ok) { desc = ImageDesc(); return stream; }
376 desc = ImageDesc(size, pd);
377 desc.setLinePad(static_cast<size_t>(linePad));
378 desc.setLineAlign(static_cast<size_t>(lineAlign));
379 desc.setInterlaceMode(InterlaceMode{static_cast<int>(interlaceValue)});
380 desc.metadata() = std::move(meta);
381 return stream;
382}
