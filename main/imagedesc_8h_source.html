360inline DataStream &operator>>(DataStream &stream, ImageDesc &desc) {
361 if(!stream.readTag(DataStream::TypeImageDesc)) { desc = ImageDesc(); return stream; }
362 Size2Du32 size;
363 PixelDesc pd;
364 uint64_t linePad = 0, lineAlign = 1;
365 bool interlaced = false;
366 Metadata meta;
367 stream >> size >> pd >> linePad >> lineAlign >> interlaced >> meta;
368 if(stream.status() != DataStream::Ok) { desc = ImageDesc(); return stream; }
369 desc = ImageDesc(size, pd);
370 desc.setLinePad(static_cast<size_t>(linePad));
371 desc.setLineAlign(static_cast<size_t>(lineAlign));
372 desc.setInterlaced(interlaced);
373 desc.metadata() = std::move(meta);
374 return stream;
