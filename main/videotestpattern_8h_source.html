329 // Generic image cache: a small fixed array of slots
330 // shared by all patterns. Slot meanings are local to
331 // each pattern's branch in create() — for example
332 // AvSync uses slot 0 for "white marker" and slot 1 for
333 // "black non-marker", while a static pattern uses slot
334 // 0 for its pre-rendered background. Patterns are
335 // mutually exclusive at any given time, so reusing the
336 // same slot index across patterns is fine.
337 //
338 // Any change to _pattern, _solidColor, or the
339 // requested ImageDesc dumps the entire cache; the next
340 // create() call rebuilds whatever it needs. Two
341 // renderSolid() calls (the most expensive thing the
342 // current patterns can do on a rebuild) are essentially
343 // free, so a simpler dump-on-change policy is the right
344 // tradeoff over per-key invalidation.
345 static constexpr int CacheSlotCount = 2;
346 mutable Image _cachedImages[CacheSlotCount];
347 mutable size_t _cacheW = 0;
348 mutable size_t _cacheH = 0;
349 mutable int _cachePixelDescId = 0;

379 const Image &cachedImage(int slotIndex,
380 const ImageDesc &desc,
381 Builder &&build) const {
382 if(_cacheW != desc.width()
383 || _cacheH != desc.height()
384 || _cachePixelDescId != static_cast<int>(desc.pixelDesc().id())) {
385 invalidateImageCache();
386 _cacheW = desc.width();
387 _cacheH = desc.height();
388 _cachePixelDescId =
389 static_cast<int>(desc.pixelDesc().id());
390 }
391 Image &slot = _cachedImages[slotIndex];
392 if(!slot.isValid()) {
393 slot = Image(desc);
394 if(slot.isValid()) build(slot);
395 }
396 return slot;
