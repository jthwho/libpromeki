347 // Generic image cache: a small fixed array of slots
348 // shared by all patterns. Slot meanings are local to
349 // each pattern's branch in create() — for example
350 // AvSync uses slot 0 for "white marker" and slot 1 for
351 // "black non-marker", while a static pattern uses slot
352 // 0 for its pre-rendered background. Patterns are
353 // mutually exclusive at any given time, so reusing the
354 // same slot index across patterns is fine.
355 //
356 // Any change to _pattern, _solidColor, or the
357 // requested ImageDesc dumps the entire cache; the next
358 // create() call rebuilds whatever it needs. Two
359 // renderSolid() calls (the most expensive thing the
360 // current patterns can do on a rebuild) are essentially
361 // free, so a simpler dump-on-change policy is the right
362 // tradeoff over per-key invalidation.
363 static constexpr int CacheSlotCount = 2;
364 mutable Image _cachedImages[CacheSlotCount];
365 mutable size_t _cacheW = 0;
366 mutable size_t _cacheH = 0;
367 mutable int _cachePixelDescId = 0;

397 const Image &cachedImage(int slotIndex,

399 Builder &&build) const {
400 if(_cacheW != desc.width()
401 || _cacheH != desc.height()
402 || _cachePixelDescId != static_cast<int>(desc.pixelDesc().id())) {
403 invalidateImageCache();
404 _cacheW = desc.width();
405 _cacheH = desc.height();
406 _cachePixelDescId =
407 static_cast<int>(desc.pixelDesc().id());
408 }
409 Image &slot = _cachedImages[slotIndex];
410 if(!slot.isValid()) {
411 slot = Image(desc);
412 if(slot.isValid()) build(slot);
