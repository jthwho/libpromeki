266 // Generic image cache: a small fixed array of slots
267 // shared by all patterns. Slot meanings are local to
268 // each pattern's branch in create() — for example
269 // AvSync uses slot 0 for "white marker" and slot 1 for
270 // "black non-marker", while a static pattern uses slot
271 // 0 for its pre-rendered background. Patterns are
272 // mutually exclusive at any given time, so reusing the
273 // same slot index across patterns is fine.
274 //
275 // Any change to _pattern, _solidColor, or the
276 // requested ImageDesc dumps the entire cache; the next
277 // create() call rebuilds whatever it needs. Two
278 // renderSolid() calls (the most expensive thing the
279 // current patterns can do on a rebuild) are essentially
280 // free, so a simpler dump-on-change policy is the right
281 // tradeoff over per-key invalidation.
282 static constexpr int CacheSlotCount = 2;
283 mutable SharedPtr<UncompressedVideoPayload, true, UncompressedVideoPayload>
284 _cachedPayloads[CacheSlotCount];

313 template <typename Builder>
314 SharedPtr<UncompressedVideoPayload, true, UncompressedVideoPayload>
315 cachedPayload(int slotIndex, const ImageDesc &desc, Builder &&build) const {
316 if (_cacheW != desc.width() || _cacheH != desc.height() ||
317 _cachePixelFormatId != static_cast<int>(desc.pixelFormat().id())) {
318 invalidatePayloadCache();
319 _cacheW = desc.width();
320 _cacheH = desc.height();
321 _cachePixelFormatId = static_cast<int>(desc.pixelFormat().id());
322 }
323 auto &slot = _cachedPayloads[slotIndex];
324 if (!slot.isValid()) {
325 slot = UncompressedVideoPayload::allocate(desc);
326 if (slot.isValid()) build(slot);
327 }
328 return slot;
329 }
