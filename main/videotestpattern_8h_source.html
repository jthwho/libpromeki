264 // Generic image cache: a small fixed array of slots
265 // shared by all patterns. Slot meanings are local to
266 // each pattern's branch in create() — for example
267 // AvSync uses slot 0 for "white marker" and slot 1 for
268 // "black non-marker", while a static pattern uses slot
269 // 0 for its pre-rendered background. Patterns are
270 // mutually exclusive at any given time, so reusing the
271 // same slot index across patterns is fine.
272 //
273 // Any change to _pattern, _solidColor, or the
274 // requested ImageDesc dumps the entire cache; the next
275 // create() call rebuilds whatever it needs. Two
276 // renderSolid() calls (the most expensive thing the
277 // current patterns can do on a rebuild) are essentially
278 // free, so a simpler dump-on-change policy is the right
279 // tradeoff over per-key invalidation.
280 static constexpr int CacheSlotCount = 2;

312 SharedPtr<UncompressedVideoPayload, true, UncompressedVideoPayload>
313 cachedPayload(int slotIndex,
314 const ImageDesc &desc,
315 Builder &&build) const {
316 if(_cacheW != desc.width()
317 || _cacheH != desc.height()
318 || _cachePixelFormatId != static_cast<int>(desc.pixelFormat().id())) {
319 invalidatePayloadCache();
320 _cacheW = desc.width();
321 _cacheH = desc.height();
322 _cachePixelFormatId =
323 static_cast<int>(desc.pixelFormat().id());
324 }
325 auto &slot = _cachedPayloads[slotIndex];
326 if(!slot.isValid()) {
327 slot = UncompressedVideoPayload::allocate(desc);
328 if(slot.isValid()) build(slot);
329 }
330 return slot;
331 }
