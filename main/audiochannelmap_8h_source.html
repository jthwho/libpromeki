366inline DataStream &operator>>(DataStream &stream, AudioChannelMap &map) {
367 if (!stream.readTag(DataStream::TypeAudioChannelMap)) {
368 map = AudioChannelMap();
369 return stream;
370 }
371 uint32_t count = 0;
372 stream >> count;
373 AudioChannelMap::EntryList entries;
374 entries.reserve(count);
375 for (uint32_t i = 0; i < count; ++i) {
376 String streamName;
377 int32_t roleValue = 0;
378 stream >> streamName >> roleValue;
379 AudioStreamDesc s = (streamName.isEmpty() || streamName == "Undefined") ? AudioStreamDesc()
380 : AudioStreamDesc(streamName);
381 entries.pushToBack(AudioChannelMap::Entry(s, ChannelRole(roleValue)));
382 }
383 if (stream.status() != DataStream::Ok) {
384 map = AudioChannelMap();
385 return stream;
386 }
387 map = AudioChannelMap(std::move(entries));
388 return stream;
389}
