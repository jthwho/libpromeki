294 constexpr auto parse(std::format_parse_context &ctx) {
295 auto it = ctx.begin();
296 auto end = ctx.end();
297
298 auto tryKeyword = [&](const char *kw, Style s) {
299 auto p = it;
300 while (*kw && p != end && *p == *kw) {
301 ++p;
302 ++kw;
303 }
304 if (*kw == 0 && (p == end || *p == '}' || *p == ':')) {
305 it = p;
306 _style = s;
307 return true;
308 }
309 return false;
310 };
311
312 if (!tryKeyword("scaled", Style::Scaled)) {
313 tryKeyword("hms", Style::Hms);
314 }
315
316 if (it != end && *it == ':') ++it;
317
318 ctx.advance_to(it);
319 return _base.parse(ctx);
320 }
